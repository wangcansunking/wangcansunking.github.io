第5课：PPT 生成 — AI 驱动的演示文稿创作

介绍

自动生成 PPT 是高频办公场景。本课构建从主题描述到完整演示文稿的 AI Pipeline，支持结构设计、内容生成、布局建议。AI 生成结构化 JSON，再用 pptxgenjs 渲染为真正的 .pptx 文件，可以直接用 PowerPoint/WPS 打开编辑。

遇到什么问题

PPT 生成需要结构化内容（大纲→章节→要点→图表建议），纯文本 LLM 输出需要转换为 pptx 格式：

  1. 结构化输出：LLM 需要生成 JSON 格式的 PPT 结构（不能直接是文本）
  2. 布局设计：不同页面有不同的布局需求（标题页、内容页、图表页）
  3. 字数控制：PPT 每页文字不能太多，LLM 默认会写太多内容
  4. 格式转换：JSON 结构需要转换为 pptxgenjs 的 API 调用

有哪些方案

方案 A：LLM 直接生成 JSON + pptxgenjs 渲染（推荐）

// LLM 输出结构化 JSON
const slides = await llm.chat([{ role: "user", content: "生成PPT大纲" }], {
  response_format: { type: "json_object" }
});
// pptxgenjs 渲染
const pptx = new PptxGenJS();
renderSlides(pptx, JSON.parse(slides.content));

✅ 完全可控，可任意定制样式 ✅ 无需 Python，纯 Node.js ❌ 需要设计 JSON Schema 和渲染逻辑

方案 B：python-pptx（Python 方案）

✅ 功能最完整，支持所有 PPT 特性 ❌ 需要 Python 环境，TypeScript 项目额外依赖

方案 C：调用 Gamma/Beautiful.AI API

✅ 生成效果最漂亮，有设计感 ❌ 第三方服务依赖，成本高，内容不可控

怎么选择

选择方案 A（LLM JSON + pptxgenjs）。纯 TypeScript 方案，工程集成最简洁。pptxgenjs 功能足够满足大多数需求（文字、图片、图表、形状），生成的 .pptx 文件可以用任何 Office 软件编辑。

怎么执行

AI PPT 生成流程

flowchart TD
    A[用户输入主题/要求] --> B[LLM 生成演讲大纲 JSON]
    B --> C{用户确认大纲?}
    C -->|调整| B
    C -->|确认| D[LLM 逐页生成内容 JSON]
    D --> E[LLM 生成配色方案建议]
    E --> F[pptxgenjs 渲染]
    F --> G[.pptx 文件输出]
    G --> H[用户下载/编辑]

PPT JSON Schema 设计

// src/ppt/types.ts

export interface PPTSlide {
  type: "title" | "content" | "two_column" | "image" | "chart" | "quote" | "section";
  title: string;
  subtitle?: string;       // 仅 title 类型
  bullets?: string[];      // 要点列表（每条不超过 20 字）
  leftColumn?: string[];   // two_column 左列
  rightColumn?: string[];  // two_column 右列
  quote?: string;          // quote 类型引用文字
  chartData?: {            // chart 类型数据
    type: "bar" | "line" | "pie";
    title: string;
    categories: string[];
    values: number[];
  };
  imageDescription?: string; // AI 生成配图描述（可选）
  speakerNotes?: string;     // 演讲者备注
}

export interface PPTPresentation {
  title: string;
  subtitle: string;
  author: string;
  date: string;
  theme: {
    primaryColor: string;    // 主色调（hex）
    secondaryColor: string;  // 辅色调
    accentColor: string;     // 强调色
    fontTitle: string;       // 标题字体
    fontBody: string;        // 正文字体
  };
  slides: PPTSlide[];
}

LLM 生成 PPT 内容

// src/ppt/generator.ts
import OpenAI from "openai";
import type { PPTPresentation, PPTSlide } from "./types";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// 步骤1：生成演讲大纲
export async function generateOutline(
  topic: string,
  requirements?: string
): Promise<string[]> {
  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      {
        role: "system",
        content: "你是一个专业的演讲设计师，擅长设计结构清晰、逻辑严密的演示文稿大纲。",
      },
      {
        role: "user",
        content: `请为以下主题设计 PPT 大纲（10-15 页）：
主题：${topic}
${requirements ? `额外要求：${requirements}` : ""}

以 JSON 数组形式返回章节标题，例如：
["封面", "目录", "背景介绍", "核心问题", "解决方案1", "解决方案2", "案例分析", "数据支撑", "总结与展望", "Q&A"]`,
      },
    ],
    response_format: { type: "json_object" },
    temperature: 0.7,
  });

  const data = JSON.parse(response.choices[0].message.content!);
  return data.slides || data.outline || data.chapters || [];
}

// 步骤2：生成完整 PPT 内容
export async function generatePresentation(
  topic: string,
  audience: string = "专业人士",
  slideCount: number = 12
): Promise<PPTPresentation> {
  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      {
        role: "system",
        content: `你是一个 PPT 内容设计专家。生成的内容必须：
- 每页要点不超过 5 条
- 每条要点不超过 20 个字
- 逻辑清晰，递进有序
- 适合 ${audience} 受众`,
      },
      {
        role: "user",
        content: `请为主题"${topic}"生成完整的 PPT 内容，共 ${slideCount} 页。

严格按以下 JSON 格式输出：
{
  "title": "演示文稿标题",
  "subtitle": "副标题",
  "author": "作者",
  "date": "2025年",
  "theme": {
    "primaryColor": "#2B5CE6",
    "secondaryColor": "#E8EEF8",
    "accentColor": "#FF6B35",
    "fontTitle": "Microsoft YaHei",
    "fontBody": "Microsoft YaHei"
  },
  "slides": [
    {
      "type": "title",
      "title": "主标题",
      "subtitle": "副标题"
    },
    {
      "type": "content",
      "title": "页面标题",
      "bullets": ["要点1（不超过20字）", "要点2", "要点3"],
      "speakerNotes": "演讲者备注"
    }
  ]
}`,
      },
    ],
    response_format: { type: "json_object" },
    temperature: 0.6,
    max_tokens: 4000,
  });

  return JSON.parse(response.choices[0].message.content!) as PPTPresentation;
}

// 配色方案建议
export async function suggestColorScheme(
  topic: string,
  mood: "professional" | "creative" | "energetic" | "calm"
): Promise<PPTPresentation["theme"]> {
  const schemes = {
    professional: { primaryColor: "#1B3A8F", secondaryColor: "#EEF2FF", accentColor: "#FF6B35", fontTitle: "Calibri", fontBody: "Calibri" },
    creative:     { primaryColor: "#7C3AED", secondaryColor: "#F5F0FF", accentColor: "#F59E0B", fontTitle: "Poppins", fontBody: "Inter" },
    energetic:    { primaryColor: "#DC2626", secondaryColor: "#FEF2F2", accentColor: "#FBBF24", fontTitle: "Oswald", fontBody: "Open Sans" },
    calm:         { primaryColor: "#059669", secondaryColor: "#F0FDF4", accentColor: "#3B82F6", fontTitle: "Georgia", fontBody: "Roboto" },
  };
  return schemes[mood];
}

pptxgenjs 渲染

// src/ppt/renderer.ts
import PptxGenJS from "pptxgenjs";
import type { PPTPresentation, PPTSlide } from "./types";

export async function renderPresentation(
  presentation: PPTPresentation,
  outputPath: string
): Promise<void> {
  const pptx = new PptxGenJS();

  // 全局设置
  pptx.author = presentation.author;
  pptx.title = presentation.title;
  pptx.layout = "LAYOUT_WIDE"; // 16:9

  const theme = presentation.theme;

  for (const slideData of presentation.slides) {
    const slide = pptx.addSlide();

    // 背景
    slide.background = { color: "FFFFFF" };

    // 顶部装饰条
    slide.addShape(pptx.ShapeType.rect, {
      x: 0, y: 0, w: "100%", h: 0.08,
      fill: { color: theme.primaryColor.replace("#", "") },
    });

    switch (slideData.type) {
      case "title":
        renderTitleSlide(slide, pptx, slideData, theme);
        break;
      case "content":
        renderContentSlide(slide, pptx, slideData, theme);
        break;
      case "two_column":
        renderTwoColumnSlide(slide, pptx, slideData, theme);
        break;
      case "chart":
        renderChartSlide(slide, pptx, slideData, theme);
        break;
      case "quote":
        renderQuoteSlide(slide, pptx, slideData, theme);
        break;
      case "section":
        renderSectionSlide(slide, pptx, slideData, theme);
        break;
      default:
        renderContentSlide(slide, pptx, slideData, theme);
    }

    // 演讲者备注
    if (slideData.speakerNotes) {
      slide.addNotes(slideData.speakerNotes);
    }
  }

  await pptx.writeFile({ fileName: outputPath });
  console.log(`PPT 已生成: ${outputPath}`);
}

function renderTitleSlide(
  slide: PptxGenJS.Slide,
  pptx: PptxGenJS,
  data: PPTSlide,
  theme: PPTPresentation["theme"]
): void {
  // 标题
  slide.addText(data.title, {
    x: 1, y: 2.5, w: 8, h: 1.2,
    fontSize: 36,
    bold: true,
    color: theme.primaryColor.replace("#", ""),
    fontFace: theme.fontTitle,
    align: "center",
  });

  // 副标题
  if (data.subtitle) {
    slide.addText(data.subtitle, {
      x: 1, y: 3.8, w: 8, h: 0.6,
      fontSize: 18,
      color: "666666",
      fontFace: theme.fontBody,
      align: "center",
    });
  }

  // 装饰线
  slide.addShape(pptx.ShapeType.rect, {
    x: 3.5, y: 3.5, w: 3, h: 0.05,
    fill: { color: theme.accentColor.replace("#", "") },
  });
}

function renderContentSlide(
  slide: PptxGenJS.Slide,
  _pptx: PptxGenJS,
  data: PPTSlide,
  theme: PPTPresentation["theme"]
): void {
  // 标题
  slide.addText(data.title, {
    x: 0.5, y: 0.3, w: 9, h: 0.8,
    fontSize: 24,
    bold: true,
    color: theme.primaryColor.replace("#", ""),
    fontFace: theme.fontTitle,
  });

  // 要点列表
  if (data.bullets?.length) {
    const bulletText = data.bullets.map((b) => ({ text: b, options: { bullet: true } }));
    slide.addText(bulletText as any, {
      x: 0.5, y: 1.3, w: 9, h: 5,
      fontSize: 18,
      color: "333333",
      fontFace: theme.fontBody,
      lineSpacingMultiple: 1.5,
    });
  }
}

function renderTwoColumnSlide(
  slide: PptxGenJS.Slide,
  _pptx: PptxGenJS,
  data: PPTSlide,
  theme: PPTPresentation["theme"]
): void {
  slide.addText(data.title, {
    x: 0.5, y: 0.3, w: 9, h: 0.8,
    fontSize: 24, bold: true,
    color: theme.primaryColor.replace("#", ""),
    fontFace: theme.fontTitle,
  });

  const colStyle = { fontSize: 16, color: "333333", fontFace: theme.fontBody };

  if (data.leftColumn) {
    slide.addText(data.leftColumn.map((t) => ({ text: t, options: { bullet: true } })) as any, {
      x: 0.5, y: 1.3, w: 4.2, h: 5, ...colStyle,
    });
  }

  if (data.rightColumn) {
    slide.addText(data.rightColumn.map((t) => ({ text: t, options: { bullet: true } })) as any, {
      x: 5.3, y: 1.3, w: 4.2, h: 5, ...colStyle,
    });
  }
}

function renderChartSlide(
  slide: PptxGenJS.Slide,
  pptx: PptxGenJS,
  data: PPTSlide,
  theme: PPTPresentation["theme"]
): void {
  slide.addText(data.title, {
    x: 0.5, y: 0.3, w: 9, h: 0.8,
    fontSize: 24, bold: true,
    color: theme.primaryColor.replace("#", ""),
  });

  if (data.chartData) {
    const chartType =
      data.chartData.type === "bar" ? pptx.ChartType.bar :
      data.chartData.type === "line" ? pptx.ChartType.line :
      pptx.ChartType.pie;

    slide.addChart(chartType, [
      {
        name: data.chartData.title,
        labels: data.chartData.categories,
        values: data.chartData.values,
      },
    ], { x: 1, y: 1.2, w: 8, h: 5 });
  }
}

function renderQuoteSlide(
  slide: PptxGenJS.Slide,
  _pptx: PptxGenJS,
  data: PPTSlide,
  theme: PPTPresentation["theme"]
): void {
  slide.addText(`"${data.quote || data.title}"`, {
    x: 1, y: 2, w: 8, h: 3,
    fontSize: 28,
    italic: true,
    color: theme.primaryColor.replace("#", ""),
    align: "center",
    fontFace: theme.fontTitle,
  });
}

function renderSectionSlide(
  slide: PptxGenJS.Slide,
  _pptx: PptxGenJS,
  data: PPTSlide,
  theme: PPTPresentation["theme"]
): void {
  slide.background = { color: theme.primaryColor.replace("#", "") };
  slide.addText(data.title, {
    x: 1, y: 2.5, w: 8, h: 1.5,
    fontSize: 36, bold: true, color: "FFFFFF",
    align: "center", fontFace: theme.fontTitle,
  });
}

完整使用示例

// src/ppt/demo.ts
import { generatePresentation } from "./generator";
import { renderPresentation } from "./renderer";

async function createPPT(topic: string, outputPath: string) {
  console.log(`生成 PPT：${topic}`);

  // 1. AI 生成内容
  console.log("步骤1：AI 生成内容...");
  const presentation = await generatePresentation(topic, "企业管理层", 12);
  console.log(`大纲：${presentation.slides.map((s) => s.title).join(" → ")}`);

  // 2. 渲染为 pptx
  console.log("步骤2：渲染 PPT...");
  await renderPresentation(presentation, outputPath);

  console.log(`完成！共 ${presentation.slides.length} 页`);
  return outputPath;
}

// 运行
createPPT(
  "2025年 AI 技术趋势与企业数字化转型机遇",
  "./output/ai_trends_2025.pptx"
).catch(console.error);
