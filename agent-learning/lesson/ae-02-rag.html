第2课：RAG 基础 — 给 Agent 接上知识库

介绍

RAG（检索增强生成）让 Agent 能利用超出训练数据的外部知识。核心是：把知识分块→向量化→存储，然后按需检索注入 context。LLM 只有训练截止日期前的知识，无法回答最新信息或私有数据的问题，RAG 正是解决这个核心痛点的标准方案。

遇到什么问题

LLM 只有训练截止日期前的知识，无法回答最新信息或私有数据的问题：

flowchart LR
    A[文档] --> B[分块 Chunking]
    B --> C[Embedding 向量化]
    C --> D[向量数据库]
    E[用户查询] --> F[查询 Embedding]
    F --> G[相似度检索\nTop-K]
    D --> G
    G --> H[注入 Prompt]
    H --> I[LLM 回答]
    style D fill:#e8f4f8

有哪些方案

方案 A：简单向量检索（cosine similarity）

const results = vectorStore.search(queryEmbedding, topK=5);

✅ 语义理解强，能找到同义表达 ❌ 对精确关键词查询效果差（如产品编号 "SKU-2024-XL"）

方案 B：BM25 全文检索（keyword matching）

const results = bm25Index.search(query, topK=5);

✅ 精确关键词匹配，速度快 ❌ 不理解语义，"猫" 查不到 "猫咪"

方案 C：混合检索（Dense + Sparse，互补）

const denseResults = vectorStore.search(queryEmbedding, topK);
const sparseResults = bm25Index.search(query, topK);
const merged = rrfFusion(denseResults, sparseResults);

✅ 语义+关键词双重保障，召回率最高 ❌ 实现稍复杂（需要 RRF 融合）

怎么选择

基础 RAG 用方案 A（向量检索），快速搭建可用的系统。混合检索（方案 C）放在下一课讲解。本课专注向量检索的完整实现：分块、Embedding、内存向量库、Top-K 检索、prompt 注入。

怎么执行

文本分块策略

// src/rag/chunking.ts

export interface Chunk {
  text: string;
  index: number;
  metadata?: Record<string, any>;
}

// 固定长度分块（有重叠，保持上下文连贯性）
export function fixedSizeChunking(
  text: string,
  chunkSize: number = 512,
  overlap: number = 64
): Chunk[] {
  const chunks: Chunk[] = [];
  let start = 0;

  while (start < text.length) {
    const end = Math.min(start + chunkSize, text.length);
    chunks.push({ text: text.slice(start, end), index: chunks.length });
    start += chunkSize - overlap;
  }
  return chunks;
}

// 按段落分块（保留自然语义边界）
export function paragraphChunking(
  text: string,
  maxChunkSize: number = 800
): Chunk[] {
  const paragraphs = text.split(/\n{2,}/).filter((p) => p.trim().length > 0);
  const chunks: Chunk[] = [];
  let current = "";

  for (const para of paragraphs) {
    if (current.length + para.length > maxChunkSize && current.length > 0) {
      chunks.push({ text: current.trim(), index: chunks.length });
      current = "";
    }
    current += (current ? "\n\n" : "") + para;
  }

  if (current.trim()) chunks.push({ text: current.trim(), index: chunks.length });
  return chunks;
}

// 递归分块（优先按大边界，递归到小边界）
export function recursiveChunking(
  text: string,
  maxSize: number = 600,
  separators: string[] = ["\n\n", "\n", "。", ".", " "]
): Chunk[] {
  function split(text: string, seps: string[]): string[] {
    if (text.length <= maxSize) return [text];
    if (seps.length === 0) return [text.slice(0, maxSize)];

    const parts = text.split(seps[0]);
    const result: string[] = [];
    let current = "";

    for (const part of parts) {
      if ((current + seps[0] + part).length <= maxSize) {
        current = current ? current + seps[0] + part : part;
      } else {
        if (current) result.push(current);
        if (part.length > maxSize) {
          result.push(...split(part, seps.slice(1)));
          current = "";
        } else {
          current = part;
        }
      }
    }
    if (current) result.push(current);
    return result;
  }

  return split(text, separators).map((text, index) => ({ text, index }));
}

OpenAI Embeddings API

// src/rag/embeddings.ts
import OpenAI from "openai";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export async function getEmbedding(
  text: string,
  model: "text-embedding-3-small" | "text-embedding-3-large" = "text-embedding-3-small"
): Promise<number[]> {
  const response = await openai.embeddings.create({ model, input: text });
  return response.data[0].embedding;
}

// 批量 Embedding（减少 API 调用次数）
export async function batchEmbeddings(
  texts: string[],
  batchSize: number = 100
): Promise<number[][]> {
  const results: number[][] = [];

  for (let i = 0; i < texts.length; i += batchSize) {
    const batch = texts.slice(i, i + batchSize);
    const response = await openai.embeddings.create({
      model: "text-embedding-3-small",
      input: batch,
    });
    results.push(...response.data.map((d) => d.embedding));
    console.log(`Embedding: ${Math.min(i + batchSize, texts.length)}/${texts.length}`);
  }

  return results;
}

向量数据库（内存实现演示）

// src/rag/vector_store.ts
import * as fs from "fs";

export interface VectorRecord {
  id: string;
  text: string;
  embedding: number[];
  metadata?: Record<string, any>;
}

export class SimpleVectorStore {
  private records: VectorRecord[] = [];

  add(record: VectorRecord): void {
    this.records.push(record);
  }

  // Top-K 余弦相似度检索
  search(
    queryEmbedding: number[],
    topK: number = 5,
    minScore: number = 0.5
  ): { record: VectorRecord; score: number }[] {
    return this.records
      .map((record) => ({
        record,
        score: this.cosineSimilarity(queryEmbedding, record.embedding),
      }))
      .filter((r) => r.score >= minScore)
      .sort((a, b) => b.score - a.score)
      .slice(0, topK);
  }

  private cosineSimilarity(a: number[], b: number[]): number {
    let dot = 0, magA = 0, magB = 0;
    for (let i = 0; i < a.length; i++) {
      dot += a[i] * b[i];
      magA += a[i] * a[i];
      magB += b[i] * b[i];
    }
    return dot / (Math.sqrt(magA) * Math.sqrt(magB));
  }

  save(path: string): void {
    fs.writeFileSync(path, JSON.stringify(this.records));
  }

  load(path: string): void {
    if (fs.existsSync(path)) {
      this.records = JSON.parse(fs.readFileSync(path, "utf-8"));
      console.log(`加载 ${this.records.length} 条向量记录`);
    }
  }

  get size(): number { return this.records.length; }
}

检索（Top-K）与 Prompt 注入

// src/rag/rag_pipeline.ts
import OpenAI from "openai";
import { recursiveChunking } from "./chunking";
import { batchEmbeddings, getEmbedding } from "./embeddings";
import { SimpleVectorStore } from "./vector_store";
import * as fs from "fs";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export class RAGPipeline {
  private vectorStore: SimpleVectorStore;

  constructor(private storePath?: string) {
    this.vectorStore = new SimpleVectorStore();
    if (storePath) this.vectorStore.load(storePath);
  }

  async indexDocument(text: string, metadata?: Record<string, any>): Promise<number> {
    const chunks = recursiveChunking(text, 600);
    const embeddings = await batchEmbeddings(chunks.map((c) => c.text));

    for (let i = 0; i < chunks.length; i++) {
      this.vectorStore.add({
        id: `chunk_${Date.now()}_${i}`,
        text: chunks[i].text,
        embedding: embeddings[i],
        metadata: { ...metadata, chunkIndex: i },
      });
    }

    if (this.storePath) this.vectorStore.save(this.storePath);
    console.log(`索引完成: ${chunks.length} 个块，共 ${this.vectorStore.size} 条记录`);
    return chunks.length;
  }

  async query(question: string, topK: number = 5): Promise<{
    answer: string;
    sources: string[];
    scores: number[];
  }> {
    // 1. 查询 Embedding
    const questionEmbedding = await getEmbedding(question);

    // 2. Top-K 检索
    const results = this.vectorStore.search(questionEmbedding, topK, 0.6);

    if (results.length === 0) {
      return { answer: "知识库中没有找到相关信息。", sources: [], scores: [] };
    }

    console.log(`检索到 ${results.length} 条 (最高相似度: ${results[0].score.toFixed(3)})`);

    // 3. Prompt 注入（把检索结果插入 system prompt）
    const context = results
      .map((r, i) => `[来源 ${i + 1}] ${r.record.text}`)
      .join("\n\n");

    const response = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        {
          role: "system",
          content: `你是一个知识库助手。严格基于以下上下文回答，不要编造：

${context}

如果上下文中没有答案，说"根据知识库无法回答此问题"。`,
        },
        { role: "user", content: question },
      ],
      temperature: 0.1,
    });

    return {
      answer: response.choices[0].message.content || "",
      sources: results.map((r) => r.record.text.slice(0, 100) + "..."),
      scores: results.map((r) => r.score),
    };
  }
}

完整使用示例

// src/rag/demo.ts
import { RAGPipeline } from "./rag_pipeline";
import * as fs from "fs";

async function runDemo() {
  const rag = new RAGPipeline("./rag_store.json");

  // 构建知识库
  console.log("=== 索引文档 ===");
  await rag.indexDocument(
    fs.readFileSync("./docs/company_guide.txt", "utf-8"),
    { source: "公司指南" }
  );

  // RAG 问答
  console.log("\n=== RAG 问答 ===");
  const questions = [
    "公司的差旅报销标准是什么？",
    "如何申请年假？",
  ];

  for (const q of questions) {
    console.log(`\n问：${q}`);
    const { answer, sources } = await rag.query(q);
    console.log(`答：${answer}`);
    console.log(`引用：${sources[0]}`);
  }
}

runDemo().catch(console.error);
