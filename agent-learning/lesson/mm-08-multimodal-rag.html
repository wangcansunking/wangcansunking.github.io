第8课：多模态 RAG — 图文混合知识库

介绍

传统 RAG 只处理文本。多模态 RAG 可以检索图像、表格、图表，让知识库真正理解非文本内容。本课构建图文混合知识库：图像先用 Vision API 生成文字描述再 embedding，检索时同时搜索文本和图像描述，检索结果注入 context 供 LLM 回答。

遇到什么问题

图像无法直接向量化检索，需要先"描述"再存储；检索时需要同时搜索文本和图像：

  1. 图像无法直接 embedding：文本 embedding 模型（如 text-embedding-3-small）无法处理图像
  2. 图像描述的信息损耗：Vision API 生成的描述可能遗漏关键细节，需要精心设计 prompt
  3. 混合检索：文本块和图像描述的 embedding 在同一向量空间，但检索时需要标记来源类型
  4. Context 注入：检索到图像时，需要把原始图像（而非描述文字）注入 Vision 模型

有哪些方案

方案 A：图像→文字描述→Embedding（推荐）

图像 → Vision API → 文字描述 → Embedding → 存向量库（含图像路径）
检索时：返回描述 + 原始图像路径 → 注入 Vision 模型

✅ 实现简单，不需要多模态 embedding 模型 ✅ 文字描述可以编辑和增强（加标签、关键词） ❌ 描述可能遗漏视觉细节

方案 B：多模态 Embedding 模型（如 CLIP）

直接把图像转为向量，与文本向量共享同一向量空间。

✅ 保留完整图像信息，跨模态检索准确度更高 ❌ 需要部署 CLIP 或调用专用 API，成本和复杂度高

方案 C：独立图像索引，并行检索

图像和文本分别建索引，检索时并行查询再合并排序。

✅ 不同模态可以用各自最优的检索方式 ❌ 需要维护多个索引，RRF 融合逻辑复杂

怎么选择

选择方案 A（图像→描述→Embedding）。工程实现最简单，与现有文本 RAG 无缝融合，成本可控。Vision API 生成的详细描述质量已经很高，足以支撑大多数检索场景。

图像 Embedding（Vision 生成描述）

// src/multimodal_rag/image_indexer.ts
import OpenAI from "openai";
import * as fs from "fs";
import * as path from "path";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export interface ImageRecord {
  id: string;
  type: "image";
  filePath: string;
  description: string;
  embedding: number[];
  metadata: {
    fileName: string;
    fileSize: number;
    createdAt: string;
    tags?: string[];
  };
}

export async function describeImage(imagePath: string): Promise<string> {
  const buffer = fs.readFileSync(imagePath);
  const base64 = buffer.toString("base64");
  const ext = path.extname(imagePath).toLowerCase();
  const mimeType = ext === ".png" ? "image/png" : "image/jpeg";

  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [{
      role: "user",
      content: [
        { type: "image_url", image_url: { url: `data:${mimeType};base64,${base64}`, detail: "high" } },
        {
          type: "text",
          text: `请为这张图片生成详细的索引描述（用于知识库检索），包含：
1. 图片主要内容和主题
2. 如果是图表：所有数据点、数值、趋势、标题、图例
3. 如果是文档截图：所有可见文字
4. 如果是产品/实物：品牌、型号、特征、颜色
5. 关键词列表（5-10个）
用中文描述，约 200-300 字。`,
        },
      ],
    }],
    max_tokens: 500,
  });

  return response.choices[0].message.content || "";
}

export async function getEmbedding(text: string): Promise<number[]> {
  const response = await openai.embeddings.create({
    model: "text-embedding-3-small",
    input: text,
  });
  return response.data[0].embedding;
}

export async function indexImage(imagePath: string, tags?: string[]): Promise<ImageRecord> {
  console.log(`索引: ${path.basename(imagePath)}`);
  const description = await describeImage(imagePath);
  const embedding = await getEmbedding(description);

  return {
    id: `img_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`,
    type: "image",
    filePath: imagePath,
    description,
    embedding,
    metadata: {
      fileName: path.basename(imagePath),
      fileSize: fs.statSync(imagePath).size,
      createdAt: new Date().toISOString(),
      tags,
    },
  };
}

export async function indexImageDirectory(dirPath: string): Promise<ImageRecord[]> {
  const exts = [".jpg", ".jpeg", ".png", ".webp"];
  const files = fs.readdirSync(dirPath).filter((f) =>
    exts.includes(path.extname(f).toLowerCase())
  );

  console.log(`发现 ${files.length} 张图像`);
  const records: ImageRecord[] = [];

  for (let i = 0; i < files.length; i++) {
    const record = await indexImage(path.join(dirPath, files[i]));
    records.push(record);
    if (i < files.length - 1) await new Promise((r) => setTimeout(r, 500));
  }

  return records;
}

多模态向量库（图文混存）

// src/multimodal_rag/vector_store.ts
import * as fs from "fs";
import type { ImageRecord } from "./image_indexer";

export interface TextRecord {
  id: string;
  type: "text";
  text: string;
  embedding: number[];
  metadata: { source: string; page?: number };
}

export type VectorRecord = ImageRecord | TextRecord;

export class MultimodalVectorStore {
  private records: VectorRecord[] = [];
  private readonly storePath: string;

  constructor(storePath: string = "./multimodal_store.json") {
    this.storePath = storePath;
    if (fs.existsSync(storePath)) {
      this.records = JSON.parse(fs.readFileSync(storePath, "utf-8"));
      console.log(`加载 ${this.records.length} 条向量记录`);
    }
  }

  add(record: VectorRecord): void {
    this.records.push(record);
  }

  search(
    queryEmbedding: number[],
    topK: number = 5,
    typeFilter?: "text" | "image"
  ): { record: VectorRecord; score: number }[] {
    const candidates = typeFilter
      ? this.records.filter((r) => r.type === typeFilter)
      : this.records;

    return candidates
      .map((record) => ({ record, score: this.cosineSimilarity(queryEmbedding, record.embedding) }))
      .sort((a, b) => b.score - a.score)
      .slice(0, topK);
  }

  private cosineSimilarity(a: number[], b: number[]): number {
    let dot = 0, magA = 0, magB = 0;
    for (let i = 0; i < a.length; i++) {
      dot += a[i] * b[i];
      magA += a[i] * a[i];
      magB += b[i] * b[i];
    }
    return dot / (Math.sqrt(magA) * Math.sqrt(magB));
  }

  save(): void {
    fs.writeFileSync(this.storePath, JSON.stringify(this.records, null, 2));
    console.log(`向量库保存: ${this.records.length} 条`);
  }

  getStats() {
    const images = this.records.filter((r) => r.type === "image").length;
    return { total: this.records.length, images, texts: this.records.length - images };
  }
}

混合检索与 Context 注入

// src/multimodal_rag/retriever.ts
import OpenAI from "openai";
import * as fs from "fs";
import { getEmbedding } from "./image_indexer";
import type { MultimodalVectorStore } from "./vector_store";
import type { ImageRecord, TextRecord } from "./image_indexer";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export async function multimodalRAGQuery(
  question: string,
  store: MultimodalVectorStore,
  topK: number = 5
): Promise<string> {
  // 1. 问题 Embedding
  const questionEmbedding = await getEmbedding(question);

  // 2. 混合检索
  const results = store.search(questionEmbedding, topK);

  const imageResults = results.filter((r) => r.record.type === "image") as { record: ImageRecord; score: number }[];
  const textResults = results.filter((r) => r.record.type === "text") as { record: TextRecord; score: number }[];

  console.log(`[RAG] 检索结果: ${imageResults.length} 张图像, ${textResults.length} 段文本`);

  if (imageResults.length > 0) {
    // 3a. 有图像：用 Vision 模型，传入原始图像
    const userContent: OpenAI.ChatCompletionContentPart[] = [];

    for (const { record } of imageResults) {
      try {
        const buffer = fs.readFileSync(record.filePath);
        const mimeType = record.filePath.endsWith(".png") ? "image/png" : "image/jpeg";
        userContent.push({
          type: "image_url",
          image_url: { url: `data:${mimeType};base64,${buffer.toString("base64")}` },
        });
      } catch {
        // 原始文件不存在，退回描述文字
        userContent.push({ type: "text", text: `[图像: ${record.metadata.fileName}]\n${record.description}` });
      }
    }

    if (textResults.length > 0) {
      const textContext = textResults.map((r) => r.record.text).join("\n\n---\n\n");
      userContent.push({ type: "text", text: `参考文档：\n${textContext}` });
    }

    userContent.push({ type: "text", text: `问题：${question}` });

    const response = await openai.chat.completions.create({
      model: "gpt-4o",
      messages: [
        { role: "system", content: "基于提供的图像和文档内容回答问题，引用具体来源。" },
        { role: "user", content: userContent },
      ],
      max_tokens: 1500,
    });

    return response.choices[0].message.content || "";
  } else {
    // 3b. 纯文本检索
    const context = textResults.map((r) => r.record.text).join("\n\n---\n\n");

    const response = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        { role: "system", content: "基于提供的上下文回答问题。" },
        { role: "user", content: `上下文：\n${context}\n\n问题：${question}` },
      ],
    });

    return response.choices[0].message.content || "";
  }
}

完整演示

// src/multimodal_rag/demo.ts
import { indexImageDirectory, getEmbedding } from "./image_indexer";
import { MultimodalVectorStore } from "./vector_store";
import { multimodalRAGQuery } from "./retriever";

async function buildKnowledgeBase() {
  const store = new MultimodalVectorStore("./kb.json");

  // 索引图像目录
  const imageRecords = await indexImageDirectory("./kb_images");
  for (const record of imageRecords) store.add(record);

  // 添加文本记录
  const textChunks = [
    "产品A的Q3销售额增长35%，主要驱动来自华南市场扩张和新品发布。",
    "2025年全球AI市场规模预计达到5000亿美元，年复合增长率38%。",
  ];

  for (const chunk of textChunks) {
    const embedding = await getEmbedding(chunk);
    store.add({
      id: `text_${Date.now()}`,
      type: "text",
      text: chunk,
      embedding,
      metadata: { source: "manual" },
    });
  }

  store.save();
  console.log("知识库统计:", store.getStats());
  return store;
}

async function runDemo() {
  const store = await buildKnowledgeBase();

  const questions = [
    "销售增长最快的区域是哪里？",
    "图表中显示了什么数据趋势？",
    "AI市场的规模预测是多少？",
  ];

  for (const q of questions) {
    console.log(`\n问题: ${q}`);
    const answer = await multimodalRAGQuery(q, store);
    console.log(`回答: ${answer}`);
  }
}

runDemo().catch(console.error);
