第3课：视频生成 — Wan2.1 / HunyuanVideo / Kling / Runway

介绍

视频生成是 2025 年 AI 最热门的方向。本课介绍 2025 年最新的视频生成技术栈：

开源模型（可本地/云部署）：

  • Wan2.1（阿里巴巴，2025最强开源）— 支持文生视频/图生视频，支持中英文，可在 12GB 显存运行
  • HunyuanVideo（腾讯，11.8k ⭐）— 高质量开源，720p 视频，人物动作真实
  • CogVideoX（智谱，12.5k ⭐）— 开源，可商用，支持 6B 和 2B 两个版本

商业 API：

  • Kling（快手）— 中文场景最优，API 稳定，国内访问无需代理
  • Runway Gen-3— 图生视频国际标准，有官方 SDK
  • Sora（OpenAI）— 最高质量，价格最高

本地工作流：

  • ComfyUI（106k ⭐）— 最流行的可视化工作流工具，支持 Wan2.1/CogVideoX 节点

视频生成有一个独特的工程挑战：生成时间长（30秒到几分钟），必须用异步任务模式处理，文件体积大（需要流式下载和存储管理）。

遇到什么问题

视频生成时间长（异步任务），需要轮询状态；文件大，需要正确处理存储和传输：

  1. 生成耗时：一个 5 秒视频可能需要 30-120 秒生成，不能用同步 HTTP 请求
  2. 轮询策略：过于频繁轮询浪费资源，间隔太长增加延迟，需要指数退避
  3. 文件管理：视频文件通常 5-50MB，需要流式下载、临时存储、及时清理
  4. 任务失败处理：生成可能因内容审查或服务异常失败，需要优雅降级

# wan2.1_demo.py
import torch
from diffusers import WanPipeline
from diffusers.utils import export_to_video

def generate_video_wan(
    prompt: str,
    output_path: str = "output.mp4",
    num_frames: int = 81,
    model_variant: str = "14B"  # "14B" 或 "1.3B"
) -> str:
    """用 Wan2.1 生成视频（完全本地，无 API 费用）"""

    model_id = f"Wan-AI/Wan2.1-T2V-{model_variant}"

    print(f"加载模型 {model_id}...")
    pipe = WanPipeline.from_pretrained(
        model_id,
        torch_dtype=torch.bfloat16 if model_variant == "14B" else torch.float16
    )
    pipe.enable_model_cpu_offload()  # 显存不够时自动 offload 到内存

    print(f"生成视频: {prompt}")
    output = pipe(
        prompt=prompt,
        num_frames=num_frames,
        guidance_scale=5.0,
        num_inference_steps=50,
        height=480,
        width=832,
        generator=torch.Generator("cuda").manual_seed(42)
    )

    export_to_video(output.frames[0], output_path, fps=15)
    print(f"✅ 视频已保存: {output_path}")
    return output_path

# 测试

Runway API 实现

// src/video/runway.ts
import RunwayML from "@runwayml/sdk";
import * as fs from "fs";

const runway = new RunwayML({ apiKey: process.env.RUNWAYML_API_SECRET });

interface VideoOptions {
  promptText: string;
  duration?: 5 | 10;       // 5 或 10 秒
  ratio?: "1280:768" | "768:1280" | "1104:832" | "832:1104"; // 横版/竖版/方形
  model?: "gen3a_turbo" | "gen3a"; // turbo 更快，gen3a 质量更高
}

// 文生视频
export async function textToVideo(
  options: VideoOptions,
  outputPath: string
): Promise<string> {
  console.log("提交视频生成任务...");

  const task = await runway.textToVideo.create({
    model: options.model || "gen3a_turbo",
    promptText: options.promptText,
    duration: options.duration || 5,
    ratio: options.ratio || "1280:768",
  });

  console.log(`任务 ID: ${task.id}，开始轮询...`);

  const completedTask = await pollUntilDone(task.id);
  return downloadVideo(completedTask.output![0], outputPath);
}

// 图生视频（让静态图像动起来）
export async function imageToVideo(
  imagePath: string,
  promptText: string,
  outputPath: string,
  options?: Partial<VideoOptions>
): Promise<string> {
  // 读取图像并转 base64
  const imageBuffer = fs.readFileSync(imagePath);
  const base64Image = imageBuffer.toString("base64");
  const mimeType = imagePath.endsWith(".png") ? "image/png" : "image/jpeg";
  const dataUri = `data:${mimeType};base64,${base64Image}`;

  console.log(`提交图生视频任务: ${promptText}`);

  const task = await runway.imageToVideo.create({
    model: options?.model || "gen3a_turbo",
    promptImage: dataUri,
    promptText,
    duration: options?.duration || 5,
    ratio: options?.ratio || "1280:768",
  });

  console.log(`任务 ID: ${task.id}`);

  const completedTask = await pollUntilDone(task.id);
  return downloadVideo(completedTask.output![0], outputPath);
}

// 轮询直到任务完成（指数退避）
async function pollUntilDone(
  taskId: string,
  maxWaitMs: number = 300000 // 最多等 5 分钟
): Promise<RunwayML.Task> {
  const startTime = Date.now();
  let pollInterval = 5000; // 初始 5 秒
  const maxInterval = 30000; // 最大 30 秒

  while (Date.now() - startTime < maxWaitMs) {
    const task = await runway.tasks.retrieve(taskId);

    console.log(`[${new Date().toLocaleTimeString()}] 状态: ${task.status}${
      task.progress ? ` (${Math.round(task.progress * 100)}%)` : ""
    }`);

    if (task.status === "SUCCEEDED") {
      console.log("✓ 视频生成完成");
      return task;
    }

    if (task.status === "FAILED") {
      throw new Error(`视频生成失败: ${task.failure || "未知错误"}`);
    }

    // 指数退避：每次轮询后等待时间递增
    await new Promise((r) => setTimeout(r, pollInterval));
    pollInterval = Math.min(pollInterval * 1.5, maxInterval);
  }

  throw new Error(`超时：视频生成超过 ${maxWaitMs / 1000} 秒`);
}

// 流式下载视频
async function downloadVideo(url: string, outputPath: string): Promise<string> {
  console.log("下载视频中...");

  const response = await fetch(url);
  if (!response.ok) throw new Error(`下载失败: ${response.status}`);

  const totalSize = Number(response.headers.get("content-length") || 0);
  const buffer = Buffer.from(await response.arrayBuffer());

  fs.writeFileSync(outputPath, buffer);
  console.log(`视频已保存: ${outputPath} (${(buffer.length / 1024 / 1024).toFixed(1)}MB)`);

  return outputPath;
}

视频处理（时长、分辨率、帧率）

// src/video/processor.ts
import * as fs from "fs";
import { exec } from "child_process";
import { promisify } from "util";

const execAsync = promisify(exec);

// 视频元数据提取（需要 ffprobe）
export async function getVideoInfo(videoPath: string): Promise<{
  duration: number;
  width: number;
  height: number;
  fps: number;
  size: number;
  format: string;
}> {
  const { stdout } = await execAsync(
    `ffprobe -v quiet -print_format json -show_streams -show_format "${videoPath}"`
  );

  const info = JSON.parse(stdout);
  const videoStream = info.streams.find((s: any) => s.codec_type === "video");

  return {
    duration: parseFloat(info.format.duration),
    width: videoStream.width,
    height: videoStream.height,
    fps: eval(videoStream.r_frame_rate), // "30/1" → 30
    size: fs.statSync(videoPath).size,
    format: info.format.format_name,
  };
}

// 视频压缩（使用 ffmpeg）
export async function compressVideo(
  inputPath: string,
  outputPath: string,
  options?: {
    crf?: number;        // 压缩质量 0-51，越大压缩越重，推荐 23-28
    maxWidth?: number;   // 限制宽度
    fps?: number;        // 目标帧率
  }
): Promise<void> {
  const { crf = 23, maxWidth = 1280, fps = 30 } = options || {};

  const scaleFilter = maxWidth ? `-vf "scale='min(${maxWidth},iw)':-2"` : "";
  const fpsFilter = `-r ${fps}`;

  const cmd = `ffmpeg -i "${inputPath}" ${scaleFilter} ${fpsFilter} -c:v libx264 -crf ${crf} -preset medium -c:a aac -y "${outputPath}"`;

  console.log("压缩视频中...");
  await execAsync(cmd);

  const originalSize = fs.statSync(inputPath).size;
  const compressedSize = fs.statSync(outputPath).size;
  const ratio = ((1 - compressedSize / originalSize) * 100).toFixed(1);
  console.log(`压缩完成: ${(originalSize / 1024 / 1024).toFixed(1)}MB → ${(compressedSize / 1024 / 1024).toFixed(1)}MB (${ratio}%)`);
}

// 生成视频缩略图
export async function generateThumbnail(
  videoPath: string,
  outputPath: string,
  timeSeconds: number = 1
): Promise<void> {
  await execAsync(
    `ffmpeg -i "${videoPath}" -ss ${timeSeconds} -vframes 1 -y "${outputPath}"`
  );
  console.log(`缩略图已保存: ${outputPath}`);
}

// 视频转 GIF（适合预览）
export async function videoToGif(
  videoPath: string,
  outputPath: string,
  options?: { fps?: number; scale?: number; duration?: number }
): Promise<void> {
  const { fps = 10, scale = 480, duration = 3 } = options || {};

  await execAsync(
    `ffmpeg -i "${videoPath}" -t ${duration} -vf "fps=${fps},scale=${scale}:-1:flags=lanczos,split[s0][s1];[s0]palettegen[p];[s1][p]paletteuse" -loop 0 -y "${outputPath}"`
  );
  console.log(`GIF 已保存: ${outputPath}`);
}
