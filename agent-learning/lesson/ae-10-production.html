第10课：生产 Agent — 可靠性、监控、成本控制

介绍

Agent 从原型到生产需要解决三大问题：可靠性（API 故障时优雅降级）、可观测性（追踪每次调用的延迟、Token 用量、错误率）、成本控制（缓存、模型路由、输出裁剪）。本课实现完整的生产级 Agent 基础设施：熔断器、调用追踪、成本监控、提示词缓存。

遇到什么问题

生产环境中 Agent 面临的挑战远超开发阶段：

flowchart TD
    A[用户请求] --> B{API 可用？}
    B -->|可用| C[调用 LLM]
    B -->|超时/限流| D[熔断器]
    C --> E{响应正常？}
    E -->|正常| F[返回结果]
    E -->|错误/幻觉| G[重试/降级]
    D --> H[Fallback 模型]
    H --> F
    F --> I[记录日志\n追踪 Token 用量\n成本统计]
    G --> I
    I --> J[监控告警\n超成本/高延迟]

有哪些方案

方案 A：简单重试

for (let i = 0; i < 3; i++) {
  try { return await callLLM(prompt); }
  catch (e) { await sleep(1000 * (i + 1)); }
}

✅ 简单 ❌ 重试风暴：API 故障时大量请求同时重试，加重服务压力

方案 B：熔断器 + 指数退避

const breaker = new CircuitBreaker(openaiClient);
const result = await breaker.call(prompt);  // 故障时自动切换备用模型

✅ 防止重试风暴 ✅ 自动故障恢复

方案 C：全栈生产基础设施（推荐）

熔断器 + 调用追踪 + 成本监控 + 提示词缓存，形成完整的生产级解决方案。

✅ 高可靠性：三级 Fallback ✅ 全可观测：每次调用都有完整追踪 ✅ 低成本：缓存命中率 40-60%

怎么选择

选择方案 C，逐步建立生产基础设施。每个组件独立，可以按需引入。优先级：熔断器 > 调用追踪 > 成本监控 > 提示词缓存。

怎么执行

熔断器（Circuit Breaker）

// src/production/circuit_breaker.ts

type CircuitState = "CLOSED" | "OPEN" | "HALF_OPEN";

export interface CircuitBreakerConfig {
  failureThreshold: number;    // 触发熔断的连续失败次数
  recoveryTimeout: number;     // 熔断后等待恢复的时间（ms）
  halfOpenRequests: number;    // HALF_OPEN 状态允许的测试请求数
}

export class CircuitBreaker {
  private state: CircuitState = "CLOSED";
  private failureCount = 0;
  private lastFailureTime = 0;
  private halfOpenCount = 0;

  constructor(private config: CircuitBreakerConfig = {
    failureThreshold: 5,
    recoveryTimeout: 60_000,
    halfOpenRequests: 2,
  }) {}

  async call<T>(fn: () => Promise<T>): Promise<T> {
    if (this.state === "OPEN") {
      if (Date.now() - this.lastFailureTime > this.config.recoveryTimeout) {
        this.state = "HALF_OPEN";
        this.halfOpenCount = 0;
        console.log("[CircuitBreaker] 状态: OPEN → HALF_OPEN，开始探测");
      } else {
        throw new Error(`CircuitBreaker OPEN: 服务不可用，${Math.ceil((this.config.recoveryTimeout - (Date.now() - this.lastFailureTime)) / 1000)}s 后重试`);
      }
    }

    if (this.state === "HALF_OPEN" && this.halfOpenCount >= this.config.halfOpenRequests) {
      throw new Error("CircuitBreaker HALF_OPEN: 探测请求数已满");
    }

    try {
      const result = await fn();

      if (this.state === "HALF_OPEN") {
        this.state = "CLOSED";
        this.failureCount = 0;
        console.log("[CircuitBreaker] 状态: HALF_OPEN → CLOSED，服务恢复");
      } else {
        this.failureCount = 0;
      }

      return result;
    } catch (error) {
      this.failureCount++;
      this.lastFailureTime = Date.now();

      if (this.state === "HALF_OPEN") {
        this.state = "OPEN";
        console.log("[CircuitBreaker] 状态: HALF_OPEN → OPEN，探测失败");
      } else if (this.failureCount >= this.config.failureThreshold) {
        this.state = "OPEN";
        console.log(`[CircuitBreaker] 状态: CLOSED → OPEN，连续失败 ${this.failureCount} 次`);
      }

      throw error;
    }
  }

  getState(): CircuitState {
    return this.state;
  }
}

调用追踪（Observability）

// src/production/tracer.ts
import * as fs from "fs";

export interface CallTrace {
  traceId: string;
  timestamp: number;
  model: string;
  inputTokens: number;
  outputTokens: number;
  latencyMs: number;
  success: boolean;
  error?: string;
  cost: number;         // 美元
  tags?: Record<string, string>;
}

// 每百万 Token 的价格（美元）
const PRICING: Record<string, { input: number; output: number }> = {
  "gpt-4o": { input: 2.5, output: 10 },
  "gpt-4o-mini": { input: 0.15, output: 0.6 },
  "claude-opus-4-5": { input: 15, output: 75 },
  "claude-sonnet-4-5": { input: 3, output: 15 },
  "claude-haiku-4-5": { input: 0.25, output: 1.25 },
};

export class CallTracer {
  private traces: CallTrace[] = [];
  private readonly persistPath?: string;

  constructor(persistPath?: string) {
    this.persistPath = persistPath;
  }

  record(trace: CallTrace): void {
    this.traces.push(trace);

    if (this.persistPath) {
      // 追加写入，不重写全部
      fs.appendFileSync(
        this.persistPath,
        JSON.stringify(trace) + "\n"
      );
    }

    // 实时告警
    if (trace.cost > 0.1) {
      console.warn(`[Tracer] 高成本调用: $${trace.cost.toFixed(4)}, 模型: ${trace.model}`);
    }
    if (trace.latencyMs > 10_000) {
      console.warn(`[Tracer] 高延迟调用: ${trace.latencyMs}ms, 模型: ${trace.model}`);
    }
  }

  static calculateCost(model: string, inputTokens: number, outputTokens: number): number {
    const pricing = PRICING[model] || { input: 1, output: 3 };
    return (inputTokens / 1_000_000) * pricing.input +
           (outputTokens / 1_000_000) * pricing.output;
  }

  getSummary(lastNHours: number = 24): {
    totalCalls: number;
    successRate: number;
    avgLatency: number;
    totalCost: number;
    costByModel: Record<string, number>;
    p95Latency: number;
  } {
    const cutoff = Date.now() - lastNHours * 3_600_000;
    const recent = this.traces.filter((t) => t.timestamp > cutoff);

    if (recent.length === 0) {
      return { totalCalls: 0, successRate: 1, avgLatency: 0, totalCost: 0, costByModel: {}, p95Latency: 0 };
    }

    const latencies = recent.map((t) => t.latencyMs).sort((a, b) => a - b);
    const p95Index = Math.floor(latencies.length * 0.95);

    const costByModel: Record<string, number> = {};
    for (const trace of recent) {
      costByModel[trace.model] = (costByModel[trace.model] || 0) + trace.cost;
    }

    return {
      totalCalls: recent.length,
      successRate: recent.filter((t) => t.success).length / recent.length,
      avgLatency: latencies.reduce((s, v) => s + v, 0) / latencies.length,
      totalCost: recent.reduce((s, t) => s + t.cost, 0),
      costByModel,
      p95Latency: latencies[p95Index] || 0,
    };
  }
}

提示词缓存（Prompt Cache）

// src/production/cache.ts
import * as crypto from "crypto";

export interface CacheEntry {
  response: string;
  timestamp: number;
  hits: number;
}

export class PromptCache {
  private cache: Map<string, CacheEntry> = new Map();
  private readonly ttl: number;          // 缓存有效期（ms）
  private readonly maxSize: number;      // 最大缓存条目数

  constructor(ttl: number = 3_600_000, maxSize: number = 1000) {
    this.ttl = ttl;
    this.maxSize = maxSize;
  }

  private hashKey(model: string, messages: object[]): string {
    const content = JSON.stringify({ model, messages });
    return crypto.createHash("sha256").update(content).digest("hex").slice(0, 16);
  }

  get(model: string, messages: object[]): string | null {
    const key = this.hashKey(model, messages);
    const entry = this.cache.get(key);

    if (!entry) return null;
    if (Date.now() - entry.timestamp > this.ttl) {
      this.cache.delete(key);
      return null;
    }

    entry.hits++;
    return entry.response;
  }

  set(model: string, messages: object[], response: string): void {
    // 超出容量时，删除命中次数最少的条目（LFU）
    if (this.cache.size >= this.maxSize) {
      let minHits = Infinity;
      let minKey = "";
      for (const [k, v] of this.cache) {
        if (v.hits < minHits) { minHits = v.hits; minKey = k; }
      }
      if (minKey) this.cache.delete(minKey);
    }

    const key = this.hashKey(model, messages);
    this.cache.set(key, { response, timestamp: Date.now(), hits: 0 });
  }

  getStats(): { size: number; hitRate: string } {
    let totalHits = 0;
    for (const entry of this.cache.values()) totalHits += entry.hits;
    return {
      size: this.cache.size,
      hitRate: this.cache.size > 0 ? `${((totalHits / (totalHits + this.cache.size)) * 100).toFixed(1)}%` : "0%",
    };
  }
}

生产级 Agent 封装

// src/production/production_agent.ts
import OpenAI from "openai";
import Anthropic from "@anthropic-ai/sdk";
import { CircuitBreaker } from "./circuit_breaker";
import { CallTracer } from "./tracer";
import { PromptCache } from "./cache";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

export class ProductionAgent {
  private primaryBreaker = new CircuitBreaker();
  private fallbackBreaker = new CircuitBreaker();
  private tracer: CallTracer;
  private cache: PromptCache;

  constructor(private options: {
    primaryModel?: string;
    fallbackModel?: string;
    enableCache?: boolean;
    tracePath?: string;
  } = {}) {
    this.tracer = new CallTracer(options.tracePath);
    this.cache = new PromptCache();
    this.options.primaryModel = options.primaryModel || "gpt-4o-mini";
    this.options.fallbackModel = options.fallbackModel || "claude-haiku-4-5";
  }

  async chat(
    messages: { role: string; content: string }[],
    tags?: Record<string, string>
  ): Promise<string> {
    // 1. 检查缓存
    if (this.options.enableCache) {
      const cached = this.cache.get(this.options.primaryModel!, messages);
      if (cached) {
        console.log("[ProductionAgent] 缓存命中");
        return cached;
      }
    }

    // 2. 尝试主模型
    try {
      const result = await this.callWithTracing(
        this.options.primaryModel!,
        messages,
        () => this.primaryBreaker.call(() => this.callOpenAI(this.options.primaryModel!, messages)),
        tags
      );

      if (this.options.enableCache) {
        this.cache.set(this.options.primaryModel!, messages, result);
      }

      return result;
    } catch (primaryError) {
      console.warn(`[ProductionAgent] 主模型失败: ${primaryError instanceof Error ? primaryError.message : primaryError}`);
    }

    // 3. Fallback 到备用模型
    console.log(`[ProductionAgent] 切换到备用模型: ${this.options.fallbackModel}`);
    try {
      return await this.callWithTracing(
        this.options.fallbackModel!,
        messages,
        () => this.fallbackBreaker.call(() => this.callClaude(this.options.fallbackModel!, messages)),
        { ...tags, fallback: "true" }
      );
    } catch (fallbackError) {
      // 4. 最终降级：返回错误占位符
      console.error("[ProductionAgent] 所有模型失败，返回降级响应");
      return "服务暂时不可用，请稍后重试。";
    }
  }

  private async callWithTracing(
    model: string,
    messages: object[],
    fn: () => Promise<string>,
    tags?: Record<string, string>
  ): Promise<string> {
    const start = Date.now();
    let success = false;
    let error: string | undefined;
    let result = "";

    try {
      result = await fn();
      success = true;
      return result;
    } catch (e) {
      error = e instanceof Error ? e.message : String(e);
      throw e;
    } finally {
      const latencyMs = Date.now() - start;
      const inputTokens = Math.ceil(JSON.stringify(messages).length / 4);
      const outputTokens = Math.ceil(result.length / 4);

      this.tracer.record({
        traceId: `trace_${Date.now()}`,
        timestamp: Date.now(),
        model,
        inputTokens,
        outputTokens,
        latencyMs,
        success,
        error,
        cost: CallTracer.calculateCost(model, inputTokens, outputTokens),
        tags,
      });
    }
  }

  private async callOpenAI(model: string, messages: { role: string; content: string }[]): Promise<string> {
    const response = await openai.chat.completions.create({
      model,
      messages: messages as OpenAI.ChatCompletionMessageParam[],
      temperature: 0.7,
    });
    return response.choices[0].message.content || "";
  }

  private async callClaude(model: string, messages: { role: string; content: string }[]): Promise<string> {
    const system = messages.find((m) => m.role === "system")?.content;
    const userMessages = messages.filter((m) => m.role !== "system");

    const response = await anthropic.messages.create({
      model,
      max_tokens: 2048,
      system,
      messages: userMessages as Anthropic.MessageParam[],
    });

    return response.content[0].type === "text" ? response.content[0].text : "";
  }

  printReport(): void {
    const summary = this.tracer.getSummary(24);
    console.log("\n=== 生产监控报告（过去 24 小时）===");
    console.log(`总调用数：${summary.totalCalls}`);
    console.log(`成功率：${(summary.successRate * 100).toFixed(1)}%`);
    console.log(`平均延迟：${summary.avgLatency.toFixed(0)}ms`);
    console.log(`P95 延迟：${summary.p95Latency.toFixed(0)}ms`);
    console.log(`总成本：$${summary.totalCost.toFixed(4)}`);
    console.log("各模型成本：");
    for (const [model, cost] of Object.entries(summary.costByModel)) {
      console.log(`  ${model}: $${cost.toFixed(4)}`);
    }
    console.log("\n缓存统计：", this.cache.getStats());
    console.log("熔断器状态：", {
      primary: this.primaryBreaker.getState(),
      fallback: this.fallbackBreaker.getState(),
    });
  }
}

// 完整演示
async function runDemo() {
  const agent = new ProductionAgent({
    primaryModel: "gpt-4o-mini",
    fallbackModel: "claude-haiku-4-5",
    enableCache: true,
    tracePath: "./agent_traces.jsonl",
  });

  const questions = [
    "什么是 Agent 工程？",
    "RAG 和 Fine-tuning 的区别是什么？",
    "什么是 Agent 工程？",  // 重复问题，应命中缓存
  ];

  for (const q of questions) {
    console.log(`\n问题：${q}`);
    const answer = await agent.chat([
      { role: "system", content: "你是 AI 工程助手，用中文简洁回答。" },
      { role: "user", content: q },
    ]);
    console.log(`回答：${answer.slice(0, 100)}...`);
  }

  agent.printReport();
}

runDemo().catch(console.error);
