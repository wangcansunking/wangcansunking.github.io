第2课：图像生成 — DALL-E 3 / FLUX / ComfyUI

介绍

2025 年图像生成格局已发生重大变化：FLUX（Black Forest Labs）凭借出色的文字渲染和逼真写实效果超越 SD XL，成为开源最佳；ComfyUI（106k ⭐）是最流行的可视化 AI 工作流平台，支持 FLUX、SD、视频等全系列模型。云端方案上，DALL-E 3 仍是零部署首选，Stability AI 提供精细参数控制。本课覆盖四个方案的完整代码，以及提示词工程技巧和图像后处理。

技术栈：

  • DALL-E 3（OpenAI）— 最佳云端方案，支持中文，零部署
  • FLUX.1（Black Forest Labs）— 2025 开源最佳，文字渲染超强
  • ComfyUI（106k ⭐）— 最流行可视化工作流，本地免费运行
  • Stability AI（SD3.5）— 支持负向提示词，精细风格控制

遇到什么问题

不同生成模型的 API 格式、参数、质量差异大：

  1. DALL-E 3：OpenAI 官方，质量最高，有内容审查，不支持负向提示词，返回 URL（1小时有效期）
  2. FLUX.1：开源最强，文字渲染准确，写实效果超越 SD XL，支持本地或 Replicate/Together AI 云端
  3. ComfyUI：本地运行，支持所有开源模型（FLUX/SD/视频），可视化节点编辑，完全免费
  4. Stability AI：支持负向提示词，参数控制细腻（style_preset、aspect_ratio），成本更低

有哪些方案

方案 A：DALL-E 3（OpenAI SDK）

const response = await openai.images.generate({
  model: "dall-e-3",
  prompt: "一只在宇宙中冲浪的猫，赛博朋克风格",
  size: "1024x1024",
  quality: "hd",
  style: "vivid",
});

✅ 质量最高，理解提示词能力强，支持中文 ✅ 直接用 OpenAI SDK，无额外依赖 ❌ 价格较贵（$0.04-$0.12/张） ❌ 不支持负向提示词，内容审查严格

方案 B：Stability AI REST API

const formData = new FormData();
formData.append("prompt", "elegant perfume bottle, white background");
formData.append("negative_prompt", "blurry, low quality, distorted");

✅ 支持负向提示词，风格控制更精细 ✅ 成本更低（约 $0.003/张） ❌ 需要手写 HTTP 请求，英文提示词 ❌ 整体质量略低于 DALL-E 3

快速决策：

  • 商业项目，最快交付：DALL-E 3（质量和可靠性最高）
  • 开源，文字要准确：FLUX.1（Replicate 云端或本地）
  • 本地探索，免费无限制：ComfyUI + FLUX
  • 精细风格控制/负向提示词：Stability AI SD3.5

DALL-E 3 完整代码

// src/image_gen/dalle3.ts
import OpenAI from "openai";
import * as fs from "fs";
import * as path from "path";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

interface DALLE3Options {
  prompt: string;
  size?: "1024x1024" | "1792x1024" | "1024x1792"; // 正方形/横版/竖版
  quality?: "standard" | "hd";                      // hd 更贵但更精细
  style?: "natural" | "vivid";                      // natural 写实，vivid 鲜艳
}

export async function generateWithDALLE3(options: DALLE3Options): Promise<{
  url: string;
  revisedPrompt: string;
}> {
  const response = await openai.images.generate({
    model: "dall-e-3",
    prompt: options.prompt,
    size: options.size || "1024x1024",
    quality: options.quality || "standard",
    style: options.style || "vivid",
    n: 1,
    response_format: "url",
  });

  const item = response.data[0];
  const imageUrl = item.url!;
  const revisedPrompt = item.revised_prompt || options.prompt;

  console.log("原始提示词:", options.prompt);
  if (revisedPrompt !== options.prompt) {
    console.log("DALL-E 改写为:", revisedPrompt);
  }

  return { url: imageUrl, revisedPrompt };
}

// 生成并立即保存（URL 只有约 1 小时有效期）
export async function generateAndSave(
  prompt: string,
  outputPath: string,
  options?: Partial<DALLE3Options>
): Promise<string> {
  const { url } = await generateWithDALLE3({ prompt, ...options });

  const response = await fetch(url);
  const buffer = Buffer.from(await response.arrayBuffer());

  const dir = path.dirname(outputPath);
  if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });

  fs.writeFileSync(outputPath, buffer);
  console.log(`图像已保存: ${outputPath} (${(buffer.length / 1024).toFixed(1)}KB)`);

  return outputPath;
}

// 批量生成（注意速率限制：DALL-E 3 标准账户 5 img/min）
export async function batchGenerate(
  prompts: string[],
  outputDir: string
): Promise<string[]> {
  const results: string[] = [];

  for (let i = 0; i < prompts.length; i++) {
    const outputPath = path.join(outputDir, `image_${i + 1}.png`);
    console.log(`[${i + 1}/${prompts.length}] 生成中: ${prompts[i].substring(0, 50)}...`);

    await generateAndSave(prompts[i], outputPath);
    results.push(outputPath);

    // 速率限制保护：每张间隔 12 秒（5 img/min）
    if (i < prompts.length - 1) {
      await new Promise((r) => setTimeout(r, 12000));
    }
  }

  return results;
}

Stability AI API

// src/image_gen/stability.ts
import * as fs from "fs";

const STABILITY_API_KEY = process.env.STABILITY_API_KEY!;
const BASE_URL = "https://api.stability.ai/v2beta";

interface StabilityOptions {
  prompt: string;
  negativePrompt?: string;  // DALL-E 3 不支持，这是 Stability 的优势
  aspectRatio?: "1:1" | "16:9" | "9:16" | "4:3" | "3:4" | "21:9";
  style?:
    | "photographic"
    | "anime"
    | "cinematic"
    | "digital-art"
    | "comic-book"
    | "neon-punk"
    | "line-art"
    | "3d-model";
  outputFormat?: "png" | "jpeg" | "webp";
  seed?: number; // 固定种子，用于复现结果
}

export async function generateWithStability(options: StabilityOptions): Promise<Buffer> {
  const formData = new FormData();
  formData.append("prompt", options.prompt);

  if (options.negativePrompt) formData.append("negative_prompt", options.negativePrompt);
  if (options.aspectRatio)    formData.append("aspect_ratio", options.aspectRatio);
  if (options.style)          formData.append("style_preset", options.style);
  if (options.outputFormat)   formData.append("output_format", options.outputFormat);
  if (options.seed !== undefined) formData.append("seed", String(options.seed));

  const response = await fetch(`${BASE_URL}/stable-image/generate/core`, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${STABILITY_API_KEY}`,
      Accept: "image/*",
    },
    body: formData,
  });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(`Stability AI 错误: ${JSON.stringify(error)}`);
  }

  return Buffer.from(await response.arrayBuffer());
}

// SD3 超高质量生成
export async function generateWithSD3(
  prompt: string,
  outputPath: string,
  negativePrompt?: string
): Promise<void> {
  const formData = new FormData();
  formData.append("prompt", prompt);
  formData.append("model", "sd3.5-large");
  formData.append("output_format", "png");
  if (negativePrompt) formData.append("negative_prompt", negativePrompt);

  const response = await fetch(`${BASE_URL}/stable-image/generate/sd3`, {
    method: "POST",
    headers: { Authorization: `Bearer ${STABILITY_API_KEY}`, Accept: "image/*" },
    body: formData,
  });

  if (!response.ok) throw new Error(`SD3 失败: ${response.statusText}`);
  fs.writeFileSync(outputPath, Buffer.from(await response.arrayBuffer()));
  console.log(`SD3 图像已保存: ${outputPath}`);
}

提示词工程技巧（正向/负向提示词）

// src/image_gen/prompt_engineering.ts

// 提示词结构：主体 + 动作 + 场景 + 风格 + 光线 + 质量关键词
interface PromptTemplate {
  subject: string;       // 主体（必填）
  action?: string;       // 动作/状态
  setting?: string;      // 场景/背景
  style?: string;        // 艺术风格
  lighting?: string;     // 光线效果
  camera?: string;       // 视角/镜头
  quality?: string[];    // 质量关键词
}

export function buildPrompt(template: PromptTemplate): string {
  const parts = [template.subject];
  if (template.action)  parts.push(template.action);
  if (template.setting) parts.push(`in ${template.setting}`);
  if (template.style)   parts.push(`${template.style} style`);
  if (template.lighting) parts.push(`${template.lighting} lighting`);
  if (template.camera)  parts.push(template.camera);
  if (template.quality?.length) parts.push(template.quality.join(", "));
  return parts.join(", ");
}

// 预设风格库（英文，适用于 Stability AI）
export const STYLE_PRESETS = {
  photorealistic: "photorealistic, 8k resolution, professional photography, sharp focus, bokeh",
  anime: "anime style, vibrant colors, detailed linework, Studio Ghibli inspired",
  oil_painting: "oil painting, impasto technique, rich textures, museum quality artwork",
  watercolor: "watercolor painting, soft colors, delicate brushstrokes, translucent",
  cyberpunk: "cyberpunk, neon lights, rain-slicked streets, futuristic atmosphere",
  minimalist: "minimalist design, clean lines, flat colors, white background",
  product_photo: "professional product photography, studio lighting, white background, commercial",
};

// 质量提升关键词（放在提示词末尾）
export const QUALITY_BOOSTS = {
  general: "masterpiece, best quality, highly detailed, 8k",
  artistic: "trending on artstation, award-winning, breathtaking, stunning composition",
  photo: "DSLR, professional photography, perfect exposure, tack sharp",
};

// 通用负向提示词（Stability AI 使用）
export const NEGATIVE_PROMPTS = {
  general: "blurry, low quality, distorted, ugly, bad anatomy, watermark, signature, text, logo",
  portrait: "extra limbs, bad hands, deformed face, extra fingers, missing fingers, low resolution",
  landscape: "people, cars, text watermark, overexposed, underexposed, washed out",
  product: "hands, people, cluttered background, shadow, reflection issues, distorted",
};

// 中文提示词增强（DALL-E 3 支持中文，但加英文质量词效果更好）
export function enhanceChinesePrompt(chinesePrompt: string): string {
  return `${chinesePrompt}, 高质量, 细节丰富, 专业摄影级别, masterpiece, highly detailed`;
}

// 产品图生成提示词
export function generateProductPrompt(
  productName: string,
  style: "photo" | "illustration" | "3d"
): { positive: string; negative: string } {
  const prompts = {
    photo: {
      positive: buildPrompt({
        subject: productName,
        setting: "clean white studio background",
        lighting: "soft studio professional",
        camera: "product photography, front view",
        quality: QUALITY_BOOSTS.photo.split(", "),
      }),
      negative: NEGATIVE_PROMPTS.product,
    },
    illustration: {
      positive: buildPrompt({
        subject: productName,
        style: "flat design illustration",
        quality: ["vector style", "clean", "modern", "colorful", "minimal"],
      }),
      negative: "photorealistic, 3d render, shadows, noise",
    },
    "3d": {
      positive: buildPrompt({
        subject: productName,
        style: "3D render, octane render",
        lighting: "dramatic studio",
        quality: ["hyperrealistic", "ray tracing", "subsurface scattering"],
      }),
      negative: NEGATIVE_PROMPTS.general,
    },
  };

  return prompts[style];
}

生成图像的后处理（保存、压缩）

// src/image_gen/post_process.ts
import * as fs from "fs";
import * as path from "path";

// 图像压缩
export async function compressImage(
  inputPath: string,
  outputPath: string,
  options?: { quality?: number; maxWidth?: number; format?: "jpeg" | "png" | "webp" }
): Promise<{ originalSize: number; compressedSize: number; ratio: string }> {
  const sharp = (await import("sharp")).default;
  const { quality = 85, maxWidth = 1024, format = "jpeg" } = options || {};

  let pipeline = sharp(inputPath);
  if (maxWidth) pipeline = pipeline.resize(maxWidth, undefined, { withoutEnlargement: true });
  pipeline = pipeline[format]({ quality });
  await pipeline.toFile(outputPath);

  const originalSize = fs.statSync(inputPath).size;
  const compressedSize = fs.statSync(outputPath).size;

  return {
    originalSize,
    compressedSize,
    ratio: `${((1 - compressedSize / originalSize) * 100).toFixed(1)}% 压缩`,
  };
}

// 生成响应式图像（多尺寸，适合 Web）
export async function generateResponsiveSizes(
  imagePath: string,
  outputDir: string,
  sizes: number[] = [320, 640, 1024, 1920]
): Promise<Record<number, string>> {
  const sharp = (await import("sharp")).default;
  if (!fs.existsSync(outputDir)) fs.mkdirSync(outputDir, { recursive: true });

  const results: Record<number, string> = {};
  for (const size of sizes) {
    const outputPath = path.join(outputDir, `image_${size}w.webp`);
    await sharp(imagePath)
      .resize(size, undefined, { withoutEnlargement: true })
      .webp({ quality: 80 })
      .toFile(outputPath);
    results[size] = outputPath;
  }

  return results;
}
