第4课：文档处理 — PDF + Word + Excel 智能解析

介绍

大量企业数据存在于 PDF、Word、Excel 中。本课构建文档处理 Pipeline，让 AI 能提取、理解、分析各类文档。覆盖 pdf-parse、mammoth、xlsx 三大解析库，以及针对超出上下文限制的大文档的分块处理策略和问答系统。

遇到什么问题

不同文档格式的解析方式不同，且大文档超过上下文限制需要分块处理：

  1. 格式差异：PDF 可能是扫描版（需要 OCR）或数字版（可直接提取文本）
  2. 上下文限制：100 页 PDF 约 50K-200K tokens，超出大多数模型的上下文窗口
  3. 表格提取：PDF 中的表格很难保持结构，需要特殊处理
  4. 分块策略：简单按字数切割会破坏语义，需要按自然段落或语义边界切割

有哪些方案

方案 A：直接解析 + 全量传入（适合短文档）

const text = await pdfParse(buffer);
const answer = await llm.chat([{ role: "user", content: text + "\n\n" + question }]);

✅ 简单，不丢失信息 ❌ 超过上下文就报错，无法扩展

方案 B：分块 + 向量检索（RAG 方式）

先分块、embedding、存向量库，查询时只取相关块。

✅ 可处理任意长度文档 ✅ 检索精准 ❌ 需要向量库，有嵌入成本，实现复杂

方案 C：Map-Reduce 方式（适合摘要任务）

// 每块分别摘要，再合并摘要
const chunkSummaries = await Promise.all(chunks.map(summarize));
const finalSummary = await summarize(chunkSummaries.join("\n\n"));

✅ 不需要向量库，适合摘要/分析任务 ❌ 不适合需要跨块检索的问答

怎么选择

**短文档（<50K tokens）**用方案 A；需要精准问答的长文档用方案 B（RAG）；摘要/分析任务用方案 C（Map-Reduce）。本课实现三种方式，并提供选择建议。

怎么执行

文档处理 Pipeline

flowchart TD
    A[输入文档] --> B{文档类型}
    B -->|PDF| C[pdf-parse 提取文本]
    B -->|Word .docx| D[mammoth 提取文本]
    B -->|Excel .xlsx| E[xlsx 提取表格]

    H --> L[LLM 回答]
    J --> L
    K --> L

PDF 解析

// src/documents/pdf_parser.ts
import * as fs from "fs";

export interface ParsedDocument {
  text: string;
  pages: string[];  // 每页的文本
  metadata: {
    pageCount: number;
    title?: string;
    author?: string;
    estimatedTokens: number;
  };
}

export async function parsePDF(filePath: string): Promise<ParsedDocument> {
  const pdfParse = (await import("pdf-parse")).default;
  const buffer = fs.readFileSync(filePath);

  const data = await pdfParse(buffer);

  // 按页分割文本（pdf-parse 用 \n\n 分页）
  const pages = data.text
    .split(/\f/)  // 换页符
    .map((p: string) => p.trim())
    .filter((p: string) => p.length > 0);

  return {
    text: data.text,
    pages,
    metadata: {
      pageCount: data.numpages,
      title: data.info?.Title,
      author: data.info?.Author,
      estimatedTokens: Math.ceil(data.text.length / 3),
    },
  };
}

// 扫描版 PDF（需要 OCR）
export async function parsePDFWithOCR(filePath: string): Promise<string> {
  // 使用 Gemini Vision 处理扫描版 PDF
  const { GoogleGenerativeAI } = await import("@google/generative-ai");
  const { GoogleAIFileManager } = await import("@google/generative-ai/server");

  const fileManager = new GoogleAIFileManager(process.env.GEMINI_API_KEY!);
  const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);

  // 上传 PDF
  const upload = await fileManager.uploadFile(filePath, {
    mimeType: "application/pdf",
    displayName: "OCR 文档",
  });

  // 等待处理
  let file = upload.file;
  while (file.state === "PROCESSING") {
    await new Promise((r) => setTimeout(r, 2000));
    file = await fileManager.getFile(file.name);
  }

  // 用 Gemini Vision 提取文字
  const model = genAI.getGenerativeModel({ model: "gemini-1.5-pro" });
  const result = await model.generateContent([
    { fileData: { mimeType: "application/pdf", fileUri: file.uri } },
    "请提取这份 PDF 中的所有文字内容，保持原始格式和排版。如果有表格，用 Markdown 表格格式输出。",
  ]);

  await fileManager.deleteFile(file.name);
  return result.response.text();
}

Word 解析

// src/documents/word_parser.ts
import * as fs from "fs";

export interface WordDocument {
  text: string;           // 纯文本
  markdown: string;       // 保留格式的 Markdown
  tables: string[][];     // 提取的表格数据
}

export async function parseWord(filePath: string): Promise<WordDocument> {
  const mammoth = await import("mammoth");

  const buffer = fs.readFileSync(filePath);

  // 提取纯文本
  const textResult = await mammoth.extractRawText({ buffer });

  // 提取带格式的 Markdown
  const htmlResult = await mammoth.convertToHtml({ buffer });
  const markdown = htmlToMarkdown(htmlResult.value);

  // 提取表格（mammoth 会把表格转成 HTML table）
  const tables = extractTablesFromHTML(htmlResult.value);

  if (textResult.messages.length > 0) {
    console.warn("Word 解析警告:", textResult.messages);
  }

  return {
    text: textResult.value,
    markdown,
    tables,
  };
}

function htmlToMarkdown(html: string): string {
  // 简单的 HTML → Markdown 转换（生产环境建议用 turndown 库）
  return html
    .replace(/<h1[^>]*>(.*?)<\/h1>/gi, "# $1\n")
    .replace(/<h2[^>]*>(.*?)<\/h2>/gi, "## $1\n")
    .replace(/<h3[^>]*>(.*?)<\/h3>/gi, "### $1\n")
    .replace(/<strong[^>]*>(.*?)<\/strong>/gi, "**$1**")
    .replace(/<em[^>]*>(.*?)<\/em>/gi, "*$1*")
    .replace(/<li[^>]*>(.*?)<\/li>/gi, "- $1")
    .replace(/<p[^>]*>(.*?)<\/p>/gi, "$1\n\n")
    .replace(/<br\s*\/?>/gi, "\n")
    .replace(/<[^>]+>/g, "") // 移除其他标签
    .trim();
}

function extractTablesFromHTML(html: string): string[][] {
  const tableRegex = /<table[^>]*>([\s\S]*?)<\/table>/gi;
  const rowRegex = /<tr[^>]*>([\s\S]*?)<\/tr>/gi;
  const cellRegex = /<t[dh][^>]*>([\s\S]*?)<\/t[dh]>/gi;

  const tables: string[][] = [];
  let tableMatch;

  while ((tableMatch = tableRegex.exec(html)) !== null) {
    const rows: string[] = [];
    let rowMatch;
    const rowRegexCopy = new RegExp(rowRegex.source, rowRegex.flags);

    while ((rowMatch = rowRegexCopy.exec(tableMatch[1])) !== null) {
      const cells: string[] = [];
      let cellMatch;
      const cellRegexCopy = new RegExp(cellRegex.source, cellRegex.flags);

      while ((cellMatch = cellRegexCopy.exec(rowMatch[1])) !== null) {
        cells.push(cellMatch[1].replace(/<[^>]+>/g, "").trim());
      }

      if (cells.length > 0) rows.push(cells.join(" | "));
    }

    if (rows.length > 0) tables.push(rows);
  }

  return tables;
}

Excel 解析

// src/documents/excel_parser.ts
import * as fs from "fs";
import * as XLSX from "xlsx";

export interface ExcelData {
  sheets: Record<string, {
    name: string;
    headers: string[];
    rows: Record<string, any>[];
    markdown: string;
  }>;
}

export function parseExcel(filePath: string): ExcelData {
  const buffer = fs.readFileSync(filePath);
  const workbook = XLSX.read(buffer, { type: "buffer", cellDates: true });

  const result: ExcelData = { sheets: {} };

  for (const sheetName of workbook.SheetNames) {
    const sheet = workbook.Sheets[sheetName];

    // 转为 JSON 数组（第一行为 headers）
    const jsonData = XLSX.utils.sheet_to_json(sheet, { header: 1 }) as any[][];

    if (jsonData.length === 0) continue;

    const headers = jsonData[0].map(String);
    const rows = jsonData.slice(1).map((row) => {
      const obj: Record<string, any> = {};
      headers.forEach((h, i) => {
        obj[h] = row[i] ?? "";
      });
      return obj;
    });

    // 转为 Markdown 表格（方便传给 LLM）
    const markdown = rowsToMarkdown(headers, rows);

    result.sheets[sheetName] = { name: sheetName, headers, rows, markdown };
  }

  return result;
}

function rowsToMarkdown(headers: string[], rows: Record<string, any>[]): string {
  const headerRow = "| " + headers.join(" | ") + " |";
  const separatorRow = "| " + headers.map(() => "---").join(" | ") + " |";
  const dataRows = rows.slice(0, 100).map((row) => {
    // 限制 100 行，避免超出上下文
    const cells = headers.map((h) => String(row[h] ?? "").replace(/\|/g, "\\|"));
    return "| " + cells.join(" | ") + " |";
  });

  const footer = rows.length > 100
    ? `\n> 注：仅显示前 100 行，共 ${rows.length} 行`
    : "";

  return [headerRow, separatorRow, ...dataRows].join("\n") + footer;
}

// Excel 数据分析
export async function analyzeExcel(filePath: string, question: string): Promise<string> {
  const excelData = parseExcel(filePath);
  const OpenAI = (await import("openai")).default;
  const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

  // 构建包含所有 sheet 数据的 prompt
  const sheetsContent = Object.values(excelData.sheets)
    .map((sheet) => `## Sheet: ${sheet.name}\n\n${sheet.markdown}`)
    .join("\n\n---\n\n");

  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      {
        role: "system",
        content: "你是一个数据分析专家，能从 Excel 数据中提取洞察和回答问题。",
      },
      {
        role: "user",
        content: `以下是 Excel 文件的内容：\n\n${sheetsContent}\n\n问题：${question}`,
      },
    ],
    temperature: 0.1,
    response_format: { type: "text" },
  });

  return response.choices[0].message.content || "";
}

分块策略

// src/documents/chunking.ts

export interface Chunk {
  text: string;
  index: number;
  startChar: number;
  endChar: number;
  metadata?: Record<string, any>;
}

// 策略1：按固定长度分块（最简单）
export function fixedSizeChunking(
  text: string,
  chunkSize: number = 1000,
  overlap: number = 200
): Chunk[] {
  const chunks: Chunk[] = [];
  let start = 0;

  while (start < text.length) {
    const end = Math.min(start + chunkSize, text.length);
    chunks.push({
      text: text.slice(start, end),
      index: chunks.length,
      startChar: start,
      endChar: end,
    });
    start += chunkSize - overlap; // 重叠保证上下文连贯
  }

  return chunks;
}

// 策略2：按段落分块（保留自然结构）
export function paragraphChunking(
  text: string,
  maxChunkSize: number = 2000
): Chunk[] {
  const paragraphs = text.split(/\n{2,}/).filter((p) => p.trim().length > 0);
  const chunks: Chunk[] = [];
  let currentChunk = "";
  let startChar = 0;

  for (const paragraph of paragraphs) {
    if (currentChunk.length + paragraph.length > maxChunkSize && currentChunk.length > 0) {
      chunks.push({
        text: currentChunk.trim(),
        index: chunks.length,
        startChar,
        endChar: startChar + currentChunk.length,
      });
      startChar += currentChunk.length;
      currentChunk = "";
    }
    currentChunk += (currentChunk ? "\n\n" : "") + paragraph;
  }

  if (currentChunk.trim()) {
    chunks.push({
      text: currentChunk.trim(),
      index: chunks.length,
      startChar,
      endChar: startChar + currentChunk.length,
    });
  }

  return chunks;
}

// 策略3：语义分块（递归分割，优先按更大的边界）
export function recursiveChunking(
  text: string,
  maxChunkSize: number = 1500,
  separators: string[] = ["\n\n", "\n", "。", ".", " "]
): Chunk[] {
  function splitText(text: string, separators: string[]): string[] {
    if (text.length <= maxChunkSize) return [text];

    const separator = separators[0];
    if (!separator) return [text.slice(0, maxChunkSize)];

    const parts = text.split(separator);
    const result: string[] = [];
    let current = "";

    for (const part of parts) {
      if ((current + separator + part).length <= maxChunkSize) {
        current = current ? current + separator + part : part;
      } else {
        if (current) result.push(current);
        if (part.length > maxChunkSize) {
          result.push(...splitText(part, separators.slice(1)));
          current = "";
        } else {
          current = part;
        }
      }
    }

    if (current) result.push(current);
    return result;
  }

  const parts = splitText(text, separators);
  return parts.map((text, index) => ({
    text,
    index,
    startChar: 0,
    endChar: text.length,
  }));
}

大文档问答系统（Map-Reduce）

// src/documents/qa_system.ts
import { parsePDF } from "./pdf_parser";
import { recursiveChunking } from "./chunking";
import OpenAI from "openai";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// Map-Reduce 方式：适合摘要和分析
export async function summarizeDocument(filePath: string): Promise<string> {
  const doc = await parsePDF(filePath);
  const chunks = recursiveChunking(doc.text, 3000);

  console.log(`文档分为 ${chunks.length} 个块`);

  // Map：每块独立摘要
  const summaries = await Promise.all(
    chunks.map(async (chunk, i) => {
      const response = await openai.chat.completions.create({
        model: "gpt-4o-mini", // 用小模型处理各块，节省成本
        messages: [
          { role: "user", content: `请总结以下文本的核心内容（100字以内）：\n\n${chunk.text}` },
        ],
        max_tokens: 200,
      });
      console.log(`块 ${i + 1}/${chunks.length} 摘要完成`);
      return response.choices[0].message.content || "";
    })
  );

  // Reduce：合并所有摘要
  const combinedSummaries = summaries.join("\n\n---\n\n");

  const finalResponse = await openai.chat.completions.create({
    model: "gpt-4o", // 最终摘要用强模型
    messages: [
      {
        role: "user",
        content: `以下是一份文档各部分的摘要，请综合生成一个完整的总结（500字以内）：\n\n${combinedSummaries}`,
      },
    ],
  });

  return finalResponse.choices[0].message.content || "";
}

// 简单的 QA（适合短文档）
export async function answerQuestion(
  filePath: string,
  question: string
): Promise<string> {
  const ext = filePath.split(".").pop()?.toLowerCase();

  let text = "";
  if (ext === "pdf") {
    const doc = await parsePDF(filePath);
    text = doc.text;
  } else if (ext === "docx") {
    const { parseWord } = await import("./word_parser");
    const doc = await parseWord(filePath);
    text = doc.text;
  } else if (ext === "xlsx") {
    const { parseExcel } = await import("./excel_parser");
    const data = parseExcel(filePath);
    text = Object.values(data.sheets)
      .map((s) => `# ${s.name}\n${s.markdown}`)
      .join("\n\n");
  }

  const estimatedTokens = text.length / 3;
  const model = estimatedTokens > 50000 ? "gpt-4o" : "gpt-4o-mini";

  const response = await openai.chat.completions.create({
    model,
    messages: [
      { role: "user", content: `文档内容：\n\n${text}\n\n问题：${question}` },
    ],
    temperature: 0.1,
  });

  return response.choices[0].message.content || "";
}
