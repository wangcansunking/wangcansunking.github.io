第9课：Agent Teams — MessageBus 多 Agent 协作

介绍

单个 Agent 能力有限。Agent Teams 允许多个专业化 Agent 通过共享 MessageBus 协作：每个 Agent 有自己的角色、工具集和系统 Prompt，通过消息传递协调工作，实现真正的并发多智能体系统。

源码参考：github.com/shareAI-lab/learn-claude-code — agents/s09_agent_teams.py

遇到什么问题

复杂工程任务需要不同专业知识：

任务：实现新功能 + 安全审查 + 性能优化

单 Agent：
- 切换角色导致注意力分散
- 安全专业知识 vs 性能优化互相干扰
- 所有历史混在一起，上下文污染

多 Agent Teams：
┌─────────────┐   MessageBus   ┌──────────────┐
│ Dev Agent   │ ──────────── → │ Security Agent│
│ 实现功能     │ ← ──────────── │ 安全审查      │
└─────────────┘                └──────────────┘
        ↕
┌─────────────┐
│ Perf Agent  │
│ 性能优化     │
└─────────────┘

有哪些方案

方案 A：顺序调用子 Agent（s04 方案）

父 Agent 逐个调用子 Agent，等待结果。

✅ 简单、顺序清晰 ❌ 串行执行，无法实现 Agent 间实时协作

方案 B：共享文件通信

Agent 通过写/读共享文件交换信息。

✅ 持久化 ❌ 轮询延迟，竞争条件，文件锁复杂

方案 C：MessageBus + 线程 Agent（推荐）

bus = MessageBus()
agents = [DevAgent(bus), SecurityAgent(bus), PerfAgent(bus)]
# 每个 Agent 在独立线程运行，订阅相关消息

✅ 实时通信，真正并发 ✅ 每个 Agent 保持独立上下文 ✅ 松耦合：Agent 只需关心自己订阅的消息类型

怎么选择

方案 C。MessageBus 是多智能体系统的标准通信模式。线程隔离让每个 Agent 有独立的对话历史和工具集。

MessageBus 实现

# agents/s09_agent_teams.py
import threading
import queue
import time
from dataclasses import dataclass, field
from typing import Callable, Any


@dataclass
class Message:
    sender: str
    recipient: str     # agent name or "broadcast"
    type: str          # "task" | "result" | "question" | "answer"
    content: str
    metadata: dict = field(default_factory=dict)
    timestamp: float = field(default_factory=time.time)


class MessageBus:
    def __init__(self):
        self._queues: dict[str, queue.Queue] = {}
        self._history: list[Message] = []
        self._lock = threading.Lock()

    def register(self, agent_name: str):
        """注册 Agent，创建专属消息队列"""
        with self._lock:
            self._queues[agent_name] = queue.Queue()

    def send(self, msg: Message):
        """发送消息"""
        with self._lock:
            self._history.append(msg)
            if msg.recipient == "broadcast":
                for q in self._queues.values():
                    q.put(msg)
            elif msg.recipient in self._queues:
                self._queues[msg.recipient].put(msg)

    def receive(self, agent_name: str, timeout: float = 5.0) -> Message | None:
        """接收消息（阻塞直到超时）"""
        q = self._queues.get(agent_name)
        if not q:
            return None
        try:
            return q.get(timeout=timeout)
        except queue.Empty:
            return None

    def get_history(self) -> list[Message]:
        with self._lock:
            return list(self._history)

TeamAgent 基类

class TeamAgent(threading.Thread):
    def __init__(self, name: str, role: str, bus: MessageBus,
                 tools: list, system_extra: str = ""):
        super().__init__(daemon=True)
        self.name = name
        self.role = role
        self.bus = bus
        self.tools = tools
        self.messages = []   # 每个 Agent 独立对话历史
        self.system = f"""You are the {role} on an engineering team.
{system_extra}

Team communication:
- send_message(recipient, type, content): Send to teammate
- get_messages(): Check your incoming messages
- report_done(summary): Signal task completion"""
        bus.register(name)

    def run(self):
        """Agent 主循环：处理消息 + 执行任务"""
        while True:
            msg = self.bus.receive(self.name, timeout=1.0)
            if msg:
                self._handle_message(msg)

    def _handle_message(self, msg: Message):
        """将消息转化为 Agent 任务"""
        self.messages.append({
            "role": "user",
            "content": f"[Message from {msg.sender}]\nType: {msg.type}\n\n{msg.content}",
        })
        self._agent_loop()

    def _agent_loop(self):
        """标准 Agent 循环"""
        tool_handlers = {
            **self._get_tool_handlers(),
            "send_message": lambda **kw: self._send_msg(**kw),
            "get_messages":  lambda **kw: self._get_pending(),
        }
        # 调用 LLM，执行工具，继续循环...
        # （与 s02 相同的 loop 结构）

    def _send_msg(self, recipient: str, type: str, content: str) -> str:
        self.bus.send(Message(
            sender=self.name, recipient=recipient,
            type=type, content=content,
        ))
        return f"Message sent to {recipient}"

专业化 Agent 定义

def create_team(bus: MessageBus) -> list[TeamAgent]:
    """创建专业化 Agent 团队"""

    dev_agent = TeamAgent(
        name="dev", role="Senior Developer",
        bus=bus, tools=CODE_TOOLS,
        system_extra="Focus on clean, correct implementation. "
                     "Ask security agent to review sensitive code.",
    )

    security_agent = TeamAgent(
        name="security", role="Security Engineer",
        bus=bus, tools=READONLY_TOOLS,  # 只读，不能直接修改代码
        system_extra="Review code for vulnerabilities. "
                     "Report findings to dev agent for fixes.",
    )

    perf_agent = TeamAgent(
        name="perf", role="Performance Engineer",
        bus=bus, tools=PROFILING_TOOLS,
        system_extra="Analyze performance bottlenecks. "
                     "Suggest optimizations to dev agent.",
    )

    return [dev_agent, security_agent, perf_agent]

启动团队

def run_team(task: str):
    bus = MessageBus()
    team = create_team(bus)

    # 启动所有 Agent 线程
    for agent in team:
        agent.start()

    # 向 Dev Agent 分配任务
    bus.send(Message(
        sender="orchestrator",
        recipient="dev",
        type="task",
        content=task,
    ))

    # 等待任务完成（简化版）
    time.sleep(60)

    # 打印通信历史
    for msg in bus.get_history():
        print(f"[{msg.sender} → {msg.recipient}] {msg.type}: {msg.content[:80]}")
