第8课：LangGraph — 状态机驱动的 Agent 工作流

介绍

LangGraph 用有向图（节点 = 动作，边 = 路由条件）来建模 Agent 工作流，解决了 ReAct 循环无法处理复杂分支和状态持久化的问题。本课用纯 TypeScript 实现 LangGraph 的核心概念：StateGraph、节点、条件边、Checkpointer，不依赖 LangGraph 库本身，理解其设计原理。

遇到什么问题

ReAct 是线性循环，无法处理条件分支、并行执行、人工介入、状态恢复：

flowchart TD
    subgraph ReAct["ReAct 循环（线性）"]
        A1[思考] --> B1[行动] --> C1[观察] --> A1
    end

    subgraph LangGraph["LangGraph（有向图）"]
        A2[节点A] -->|条件1| B2[节点B]
        A2 -->|条件2| C2[节点C]
        B2 --> D2[聚合节点]
        C2 --> D2
        D2 -->|需要审批| E2[人工介入]
        D2 -->|自动通过| F2[结束]
        E2 --> F2
    end

有哪些方案

方案 A：手动状态机（if/else 条件跳转）

let state = "start";
while (state !== "end") {
  if (state === "start") { state = await step1(); }
  else if (state === "review") { state = await step2(); }
  else if (state === "approve") { state = "end"; }
}

✅ 简单直接 ❌ 复杂流程代码变成意大利面，难以维护

方案 B：LangGraph 范式（图声明式定义）

const graph = new StateGraph(schema);
graph.addNode("step1", step1Fn);
graph.addNode("step2", step2Fn);
graph.addConditionalEdges("step1", router, { "review": "step2", "end": END });
const app = graph.compile();

✅ 流程可视化，声明式定义 ✅ 状态自动持久化，支持断点续传 ✅ 节点可复用，边逻辑集中管理

方案 C：LangGraph + Checkpointer（推荐）

在方案 B 基础上加入 Checkpointer，每次状态转换都持久化，支持多用户隔离和断点恢复。

✅ 生产级可靠性 ✅ 支持人工介入（暂停等待外部输入）

怎么选择

选择方案 C，用 TypeScript 实现 LangGraph 核心概念：StateGraph + Checkpointer + 条件边。理解设计原理后，可以直接使用 LangGraph.js 库（API 一致）。

怎么执行

StateGraph 核心实现

// src/langgraph/state_graph.ts

export type NodeFunction<S> = (state: S) => Promise<Partial<S>>;
export type ConditionFunction<S> = (state: S) => string;
export const END = "__end__";

export interface GraphConfig {
  threadId?: string;  // 用于多用户隔离
}

export class StateGraph<S extends Record<string, any>> {
  private nodes: Map<string, NodeFunction<S>> = new Map();
  private edges: Map<string, string> = new Map();
  private conditionalEdges: Map<
    string,
    { condition: ConditionFunction<S>; routes: Record<string, string> }
  > = new Map();
  private entryPoint: string = "";
  private schema: S;

  constructor(defaultState: S) {
    this.schema = defaultState;
  }

  addNode(name: string, fn: NodeFunction<S>): this {
    this.nodes.set(name, fn);
    return this;
  }

  addEdge(from: string, to: string): this {
    this.edges.set(from, to);
    return this;
  }

  addConditionalEdges(
    from: string,
    condition: ConditionFunction<S>,
    routes: Record<string, string>
  ): this {
    this.conditionalEdges.set(from, { condition, routes });
    return this;
  }

  setEntryPoint(name: string): this {
    this.entryPoint = name;
    return this;
  }

  compile(checkpointer?: Checkpointer<S>): CompiledGraph<S> {
    return new CompiledGraph(
      this.nodes,
      this.edges,
      this.conditionalEdges,
      this.entryPoint,
      { ...this.schema },
      checkpointer
    );
  }
}

export class CompiledGraph<S extends Record<string, any>> {
  constructor(
    private nodes: Map<string, NodeFunction<S>>,
    private edges: Map<string, string>,
    private conditionalEdges: Map<
      string,
      { condition: ConditionFunction<S>; routes: Record<string, string> }
    >,
    private entryPoint: string,
    private defaultState: S,
    private checkpointer?: Checkpointer<S>
  ) {}

  async invoke(
    initialState: Partial<S>,
    config?: GraphConfig
  ): Promise<S> {
    const threadId = config?.threadId || "default";

    // 恢复已有状态（断点续传）
    let state: S = this.checkpointer?.load(threadId) || {
      ...this.defaultState,
      ...initialState,
    };

    let current = this.entryPoint;
    let steps = 0;
    const maxSteps = 50;

    while (current !== END && steps < maxSteps) {
      steps++;
      console.log(`[Graph] 执行节点: ${current}`);

      const nodeFn = this.nodes.get(current);
      if (!nodeFn) throw new Error(`节点 "${current}" 不存在`);

      // 执行节点，合并状态更新
      const update = await nodeFn(state);
      state = { ...state, ...update };

      // 持久化状态
      this.checkpointer?.save(threadId, state);

      // 确定下一个节点
      if (this.conditionalEdges.has(current)) {
        const { condition, routes } = this.conditionalEdges.get(current)!;
        const routeKey = condition(state);
        current = routes[routeKey] || END;
        console.log(`[Graph] 条件路由: ${routeKey} → ${current}`);
      } else if (this.edges.has(current)) {
        current = this.edges.get(current)!;
      } else {
        current = END;
      }
    }

    if (steps >= maxSteps) {
      console.warn("[Graph] 达到最大步骤数，强制终止");
    }

    return state;
  }

  // 流式执行，每个节点完成后 yield 当前状态
  async *stream(
    initialState: Partial<S>,
    config?: GraphConfig
  ): AsyncGenerator<{ node: string; state: S }> {
    const threadId = config?.threadId || "default";
    let state: S = { ...this.defaultState, ...initialState };
    let current = this.entryPoint;
    let steps = 0;

    while (current !== END && steps < 50) {
      steps++;
      const nodeFn = this.nodes.get(current);
      if (!nodeFn) break;

      const update = await nodeFn(state);
      state = { ...state, ...update };
      this.checkpointer?.save(threadId, state);

      yield { node: current, state: { ...state } };

      if (this.conditionalEdges.has(current)) {
        const { condition, routes } = this.conditionalEdges.get(current)!;
        current = routes[condition(state)] || END;
      } else {
        current = this.edges.get(current) || END;
      }
    }
  }
}

Checkpointer（状态持久化）

// src/langgraph/checkpointer.ts
import * as fs from "fs";

export class Checkpointer<S> {
  private memory: Map<string, S> = new Map();
  private readonly persistPath?: string;

  constructor(persistPath?: string) {
    this.persistPath = persistPath;
    if (persistPath && fs.existsSync(persistPath)) {
      const data = JSON.parse(fs.readFileSync(persistPath, "utf-8"));
      this.memory = new Map(Object.entries(data));
    }
  }

  save(threadId: string, state: S): void {
    this.memory.set(threadId, state);
    if (this.persistPath) {
      fs.writeFileSync(
        this.persistPath,
        JSON.stringify(Object.fromEntries(this.memory), null, 2)
      );
    }
  }

  load(threadId: string): S | undefined {
    return this.memory.get(threadId);
  }

  delete(threadId: string): void {
    this.memory.delete(threadId);
  }

  listThreads(): string[] {
    return Array.from(this.memory.keys());
  }
}

完整示例：代码审查工作流

// src/langgraph/demo.ts
import OpenAI from "openai";
import { StateGraph, END } from "./state_graph";
import { Checkpointer } from "./checkpointer";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// 定义工作流状态（对应 LangGraph 的 TypedDict）
interface ReviewState {
  code: string;
  issues: string[];
  severity: "low" | "medium" | "high" | "critical" | "";
  fixedCode: string;
  approved: boolean;
  attempts: number;
}

const defaultState: ReviewState = {
  code: "",
  issues: [],
  severity: "",
  fixedCode: "",
  approved: false,
  attempts: 0,
};

// 节点1：分析代码
async function analyzeCode(state: ReviewState): Promise<Partial<ReviewState>> {
  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      {
        role: "system",
        content: `分析代码问题。以 JSON 格式输出：
{"issues": ["问题1", "问题2"], "severity": "low/medium/high/critical"}`,
      },
      { role: "user", content: `审查代码：\n${state.code}` },
    ],
    response_format: { type: "json_object" },
    temperature: 0.1,
  });

  const result = JSON.parse(response.choices[0].message.content!);
  console.log(`  发现 ${result.issues.length} 个问题，严重程度：${result.severity}`);

  return {
    issues: result.issues,
    severity: result.severity,
    attempts: state.attempts + 1,
  };
}

// 节点2：修复代码
async function fixCode(state: ReviewState): Promise<Partial<ReviewState>> {
  const issuesList = state.issues.map((i, idx) => `${idx + 1}. ${i}`).join("\n");

  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      { role: "system", content: "你是代码修复专家，修复所有已知问题，不引入新问题。" },
      {
        role: "user",
        content: `修复以下代码中的问题：\n\n原始代码：\n${state.code}\n\n需要修复的问题：\n${issuesList}\n\n只输出修复后的代码，不要解释。`,
      },
    ],
    temperature: 0.1,
  });

  const fixedCode = response.choices[0].message.content || "";
  console.log(`  修复完成，代码长度：${fixedCode.length} 字符`);
  return { fixedCode };
}

// 节点3：人工审批（模拟）
async function humanReview(state: ReviewState): Promise<Partial<ReviewState>> {
  // 实际场景中，这里会暂停等待真实的人工输入
  // 配合 Checkpointer，可以持久化状态，稍后恢复
  console.log(`  [人工审批] 严重问题需要人工确认...`);
  console.log(`  问题列表：${state.issues.slice(0, 2).join("; ")}`);

  // 模拟：严重程度为 critical 时需要人工拒绝或批准
  const autoApprove = state.severity !== "critical";
  console.log(`  决定：${autoApprove ? "批准" : "拒绝"}`);

  return { approved: autoApprove };
}

// 条件路由：根据严重程度决定流程
function routeAfterAnalysis(state: ReviewState): string {
  if (state.issues.length === 0) return "approve";
  if (state.severity === "critical") return "human_review";
  return "fix";
}

function routeAfterFix(state: ReviewState): string {
  // 修复后重新分析（最多 3 次）
  if (state.attempts < 3 && state.issues.length > 0) return "analyze";
  return "approve";
}

function routeAfterHumanReview(state: ReviewState): string {
  return state.approved ? "fix" : "reject";
}

// 终态节点
async function approve(state: ReviewState): Promise<Partial<ReviewState>> {
  console.log("  [批准] 代码通过审查");
  return { approved: true };
}

async function reject(state: ReviewState): Promise<Partial<ReviewState>> {
  console.log("  [拒绝] 代码被人工拒绝，需要重写");
  return { approved: false };
}

// 构建图（对应 LangGraph 的 StateGraph）
const checkpointer = new Checkpointer<ReviewState>("./review_checkpoints.json");

const workflow = new StateGraph(defaultState)
  .addNode("analyze", analyzeCode)
  .addNode("fix", fixCode)
  .addNode("human_review", humanReview)
  .addNode("approve", approve)
  .addNode("reject", reject)
  .setEntryPoint("analyze")
  .addConditionalEdges("analyze", routeAfterAnalysis, {
    approve: "approve",
    fix: "fix",
    human_review: "human_review",
  })
  .addConditionalEdges("fix", routeAfterFix, {
    analyze: "analyze",
    approve: "approve",
  })
  .addConditionalEdges("human_review", routeAfterHumanReview, {
    fix: "fix",
    reject: "reject",
  })
  .addEdge("approve", END)
  .addEdge("reject", END);

const app = workflow.compile(checkpointer);

async function runDemo() {
  const codeToReview = `
function login(username, password) {
  const query = "SELECT * FROM users WHERE username='" + username + "'";
  const result = db.query(query);
  if (result.length > 0 && result[0].password === password) {
    return { token: btoa(username + ":" + password) };
  }
  return null;
}
  `.trim();

  console.log("[LangGraph 演示] 开始代码审查工作流\n");

  // 使用流式执行，实时观察每个节点的输出
  const stream = app.stream(
    { code: codeToReview },
    { threadId: "review-001" }
  );

  for await (const { node, state } of stream) {
    console.log(`\n节点 "${node}" 完成`);
    if (node === "analyze") {
      console.log(`  严重程度: ${state.severity}, 问题数: ${state.issues.length}`);
    }
  }

  console.log("\n=== 工作流完成 ===");
  const checkpoint = checkpointer.load("review-001");
  console.log(`最终状态: 批准=${checkpoint?.approved}, 修复代码长度=${checkpoint?.fixedCode?.length || 0}`);
}

runDemo().catch(console.error);
