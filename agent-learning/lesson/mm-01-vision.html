第1课：视觉理解 — 让 AI 读懂图像

介绍

视觉理解是多模态 AI 的基础。Claude Vision、GPT-4V、Gemini 都支持图像输入，可以识别、分析、理解图像内容。不同 API 的图像输入格式不同（base64 vs URL vs file upload），本课覆盖三种方式的完整代码，以及图像分析、OCR 文字提取、多图对比、图表数据提取的实战实现。

遇到什么问题

不同 API 的图像输入格式不同，直接混用会出错：

  • OpenAI/GPT-4V：content 是数组，图像用 { type: "image_url", image_url: { url } }
  • Anthropic/Claude：content 是数组，图像用 { type: "image", source: { type, media_type, data } }
  • Gemini：parts 数组，图像用 { inlineData: { mimeType, data } } 或 fileData

URL 直接传入只有 OpenAI 支持（公网可访问），Claude 和 Gemini 主要用 base64；私有图像必须用 base64 或文件上传。

有哪些方案

方案 A：URL 直接传入（最简单）

// 仅 OpenAI 支持，图像必须公网可访问
const response = await openai.chat.completions.create({
  model: "gpt-4o",
  messages: [{
    role: "user",
    content: [
      { type: "image_url", image_url: { url: "https://example.com/img.jpg" } },
      { type: "text", text: "描述这张图片" }
    ]
  }]
});

✅ 简单，不增加请求体大小 ❌ 图像必须公网可访问，不支持私有图像 ❌ 不同模型格式不同，切换麻烦

方案 B：base64 内联（推荐）

const base64 = fs.readFileSync("./image.jpg").toString("base64");
// 适用所有模型，统一入口

✅ 不依赖外部存储，适合私有图像 ✅ 可以统一封装，屏蔽各模型差异 ❌ 增大请求体，大图有限制（通常 20MB）

方案 C：文件上传 API（适合大文件）

使用各平台的文件 API 先上传，再用 URI 引用。

✅ 大文件友好，可复用 ❌ 需要额外的上传步骤和文件管理

怎么选择

选择方案 B（统一转 base64）作为主方案。统一的入口让代码整洁，base64 解决了私有图像的鉴权问题，性能差异可通过缓存弥补。对超大图（>5MB）再考虑方案 C。

怎么执行

图像加载工具

// src/vision/image_loader.ts
import * as fs from "fs";
import * as path from "path";

export interface ImageData {
  base64: string;
  mimeType: string;
  size: number;
}

export function loadImageFromFile(filePath: string): ImageData {
  const buffer = fs.readFileSync(filePath);
  const mimeType = getMimeType(filePath);
  return { base64: buffer.toString("base64"), mimeType, size: buffer.length };
}

export async function loadImageFromUrl(url: string): Promise<ImageData> {
  const response = await fetch(url);
  if (!response.ok) throw new Error(`图像下载失败: ${response.status}`);
  const buffer = Buffer.from(await response.arrayBuffer());
  const mimeType = response.headers.get("content-type")?.split(";")[0] || "image/jpeg";
  return { base64: buffer.toString("base64"), mimeType, size: buffer.length };
}

export function getMimeType(filePath: string): string {
  const ext = path.extname(filePath).toLowerCase();
  const types: Record<string, string> = {
    ".jpg": "image/jpeg", ".jpeg": "image/jpeg",
    ".png": "image/png", ".gif": "image/gif", ".webp": "image/webp",
  };
  return types[ext] || "image/jpeg";
}

图像输入三种方式（完整代码）

// src/vision/providers.ts
import OpenAI from "openai";
import Anthropic from "@anthropic-ai/sdk";
import { GoogleGenerativeAI } from "@google/generative-ai";
import type { ImageData } from "./image_loader";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);

// === GPT-4V：URL 或 base64（data URI 格式）===
export async function analyzeWithGPT4V(image: ImageData, prompt: string): Promise<string> {
  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [{
      role: "user",
      content: [
        {
          type: "image_url",
          image_url: {
            url: `data:${image.mimeType};base64,${image.base64}`,
            detail: "high", // "low" 快而粗糙，"high" 慢而精准
          },
        },
        { type: "text", text: prompt },
      ],
    }],
    max_tokens: 1000,
  });
  return response.choices[0].message.content || "";
}

// === Claude Vision：base64，source.type = "base64" ===
export async function analyzeWithClaude(image: ImageData, prompt: string): Promise<string> {
  const response = await anthropic.messages.create({
    model: "claude-3-5-sonnet-20241022",
    max_tokens: 1000,
    messages: [{
      role: "user",
      content: [
        {
          type: "image",
          source: {
            type: "base64",
            media_type: image.mimeType as "image/jpeg" | "image/png" | "image/gif" | "image/webp",
            data: image.base64,
          },
        },
        { type: "text", text: prompt },
      ],
    }],
  });
  const block = response.content[0];
  return block.type === "text" ? block.text : "";
}

// === Gemini Vision：inlineData，支持多图 ===
export async function analyzeWithGemini(image: ImageData, prompt: string): Promise<string> {
  const model = genAI.getGenerativeModel({ model: "gemini-1.5-pro" });
  const result = await model.generateContent([
    { inlineData: { mimeType: image.mimeType, data: image.base64 } },
    prompt,
  ]);
  return result.response.text();
}

图像分析与 OCR 文字提取

// src/vision/analysis.ts
import { loadImageFromFile, loadImageFromUrl } from "./image_loader";
import { analyzeWithGPT4V, analyzeWithClaude, analyzeWithGemini } from "./providers";

export async function analyzeImage(
  imagePath: string,
  prompt: string,
  provider: "openai" | "anthropic" | "gemini" = "openai"
): Promise<string> {
  const image = imagePath.startsWith("http")
    ? await loadImageFromUrl(imagePath)
    : loadImageFromFile(imagePath);

  console.log(`[Vision] ${(image.size / 1024).toFixed(1)}KB, ${image.mimeType}`);

  switch (provider) {
    case "openai":    return analyzeWithGPT4V(image, prompt);
    case "anthropic": return analyzeWithClaude(image, prompt);
    case "gemini":    return analyzeWithGemini(image, prompt);
  }
}

// OCR 文字提取
export async function extractText(imagePath: string): Promise<string> {
  return analyzeImage(
    imagePath,
    `请提取图片中所有可见的文字。
要求：
- 保持原始文字的顺序和格式
- 如果是表格，用 markdown 表格格式输出
- 如果是代码，用代码块格式输出
- 只输出提取的文字，不加额外解释`,
    "openai" // GPT-4o OCR 准确率最高
  );
}

// 多图对比（推荐 Gemini，支持同时输入多张图）
export async function compareImages(imagePaths: string[], question: string): Promise<string> {
  const { GoogleGenerativeAI } = await import("@google/generative-ai");
  const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);
  const model = genAI.getGenerativeModel({ model: "gemini-1.5-pro" });

  const imageParts = await Promise.all(
    imagePaths.map(async (p) => {
      const img = p.startsWith("http")
        ? await loadImageFromUrl(p)
        : loadImageFromFile(p);
      return { inlineData: { mimeType: img.mimeType, data: img.base64 } };
    })
  );

  const result = await model.generateContent([...imageParts, question]);
  return result.response.text();
}

// 图表数据提取（结构化输出）
export async function extractChartData(imagePath: string): Promise<Record<string, any>> {
  const result = await analyzeImage(
    imagePath,
    `请分析这个图表并以 JSON 格式返回数据：
{
  "chart_type": "bar/line/pie/scatter 等",
  "title": "图表标题",
  "x_axis_label": "X轴标签",
  "y_axis_label": "Y轴标签",
  "data_series": [{"name": "系列名", "data": [{"label": "...", "value": 数字}]}],
  "key_insights": ["关键洞察1", "关键洞察2"]
}
只返回 JSON，不加其他文字。`
  );

  try {
    const jsonMatch = result.match(/\{[\s\S]*\}/);
    return jsonMatch ? JSON.parse(jsonMatch[0]) : { raw: result };
  } catch {
    return { raw: result };
  }
}

// src/vision/demo.ts
import { analyzeImage, extractText, compareImages, extractChartData } from "./analysis";

async function runDemo() {
  const sampleUrl =
    "https://upload.wikimedia.org/wikipedia/commons/thumb/3/3a/Cat03.jpg/320px-Cat03.jpg";

  // 演示1：基础图像分析
  console.log("=== 图像分析 ===");
  const desc = await analyzeImage(sampleUrl, "详细描述图片内容，包括颜色、姿态等细节");

  // 演示2：多模型对比同一张图

    analyzeImage(sampleUrl, question, "openai"),
    analyzeImage(sampleUrl, question, "gemini"),
  ]);

  console.log("Gemini:", geminiResult);

  console.log("\n=== 多图对比 ===");

}

runDemo().catch(console.error);
