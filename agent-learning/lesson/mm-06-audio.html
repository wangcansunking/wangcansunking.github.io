第6课：音频处理 — 语音识别 + 文字转语音

介绍

音频是多模态的重要维度。Whisper 做语音识别（STT），OpenAI TTS / MiniMax TTS 做文字转语音（TTS），构建语音 Agent。音频处理涉及格式转换、流式处理、实时性要求，比文本 API 复杂。本课覆盖 Whisper API（转录、翻译）、OpenAI TTS（6种声音、速度调节）、实时语音对话 Pipeline（录音→STT→LLM→TTS→播放）。

遇到什么问题

音频处理涉及格式转换、流式处理、实时性要求，比文本 API 复杂：

  1. Whisper API 文件限制：最大 25MB，超过需要先压缩或分片
  2. 音频格式转换：录音设备可能输出 WAV/OGG，API 支持的格式不同
  3. TTS 流式播放：等全部生成完再播放延迟高，需要边生成边播放
  4. 实时对话延迟：STT + LLM + TTS 三个串行调用，需要优化流水线

有哪些方案

方案 A：OpenAI Whisper + OpenAI TTS（全 OpenAI 方案）

// STT
const transcript = await openai.audio.transcriptions.create({
  file: fs.createReadStream("audio.mp3"),
  model: "whisper-1",
});
// TTS
const speech = await openai.audio.speech.create({
  model: "tts-1-hd",
  voice: "nova",
  input: text,
});

✅ 统一 SDK，API 风格一致 ✅ Whisper 是业界最佳 STT 模型 ❌ TTS 声音选择少（6种），中文效果一般

方案 B：OpenAI Whisper + MiniMax TTS

✅ 中文 TTS 效果更好（MiniMax） ✅ Whisper STT 准确率最高 ❌ 需要两套 SDK

方案 C：Azure Speech Services（企业方案）

✅ 低延迟实时流式 STT，企业级 SLA ❌ 配置复杂，成本较高

怎么选择

英文场景用方案 A；中文语音应用用方案 B（Whisper STT + MiniMax TTS 中文效果更自然）。本课同时实现两种 TTS，重点覆盖完整的语音对话 Pipeline。

怎么执行

Whisper API（语音识别、翻译）

// src/audio/whisper.ts
import OpenAI from "openai";
import * as fs from "fs";
import * as path from "path";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// 支持的格式：flac, mp3, mp4, mpeg, mpga, m4a, ogg, wav, webm
export async function transcribe(
  audioPath: string,
  options?: {
    language?: string;      // "zh" 中文，"en" 英文，不填自动检测
    prompt?: string;        // 提示词，帮助识别专有名词
    responseFormat?: "json" | "text" | "srt" | "vtt" | "verbose_json";
  }
): Promise<{ text: string; language?: string; duration?: number; segments?: any[] }> {
  const fileStats = fs.statSync(audioPath);
  const fileSizeMB = fileStats.size / 1024 / 1024;

  if (fileSizeMB > 25) {
    console.warn(`文件 ${fileSizeMB.toFixed(1)}MB 超过 25MB 限制，建议先压缩`);
  }

  const fileStream = fs.createReadStream(audioPath);

  const response = await openai.audio.transcriptions.create({
    file: fileStream as any,
    model: "whisper-1",
    language: options?.language,
    prompt: options?.prompt,
    response_format: (options?.responseFormat || "verbose_json") as any,
    timestamp_granularities: ["segment"],
  });

  if (typeof response === "string") {
    return { text: response };
  }

  return {
    text: (response as any).text,
    language: (response as any).language,
    duration: (response as any).duration,
    segments: (response as any).segments,
  };
}

// 语音翻译（将任意语言翻译为英文）
export async function translate(audioPath: string): Promise<string> {
  const fileStream = fs.createReadStream(audioPath);

  const response = await openai.audio.translations.create({
    file: fileStream as any,
    model: "whisper-1",
    response_format: "text",
  });

  return response as unknown as string;
}

// 大文件处理：先压缩再转录
export async function transcribeLargeFile(audioPath: string): Promise<string> {
  const { exec } = await import("child_process");
  const { promisify } = await import("util");
  const execAsync = promisify(exec);

  const compressedPath = audioPath.replace(/\.[^.]+$/, "_compressed.mp3");

  // 用 ffmpeg 压缩音频：降码率到 64kbps
  await execAsync(
    `ffmpeg -i "${audioPath}" -b:a 64k -ar 16000 -ac 1 -y "${compressedPath}"`
  );

  console.log(`音频已压缩: ${compressedPath}`);
  const result = await transcribe(compressedPath);

  // 清理压缩文件
  fs.unlinkSync(compressedPath);

  return result.text;
}

// 带时间戳的字幕生成
export async function generateSubtitles(
  audioPath: string,
  format: "srt" | "vtt" = "srt"
): Promise<string> {
  const fileStream = fs.createReadStream(audioPath);

  const response = await openai.audio.transcriptions.create({
    file: fileStream as any,
    model: "whisper-1",
    response_format: format,
  });

  return response as unknown as string;
}

OpenAI TTS（6种声音、速度调节）

// src/audio/openai_tts.ts
import OpenAI from "openai";
import * as fs from "fs";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// OpenAI TTS 的 6 种声音特点：
// alloy  - 中性，清晰，适合新闻播报
// echo   - 男声，深沉
// fable  - 英式口音，叙事感强
// onyx   - 男声，权威感强
// nova   - 女声，活泼，适合助手
// shimmer - 女声，温柔

export type OpenAIVoice = "alloy" | "echo" | "fable" | "onyx" | "nova" | "shimmer";

export async function textToSpeech(
  text: string,
  outputPath: string,
  options?: {
    voice?: OpenAIVoice;
    model?: "tts-1" | "tts-1-hd"; // tts-1 快，tts-1-hd 高质量
    speed?: number;                // 0.25 - 4.0，默认 1.0
    format?: "mp3" | "opus" | "aac" | "flac" | "wav" | "pcm";
  }
): Promise<Buffer> {
  const response = await openai.audio.speech.create({
    model: options?.model || "tts-1-hd",
    voice: options?.voice || "nova",
    input: text,
    speed: options?.speed || 1.0,
    response_format: options?.format || "mp3",
  });

  const buffer = Buffer.from(await response.arrayBuffer());
  fs.writeFileSync(outputPath, buffer);
  console.log(`语音已保存: ${outputPath} (${(buffer.length / 1024).toFixed(1)}KB)`);
  return buffer;
}

// 流式 TTS（边生成边播放，降低首字节延迟）
export async function streamingTTS(
  text: string,
  outputPath: string,
  onChunk?: (chunk: Buffer) => void
): Promise<void> {
  const response = await openai.audio.speech.create({
    model: "tts-1",
    voice: "nova",
    input: text,
    response_format: "mp3",
  });

  const writeStream = fs.createWriteStream(outputPath);
  const reader = (response.body as any).getReader();

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    const chunk = Buffer.from(value);
    writeStream.write(chunk);
    onChunk?.(chunk);
  }

  writeStream.end();
  console.log(`流式音频已保存: ${outputPath}`);
}

// 长文本分段 TTS（超过 4096 字需要分段）
export async function longTextToSpeech(
  text: string,
  outputDir: string,
  voice: OpenAIVoice = "nova"
): Promise<string[]> {
  const MAX_CHARS = 4000; // 保留一些余量
  const segments: string[] = [];

  // 按句子分割
  const sentences = text.split(/([。！？.!?])/);
  let current = "";

  for (let i = 0; i < sentences.length; i += 2) {
    const sentence = sentences[i] + (sentences[i + 1] || "");
    if (current.length + sentence.length > MAX_CHARS) {
      if (current) segments.push(current);
      current = sentence;
    } else {
      current += sentence;
    }
  }
  if (current) segments.push(current);

  const outputFiles: string[] = [];
  for (let i = 0; i < segments.length; i++) {
    const outputPath = `${outputDir}/segment_${i + 1}.mp3`;
    await textToSpeech(segments[i], outputPath, { voice });
    outputFiles.push(outputPath);
    console.log(`分段 ${i + 1}/${segments.length} 完成`);
  }

  return outputFiles;
}

实时语音对话 Pipeline

// src/audio/voice_pipeline.ts
import { transcribe } from "./whisper";
import { textToSpeech, streamingTTS } from "./openai_tts";
import OpenAI from "openai";
import * as fs from "fs";
import * as path from "path";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

interface ConversationTurn {
  role: "user" | "assistant";
  text: string;
  audioPath?: string;
}

// 完整语音对话 Pipeline
export class VoicePipeline {
  private conversationHistory: ConversationTurn[] = [];
  private outputDir: string;

  constructor(outputDir: string = "./voice_output") {
    this.outputDir = outputDir;
    if (!fs.existsSync(outputDir)) {
      fs.mkdirSync(outputDir, { recursive: true });
    }
  }

  // 处理一轮语音对话
  async processVoiceTurn(audioInputPath: string): Promise<{
    userText: string;
    assistantText: string;
    audioOutputPath: string;
  }> {
    const turnId = Date.now();

    // 步骤1：STT - 语音转文字
    console.log("[1/3] 语音识别中...");
    const sttStart = Date.now();
    const { text: userText } = await transcribe(audioInputPath, { language: "zh" });
    console.log(`    识别完成 (${Date.now() - sttStart}ms): "${userText}"`);

    this.conversationHistory.push({ role: "user", text: userText, audioPath: audioInputPath });

    // 步骤2：LLM - 生成回复
    console.log("[2/3] LLM 生成回复中...");
    const llmStart = Date.now();
    const messages = this.conversationHistory.map((turn) => ({
      role: turn.role,
      content: turn.text,
    }));

    const response = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        {
          role: "system",
          content: "你是一个语音助手。回复简洁，控制在 100 字以内，适合语音播报。不使用 Markdown 格式。",
        },
        ...messages,
      ],
      max_tokens: 200,
    });

    const assistantText = response.choices[0].message.content || "";
    console.log(`    生成完成 (${Date.now() - llmStart}ms): "${assistantText}"`);

    // 步骤3：TTS - 文字转语音
    console.log("[3/3] 语音合成中...");
    const ttsStart = Date.now();
    const audioOutputPath = path.join(this.outputDir, `reply_${turnId}.mp3`);
    await textToSpeech(assistantText, audioOutputPath, { voice: "nova", speed: 1.1 });
    console.log(`    合成完成 (${Date.now() - ttsStart}ms)`);

    this.conversationHistory.push({
      role: "assistant",
      text: assistantText,
      audioPath: audioOutputPath,
    });

    return { userText, assistantText, audioOutputPath };
  }

  getHistory(): ConversationTurn[] {
    return this.conversationHistory;
  }

  clearHistory(): void {
    this.conversationHistory = [];
  }
}

// 模拟完整对话演示
export async function runVoiceDemo() {
  const pipeline = new VoicePipeline("./voice_demo");

  // 模拟：将文字转成"用户语音"，再走完整 pipeline
  const userInputs = [
    "你好，请介绍一下你自己",
    "我想学习 AI 开发，从哪里开始好呢？",
  ];

  for (const input of userInputs) {
    // 先把用户文字转成音频（模拟录音）
    const userAudioPath = `./voice_demo/user_input_${Date.now()}.mp3`;
    await textToSpeech(input, userAudioPath, { voice: "echo" });

    console.log(`\n=== 用户说：${input} ===`);
    const result = await pipeline.processVoiceTurn(userAudioPath);
    console.log(`助手回复：${result.assistantText}`);
    console.log(`音频输出：${result.audioOutputPath}`);
  }
}

音频格式处理

// src/audio/format_utils.ts
import { exec } from "child_process";
import { promisify } from "util";

const execAsync = promisify(exec);

// 音频格式转换（需要 ffmpeg）
export async function convertAudio(
  inputPath: string,
  outputPath: string,
  options?: { bitrate?: string; sampleRate?: number; channels?: number }
): Promise<void> {
  const { bitrate = "64k", sampleRate = 16000, channels = 1 } = options || {};
  await execAsync(
    `ffmpeg -i "${inputPath}" -b:a ${bitrate} -ar ${sampleRate} -ac ${channels} -y "${outputPath}"`
  );
  console.log(`音频已转换: ${outputPath}`);
}

// 合并多个音频文件
export async function mergeAudioFiles(
  inputPaths: string[],
  outputPath: string
): Promise<void> {
  const listFile = `./concat_list_${Date.now()}.txt`;
  const listContent = inputPaths.map((p) => `file '${p}'`).join("\n");
  require("fs").writeFileSync(listFile, listContent);

  await execAsync(`ffmpeg -f concat -safe 0 -i "${listFile}" -c copy -y "${outputPath}"`);
  require("fs").unlinkSync(listFile);
  console.log(`音频已合并: ${outputPath}`);
}
