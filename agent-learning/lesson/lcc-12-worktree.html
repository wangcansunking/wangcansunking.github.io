第12课：Worktree Isolation — Git 工作树安全并行

介绍

多个 Agent 并发修改同一 Git 仓库会产生冲突。Git Worktree 让每个 Agent 在独立目录工作（独立分支），完成后提交并自动合并。这是 Claude Code 本身用来并行处理多任务的底层机制。

源码参考：github.com/shareAI-lab/learn-claude-code — agents/s12_worktree_task_isolation.py

遇到什么问题

多 Agent 共享工作目录时：

Agent-1: edit src/api.py → 修改第 42 行
Agent-2: edit src/api.py → 同时修改第 42 行
结果：冲突，其中一个 Agent 的修改丢失

Agent-1: git checkout main → 影响 Agent-2 正在编辑的文件
Agent-2: ???

有哪些方案

方案 A：文件锁

每个文件只允许一个 Agent 编辑。

✅ 防冲突 ❌ 阻塞并行，锁死频繁，性能差

方案 B：复制仓库

每个 Agent 完整克隆一份仓库。

✅ 完全隔离 ❌ 磁盘浪费（仓库很大时几GB），克隆耗时

方案 C：Git Worktree（推荐）

git worktree add .worktrees/agent-1-branch agent-1-branch
git worktree add .worktrees/agent-2-branch agent-2-branch

✅ 每个 worktree 是独立目录 + 独立分支 ✅ 共享 .git 对象库，无额外磁盘消耗 ✅ 并行修改同一文件不冲突 ✅ 完成后 git merge 自动合并

怎么选择

方案 C。Git Worktree 是专为此场景设计的原生 Git 功能，零额外依赖，Claude Code 本身就用这个机制实现 Agent 并行。

怎么执行

Worktree 管理器

# agents/s12_worktree_task_isolation.py
import subprocess
import shutil
from pathlib import Path


class WorktreeManager:
    def __init__(self, repo_path: str = "."):
        self.repo = Path(repo_path).resolve()
        self.worktrees_dir = self.repo / ".worktrees"
        self.worktrees_dir.mkdir(exist_ok=True)

    def create(self, branch_name: str) -> str:
        """
        创建新 worktree。
        返回 worktree 路径（Agent 在此路径工作）。
        """
        wt_path = self.worktrees_dir / branch_name

        # 如果分支不存在则创建
        result = subprocess.run(
            ["git", "show-ref", "--verify", f"refs/heads/{branch_name}"],
            cwd=self.repo, capture_output=True,
        )
        if result.returncode != 0:
            subprocess.run(
                ["git", "branch", branch_name],
                cwd=self.repo, check=True,
            )

        # 添加 worktree
        subprocess.run(
            ["git", "worktree", "add", str(wt_path), branch_name],
            cwd=self.repo, check=True,
        )
        return str(wt_path)

    def merge(self, branch_name: str, target: str = "main") -> str:
        """
        将 worktree 分支合并回目标分支。
        返回合并结果描述。
        """
        # 先在 worktree 内提交未提交的变更
        wt_path = self.worktrees_dir / branch_name
        subprocess.run(
            ["git", "add", "-A"], cwd=wt_path, check=True,
        )
        subprocess.run(
            ["git", "commit", "-m", f"agent: complete task on {branch_name}",
             "--allow-empty"],
            cwd=wt_path,
        )

        # 切到目标分支合并
        subprocess.run(
            ["git", "checkout", target], cwd=self.repo, check=True,
        )
        result = subprocess.run(
            ["git", "merge", "--no-ff", branch_name,
             "-m", f"merge: {branch_name} into {target}"],
            cwd=self.repo, capture_output=True, text=True,
        )
        if result.returncode != 0:
            return f"MERGE CONFLICT: {result.stderr[:300]}"
        return f"Merged {branch_name} → {target}"

    def remove(self, branch_name: str, delete_branch: bool = True):
        """清理 worktree（任务完成后调用）"""
        subprocess.run(
            ["git", "worktree", "remove",
             str(self.worktrees_dir / branch_name), "--force"],
            cwd=self.repo,
        )
        if delete_branch:
            subprocess.run(
                ["git", "branch", "-d", branch_name], cwd=self.repo,
            )

    def list_worktrees(self) -> str:
        result = subprocess.run(
            ["git", "worktree", "list"], cwd=self.repo,
            capture_output=True, text=True,
        )
        return result.stdout

带 Worktree 的 Agent

class IsolatedAgent:
    def __init__(self, agent_id: str, wt_manager: WorktreeManager):
        self.agent_id = agent_id
        self.wt_manager = wt_manager
        self.branch = f"agent-{agent_id}-{int(time.time())}"
        self.workdir = None

    def __enter__(self):
        """进入：创建独立 worktree"""
        self.workdir = self.wt_manager.create(self.branch)
        print(f"[{self.agent_id}] Working in: {self.workdir}")
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        """退出：合并并清理"""
        if exc_type is None:
            result = self.wt_manager.merge(self.branch)
            print(f"[{self.agent_id}] {result}")
        self.wt_manager.remove(self.branch)

    def run_task(self, task: str) -> str:
        """在独立 worktree 中执行任务"""
        # Agent 的所有文件操作都在 self.workdir 进行
        # 与其他 Agent 完全隔离
        return run_subagent(
            task_description=task,
            workdir=self.workdir,  # ← 关键：使用 worktree 路径
        )

并行执行多 Agent

import concurrent.futures


def run_parallel_agents(tasks: list[dict], repo_path: str = "."):
    """并行运行多个 Agent，每个使用独立 worktree"""
    wt_manager = WorktreeManager(repo_path)
    results = {}

    def run_one(task):
        agent_id = task["id"]
        with IsolatedAgent(agent_id, wt_manager) as agent:
            return agent.run_task(task["description"])

    with concurrent.futures.ThreadPoolExecutor(max_workers=4) as executor:
        future_to_task = {
            executor.submit(run_one, task): task for task in tasks
        }
        for future in concurrent.futures.as_completed(future_to_task):
            task = future_to_task[future]
            results[task["id"]] = future.result()

    return results


# 使用示例：3 个 Agent 并行处理不同功能
run_parallel_agents([
    {"id": "feat-auth",   "description": "Implement OAuth2 login"},
    {"id": "feat-search", "description": "Add full-text search to /posts"},
    {"id": "fix-perf",    "description": "Fix N+1 query in user dashboard"},
])
