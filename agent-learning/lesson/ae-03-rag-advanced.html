第3课：高级 RAG — 混合检索 + 重排序

介绍

基础 RAG 的召回率和精确率有限。高级 RAG 通过混合检索、查询扩展、重排序（Reranking）大幅提升效果。本课实现 BM25 关键词检索、RRF（Reciprocal Rank Fusion）融合、Cohere Rerank API 重排序、HyDE 查询扩展，以及 MRR/NDCG 评估指标。

遇到什么问题

单纯向量检索对关键词查询效果差；BM25 对语义相似不敏感。两者互补，混合使用效果最好：

flowchart TD
    A[用户查询] --> B[查询扩展\nHyDE 假设文档]
    B --> C1[向量检索\n语义相似度]
    B --> C2[BM25 检索\n关键词匹配]
    C1 --> D[RRF 融合\n两路结果合并]
    C2 --> D
    D --> E[Reranker 精排\nCohere/BGE]
    E --> F[Top-K 最终结果]
    F --> G[注入 Prompt → LLM]

有哪些方案

方案 A：仅向量检索（基础 RAG）

✅ 语义理解强 ❌ 精确关键词（产品型号、人名）检索效果差

方案 B：仅 BM25 检索

✅ 精确关键词匹配快 ❌ 不理解语义，同义词无法匹配

方案 C：混合检索 + Reranker（推荐）

const dense = await vectorSearch(queryEmbedding, topK=20);
const sparse = bm25Search(query, topK=20);
const merged = rrfFusion(dense, sparse);
const reranked = await cohereRerank(query, merged, topK=5);

✅ 召回率最高（两路互补） ✅ Reranker 精排消除噪声 ❌ 多一步 Reranker API 调用（增加延迟约 200-500ms）

怎么选择

生产环境 RAG统一用方案 C。两路检索覆盖语义和关键词，Reranker 精排提升最终精度。HyDE 查询扩展对复杂问题有额外收益（提升约 10-15% 召回率），但增加一次 LLM 调用，可以根据延迟要求选择性使用。

怎么执行

BM25 实现

// src/rag_advanced/bm25.ts

interface BM25Document {
  id: string;
  text: string;
  metadata?: Record<string, any>;
}

export class BM25Index {
  private documents: BM25Document[] = [];
  private termFrequencies: Map<string, Map<string, number>> = new Map();
  private documentFrequencies: Map<string, number> = new Map();
  private avgDocLength = 0;

  private readonly k1 = 1.5;  // 词频饱和参数
  private readonly b = 0.75;  // 文档长度归一化参数

  add(document: BM25Document): void {
    this.documents.push(document);
    const terms = this.tokenize(document.text);

    // 计算词频
    const tf = new Map<string, number>();
    for (const term of terms) {
      tf.set(term, (tf.get(term) || 0) + 1);
    }
    this.termFrequencies.set(document.id, tf);

    // 更新文档频率
    for (const term of new Set(terms)) {
      this.documentFrequencies.set(term, (this.documentFrequencies.get(term) || 0) + 1);
    }

    // 更新平均文档长度
    const totalLength = this.documents.reduce(
      (sum, doc) => sum + this.tokenize(doc.text).length, 0
    );
    this.avgDocLength = totalLength / this.documents.length;
  }

  search(query: string, topK: number = 10): { document: BM25Document; score: number }[] {
    const queryTerms = this.tokenize(query);
    const N = this.documents.length;
    const scores = new Map<string, number>();

    for (const term of queryTerms) {
      const df = this.documentFrequencies.get(term) || 0;
      if (df === 0) continue;

      // IDF 计算
      const idf = Math.log((N - df + 0.5) / (df + 0.5) + 1);

      for (const doc of this.documents) {
        const tf = this.termFrequencies.get(doc.id)?.get(term) || 0;
        const docLength = this.tokenize(doc.text).length;

        // BM25 分数
        const normalizedTF =
          (tf * (this.k1 + 1)) /
          (tf + this.k1 * (1 - this.b + this.b * (docLength / this.avgDocLength)));

        scores.set(doc.id, (scores.get(doc.id) || 0) + idf * normalizedTF);
      }
    }

    return this.documents
      .map((doc) => ({ document: doc, score: scores.get(doc.id) || 0 }))
      .filter((r) => r.score > 0)
      .sort((a, b) => b.score - a.score)
      .slice(0, topK);
  }

  private tokenize(text: string): string[] {
    // 简单分词：中英文混合处理
    return text
      .toLowerCase()
      .replace(/[^\u4e00-\u9fa5a-z0-9\s]/g, " ")
      .split(/[\s]+/)
      .filter((t) => t.length > 0);
  }
}

RRF（Reciprocal Rank Fusion）融合

// src/rag_advanced/rrf.ts

interface RankedResult {
  id: string;
  text: string;
  metadata?: Record<string, any>;
  score?: number;
}

// RRF 公式：score = Σ(1 / (k + rank_i))
// k=60 是经验值，平衡高排名和低排名的影响
export function rrfFusion(
  results1: { id: string; text: string; metadata?: any }[],
  results2: { id: string; text: string; metadata?: any }[],
  k: number = 60
): RankedResult[] {
  const scores = new Map<string, number>();
  const docs = new Map<string, { text: string; metadata?: any }>();

  // 计算第一路结果的 RRF 分数
  results1.forEach((r, rank) => {
    scores.set(r.id, (scores.get(r.id) || 0) + 1 / (k + rank + 1));
    docs.set(r.id, { text: r.text, metadata: r.metadata });
  });

  // 计算第二路结果的 RRF 分数（与第一路相加）
  results2.forEach((r, rank) => {
    scores.set(r.id, (scores.get(r.id) || 0) + 1 / (k + rank + 1));
    docs.set(r.id, { text: r.text, metadata: r.metadata });
  });

  // 按 RRF 分数排序
  return Array.from(scores.entries())
    .sort((a, b) => b[1] - a[1])
    .map(([id, score]) => ({
      id,
      text: docs.get(id)!.text,
      metadata: docs.get(id)!.metadata,
      score,
    }));
}

Cohere Rerank API

// src/rag_advanced/reranker.ts

interface RerankResult {
  text: string;
  metadata?: Record<string, any>;
  relevanceScore: number;
  originalIndex: number;
}

// Cohere Rerank：精排 Top-K 结果
export async function cohereRerank(
  query: string,
  documents: { text: string; metadata?: any }[],
  topK: number = 5
): Promise<RerankResult[]> {
  const response = await fetch("https://api.cohere.ai/v1/rerank", {
    method: "POST",
    headers: {
      Authorization: `Bearer ${process.env.COHERE_API_KEY}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      model: "rerank-multilingual-v3.0", // 支持中文
      query,
      documents: documents.map((d) => d.text),
      top_n: topK,
      return_documents: true,
    }),
  });

  if (!response.ok) throw new Error(`Cohere Rerank 失败: ${response.statusText}`);

  const data = await response.json();

  return data.results.map((r: any) => ({
    text: documents[r.index].text,
    metadata: documents[r.index].metadata,
    relevanceScore: r.relevance_score,
    originalIndex: r.index,
  }));
}

HyDE 查询扩展

// src/rag_advanced/hyde.ts
import OpenAI from "openai";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// HyDE (Hypothetical Document Embeddings)
// 先让 LLM 生成一个假设性的答案文档，用这个文档的 embedding 来检索
// 原理：假设答案比问题更接近知识库中的答案文档

export async function hydeQueryExpansion(question: string): Promise<string> {
  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{
      role: "user",
      content: `请为以下问题生成一个假设性的答案段落（100-150字）。
不需要完全准确，只需要听起来像这个问题的合理答案：

问题：${question}

请直接输出答案段落，不要说"假设"或"可能"这样的词。`,
    }],
    max_tokens: 200,
    temperature: 0.3,
  });

  return response.choices[0].message.content || question;
}

评估指标（MRR、NDCG）

// src/rag_advanced/evaluation.ts

// MRR（Mean Reciprocal Rank）：越高越好，最大值 1.0
export function calculateMRR(
  queries: string[],
  retrievedIds: string[][],   // 每个查询检索到的 ID 列表
  relevantIds: string[][]     // 每个查询的相关 ID
): number {
  const reciprocalRanks = queries.map((_, i) => {
    const retrieved = retrievedIds[i];
    const relevant = new Set(relevantIds[i]);

    for (let rank = 0; rank < retrieved.length; rank++) {
      if (relevant.has(retrieved[rank])) {
        return 1 / (rank + 1);
      }
    }
    return 0;
  });

  return reciprocalRanks.reduce((sum, rr) => sum + rr, 0) / queries.length;
}

// NDCG（Normalized Discounted Cumulative Gain）
export function calculateNDCG(
  retrievedIds: string[],
  relevantIds: string[],
  k: number = 10
): number {
  const relevantSet = new Set(relevantIds);

  function dcg(ids: string[]): number {
    return ids.slice(0, k).reduce((sum, id, i) => {
      const relevance = relevantSet.has(id) ? 1 : 0;
      return sum + relevance / Math.log2(i + 2);
    }, 0);
  }

  const idealOrder = [...relevantIds, ...retrievedIds.filter((id) => !relevantSet.has(id))];

  const dcgScore = dcg(retrievedIds);
  const idcgScore = dcg(idealOrder);

  return idcgScore > 0 ? dcgScore / idcgScore : 0;
}

完整高级 RAG Pipeline

// src/rag_advanced/advanced_rag.ts
import OpenAI from "openai";
import { BM25Index } from "./bm25";
import { rrfFusion } from "./rrf";
import { cohereRerank } from "./reranker";
import { hydeQueryExpansion } from "./hyde";
import { getEmbedding, batchEmbeddings } from "../rag/embeddings";
import { SimpleVectorStore } from "../rag/vector_store";
import { recursiveChunking } from "../rag/chunking";
import * as fs from "fs";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export class AdvancedRAGPipeline {
  private vectorStore = new SimpleVectorStore();
  private bm25Index = new BM25Index();

  async indexDocument(text: string, metadata?: Record<string, any>): Promise<void> {
    const chunks = recursiveChunking(text, 600);
    const embeddings = await batchEmbeddings(chunks.map((c) => c.text));

    for (let i = 0; i < chunks.length; i++) {
      const id = `chunk_${Date.now()}_${i}`;
      const metadata_full = { ...metadata, chunkIndex: i };

      // 同时索引到向量库和 BM25
      this.vectorStore.add({ id, text: chunks[i].text, embedding: embeddings[i], metadata: metadata_full });
      this.bm25Index.add({ id, text: chunks[i].text, metadata: metadata_full });
    }
  }

  async query(question: string, useHyde: boolean = false): Promise<string> {
    // 1. 查询扩展（可选）
    const searchQuery = useHyde
      ? await hydeQueryExpansion(question)
      : question;

    // 2. 并行双路检索
    const [queryEmbedding, bm25Results] = await Promise.all([
      getEmbedding(searchQuery),
      Promise.resolve(this.bm25Index.search(question, 20)), // BM25 用原始问题
    ]);

    const vectorResults = this.vectorStore.search(queryEmbedding, 20);

    // 3. RRF 融合
    const merged = rrfFusion(
      vectorResults.map((r) => ({ id: r.record.id, text: r.record.text, metadata: r.record.metadata })),
      bm25Results.map((r) => ({ id: r.document.id, text: r.document.text, metadata: r.document.metadata }))
    );

    // 4. Cohere Reranker 精排
    const reranked = await cohereRerank(question, merged.slice(0, 20), 5);

    console.log(`[AdvancedRAG] 向量: ${vectorResults.length}, BM25: ${bm25Results.length}, 融合: ${merged.length}, 精排后: ${reranked.length}`);

    // 5. 注入 Prompt
    const context = reranked.map((r, i) => `[${i + 1}] ${r.text}`).join("\n\n");

    const response = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        { role: "system", content: `基于以下上下文回答，不编造：\n\n${context}` },
        { role: "user", content: question },
      ],
      temperature: 0.1,
    });

    return response.choices[0].message.content || "";
  }
}
