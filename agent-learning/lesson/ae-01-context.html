第1课：Context 管理 — 永不丢失重要信息

介绍

上下文窗口是 LLM 的"短期记忆"，超出就截断。Context 管理是构建长期运行 Agent 的核心挑战。用户长时间对话后，早期重要信息被挤出上下文；大任务的工具调用结果撑爆 context。本课构建完整的 ContextManager，实现分层存储：Working Memory 保留近期对话，摘要压缩保留早期要点，重要实体永久存储。

遇到什么问题

用户长时间对话后，早期重要信息被挤出上下文；大任务的工具调用结果撑爆 context：

上下文窗口的压力——新内容不断加入，旧内容被挤出：

flowchart LR
    A[System Prompt\n1K tokens] --> B[早期对话\n用户身份/偏好\n20K tokens]
    B --> C[中期对话\n决策/需求\n30K tokens]
    C --> D[工具调用结果\n15K tokens]
    D --> E[当前消息\n2K tokens]
    E --> F{超出上下文限制?}
    F -->|是| G[截断早期内容\n信息丢失!]
    F -->|否| H[正常处理]
    style G fill:#ff6b6b,color:#fff

有哪些方案

方案 A：固定窗口（只保留最近 N 轮）

const recentMessages = messages.slice(-10);

✅ 简单，永不超出 context 限制 ❌ 丢失早期重要信息（用户偏好、关键决策）

方案 B：摘要压缩（定期用 LLM 摘要早期对话）

if (tokenCount > threshold) {
  const summary = await summarize(earlyMessages);
  messages = [summaryMessage, ...recentMessages];
}

✅ 保留要点，大幅压缩 token 数 ❌ 摘要有损，细节会丢失；需要额外 LLM 调用

方案 C：向量检索（把历史存向量库，按需检索）

await vectorStore.add(message);
const relevant = await vectorStore.search(currentQuery, topK=3);

✅ 最完整，可检索到任意历史信息 ❌ 需要向量数据库；检索可能遗漏信息

方案 D：分层存储（Working Memory + Long-term Memory）

结合摘要压缩和实体提取，将重要信息持久化到 Long-term Memory。

✅ 生产级方案，平衡完整性和效率 ✅ 重要信息（用户身份、决策）永不丢失

怎么选择

选择方案 D（分层存储）作为主方案，同时展示方案 B 的核心实现。设计如下：

  • Working Memory：保留近期 N 条完整对话
  • Summary Layer：定期压缩早期对话为摘要注入 system prompt
  • Long-term Memory：提取关键实体（用户信息、决策、偏好）永久保存

怎么执行

Token 计数

// src/context/token_counter.ts
export function countTokens(text: string): number {
  try {
    const { encoding_for_model } = require("tiktoken");
    const enc = encoding_for_model("gpt-4o");
    const count = enc.encode(text).length;
    enc.free();
    return count;
  } catch {
    // 回退：粗略估算
    const chineseChars = (text.match(/[\u4e00-\u9fa5]/g) || []).length;
    return Math.ceil(chineseChars / 1.5 + (text.length - chineseChars) / 4);
  }
}

export function countMessagesTokens(
  messages: { role: string; content: string }[]
): number {
  return messages.reduce((total, msg) => total + countTokens(msg.content) + 4, 0);
}

摘要压缩

// src/context/summarizer.ts
import OpenAI from "openai";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export interface Message {
  role: "user" | "assistant" | "system";
  content: string;
  timestamp?: number;
}

export async function summarizeMessages(messages: Message[]): Promise<string> {
  const text = messages
    .filter((m) => m.role !== "system")
    .map((m) => `${m.role === "user" ? "用户" : "助手"}: ${m.content}`)
    .join("\n");

  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{
      role: "user",
      content: `请压缩以下对话，保留所有关键信息（用户身份、重要决策、明确需求、约束条件）：

${text}

格式：
- 对话摘要（2-3句）
- 关键信息：[列表]
- 未解决事项：[列表]`,
    }],
    max_tokens: 500,
    temperature: 0.1,
  });

  return response.choices[0].message.content || "";
}

// 提取重要信息实体
export async function extractEntities(messages: Message[]): Promise<{
  userInfo: Record<string, string>;
  decisions: string[];
  preferences: string[];
  constraints: string[];
}> {
  const text = messages.map((m) => `${m.role}: ${m.content}`).join("\n");

  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{
      role: "user",
      content: `从对话中提取关键实体，JSON 格式：
{
  "userInfo": {"name": "...", "role": "..."},
  "decisions": ["决策1"],
  "preferences": ["偏好1"],
  "constraints": ["约束1"]
}
只提取明确提到的内容。

对话：
${text}`,
    }],
    response_format: { type: "json_object" },
    temperature: 0,
  });

  return JSON.parse(response.choices[0].message.content!);
}

完整 ContextManager 类

// src/context/context_manager.ts
import { countMessagesTokens } from "./token_counter";
import { summarizeMessages, extractEntities, type Message } from "./summarizer";

interface LongTermMemory {
  userInfo: Record<string, string>;
  decisions: string[];
  preferences: string[];
  constraints: string[];
  summaries: { timestamp: number; content: string }[];
}

export class ContextManager {
  private workingMemory: Message[] = [];
  private longTermMemory: LongTermMemory = {
    userInfo: {},
    decisions: [],
    preferences: [],
    constraints: [],
    summaries: [],
  };

  private readonly maxWorkingTokens: number;

  constructor(maxWorkingTokens: number = 30000) {
    this.maxWorkingTokens = maxWorkingTokens;
  }

  async addMessage(message: Message): Promise<void> {
    this.workingMemory.push({ ...message, timestamp: Date.now() });

    const tokens = countMessagesTokens(this.workingMemory);
    if (tokens > this.maxWorkingTokens * 0.8) {
      await this.compress();
    }
  }

  buildContext(systemPrompt: string): Message[] {
    const messages: Message[] = [];

    // 1. System + Long-term Memory
    messages.push({ role: "system", content: this.buildSystemPrompt(systemPrompt) });

    // 2. 历史摘要
    if (this.longTermMemory.summaries.length > 0) {
      const summaryText = this.longTermMemory.summaries
        .map((s) => s.content)
        .join("\n\n---\n\n");
      messages.push({ role: "system", content: `[历史对话摘要]\n${summaryText}` });
    }

    // 3. Working Memory（近期完整对话）
    messages.push(...this.workingMemory.filter((m) => m.role !== "system"));

    return messages;
  }

  private buildSystemPrompt(base: string): string {
    const parts = [base];
    const { userInfo, preferences, constraints, decisions } = this.longTermMemory;

    if (Object.keys(userInfo).length > 0) {
      parts.push(`\n用户信息：${Object.entries(userInfo).map(([k, v]) => `${k}: ${v}`).join(", ")}`);
    }
    if (preferences.length > 0) parts.push(`\n用户偏好：${preferences.join("；")}`);
    if (constraints.length > 0) parts.push(`\n重要约束：${constraints.join("；")}`);
    if (decisions.length > 0) parts.push(`\n已确认决策：${decisions.join("；")}`);

    return parts.join("\n");
  }

  private async compress(): Promise<void> {
    const msgs = this.workingMemory.filter((m) => m.role !== "system");
    const keepCount = Math.floor(msgs.length * 0.4);
    const toCompress = msgs.slice(0, msgs.length - keepCount);

    console.log(`[ContextManager] 压缩 ${toCompress.length} 条消息，保留 ${keepCount} 条`);

    const [summary, entities] = await Promise.all([
      summarizeMessages(toCompress),
      extractEntities(toCompress),
    ]);

    this.longTermMemory.summaries.push({ timestamp: Date.now(), content: summary });
    Object.assign(this.longTermMemory.userInfo, entities.userInfo);
    this.longTermMemory.decisions.push(...entities.decisions);
    this.longTermMemory.preferences.push(...entities.preferences);
    this.longTermMemory.constraints.push(...entities.constraints);

    this.workingMemory = [
      ...this.workingMemory.filter((m) => m.role === "system"),
      ...msgs.slice(-keepCount),
    ];

    console.log(`[ContextManager] 压缩后 tokens: ${countMessagesTokens(this.workingMemory)}`);
  }

  getStats() {
    return {
      workingMemory: this.workingMemory.length,
      tokens: countMessagesTokens(this.workingMemory),
      summaries: this.longTermMemory.summaries.length,
      entities: Object.keys(this.longTermMemory.userInfo).length,
    };
  }

  serialize(): string {
    return JSON.stringify({ workingMemory: this.workingMemory, longTermMemory: this.longTermMemory });
  }

  static deserialize(data: string): ContextManager {
    const parsed = JSON.parse(data);
    const manager = new ContextManager();
    (manager as any).workingMemory = parsed.workingMemory;
    (manager as any).longTermMemory = parsed.longTermMemory;
    return manager;
  }
}

使用示例

// src/context/demo.ts
import OpenAI from "openai";
import { ContextManager } from "./context_manager";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
const ctx = new ContextManager(20000);

async function chat(userInput: string): Promise<string> {
  await ctx.addMessage({ role: "user", content: userInput });

  const messages = ctx.buildContext("你是一个专业的 AI 助手，善于记住用户的偏好和需求。");

  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: messages as any,
  });

  const reply = response.choices[0].message.content || "";
  await ctx.addMessage({ role: "assistant", content: reply });

  console.log("[Context]", ctx.getStats());
  return reply;
}

// 模拟长时间对话
async function runDemo() {
  await chat("我叫李明，是一家初创公司的 CTO，公司有 15 名工程师");
  await chat("我们正在构建 AI 客服系统，预算 100 万人民币");
  await chat("需要支持中英文双语");
  // ... 30 轮后
  const result = await chat("请根据我的情况制定技术选型方案");
  console.log(result);
}

runDemo().catch(console.error);
