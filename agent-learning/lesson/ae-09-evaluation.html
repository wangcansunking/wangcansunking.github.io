第9课：Agent 评估 — 如何量化 Agent 质量

介绍

Agent 和 RAG 系统很难用传统单元测试评估——输出是自然语言，没有唯一正确答案。本课实现三种评估范式：LLM-as-Judge（用 LLM 打分替代人工标注）、基于规则的自动评估（检查工具调用、格式、内容）、回归测试套件（防止迭代中质量下降）。

遇到什么问题

传统测试方法无法评估 Agent 输出质量：

flowchart TD
    A[Agent 输出] --> B{评估方式？}
    B -->|精确匹配| C[assert output === expected\n❌ 自然语言有多种正确表达]
    B -->|关键词匹配| D[output.includes keyword\n❌ 无法判断逻辑正确性]
    B -->|LLM-as-Judge| E[GPT-4o 打分 1-10\n✅ 接近人类判断]
    B -->|规则检查| F[格式/工具调用/长度\n✅ 快速客观]
    B -->|回归测试| G[对比历史基准分数\n✅ 防止质量退化]
    E --> H[综合评估报告]
    F --> H
    G --> H

有哪些方案

方案 A：LLM-as-Judge（LLM 评审）

const score = await judge.evaluate({
  question: "解释 RAG 的工作原理",
  answer: agentOutput,
  criteria: "accuracy"
});
// → { score: 8.5, reasoning: "回答准确，覆盖了关键步骤..." }

✅ 最接近人类判断，适合开放性任务 ❌ 需要 API 调用，有一定随机性

方案 B：规则检查器

const check = evaluator.checkFormat(output, {
  minLength: 100,
  maxLength: 2000,
  requiredSections: ["问题", "解决方案"],
  forbiddenPhrases: ["我不知道", "无法确定"],
});

✅ 快速、零成本，适合格式和行为约束 ❌ 无法评估语义质量

方案 C：综合评估套件（推荐）

结合 LLM-as-Judge + 规则检查 + 回归测试基准，形成完整的持续评估体系。

✅ 多维度覆盖，结果更可靠 ✅ 可集成 CI/CD，每次发布前自动运行

怎么选择

选择方案 C：规则检查做快速过滤，LLM-as-Judge 做质量评分，回归测试防止退化。三层防护确保 Agent 质量持续改善而不是退步。

怎么执行

LLM-as-Judge

// src/evaluation/llm_judge.ts
import OpenAI from "openai";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export interface JudgeResult {
  score: number;       // 0-10
  passed: boolean;     // 是否达标（默认 7 分以上）
  reasoning: string;   // 评分理由
  dimensions: Record<string, number>;  // 各维度分数
}

export interface JudgeCriteria {
  name: string;
  description: string;
  weight: number;      // 权重 0-1，所有权重之和应为 1
}

const DEFAULT_CRITERIA: JudgeCriteria[] = [
  { name: "accuracy", description: "事实准确性：回答是否包含正确的信息", weight: 0.4 },
  { name: "completeness", description: "完整性：是否覆盖了问题的所有重要方面", weight: 0.3 },
  { name: "clarity", description: "清晰度：表达是否清晰易懂", weight: 0.2 },
  { name: "relevance", description: "相关性：是否直接回应了问题，没有跑题", weight: 0.1 },
];

export async function judgeOutput(
  question: string,
  answer: string,
  groundTruth?: string,
  criteria: JudgeCriteria[] = DEFAULT_CRITERIA,
  threshold: number = 7
): Promise<JudgeResult> {
  const criteriaList = criteria
    .map((c) => `- ${c.name}（权重${c.weight}）：${c.description}`)
    .join("\n");

  const referenceSection = groundTruth
    ? `\n参考答案（仅供参考）：${groundTruth}`
    : "";

  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      {
        role: "system",
        content: "你是一个严格客观的质量评审专家。请根据评分维度对 AI 回答评分。",
      },
      {
        role: "user",
        content: `评估以下 AI 回答的质量：

问题：${question}${referenceSection}

AI 回答：
${answer}

评分维度：
${criteriaList}

请以 JSON 格式输出：
{
  "dimensions": {"accuracy": 8, "completeness": 7, ...},
  "overall_score": 7.8,
  "reasoning": "评分理由（50字以内）"
}`,
      },
    ],
    response_format: { type: "json_object" },
    temperature: 0.1,
  });

  const parsed = JSON.parse(response.choices[0].message.content!);
  const score = parsed.overall_score || parsed.score || 0;

  return {
    score,
    passed: score >= threshold,
    reasoning: parsed.reasoning || "",
    dimensions: parsed.dimensions || {},
  };
}

// 批量评估多个测试用例
export async function batchJudge(
  testCases: Array<{ question: string; answer: string; groundTruth?: string }>,
  criteria?: JudgeCriteria[]
): Promise<{
  results: Array<JudgeResult & { question: string }>;
  summary: { avgScore: number; passRate: number };
}> {
  const results = await Promise.all(
    testCases.map(async (tc) => {
      const result = await judgeOutput(tc.question, tc.answer, tc.groundTruth, criteria);
      return { ...result, question: tc.question };
    })
  );

  const avgScore = results.reduce((sum, r) => sum + r.score, 0) / results.length;
  const passRate = results.filter((r) => r.passed).length / results.length;

  return { results, summary: { avgScore, passRate } };
}

规则检查器

// src/evaluation/rule_checker.ts

export interface RuleCheckResult {
  passed: boolean;
  violations: string[];
  score: number;  // 通过的规则比例
}

export interface CheckRules {
  minLength?: number;
  maxLength?: number;
  requiredKeywords?: string[];         // 必须包含的关键词
  forbiddenPhrases?: string[];         // 禁止出现的短语
  requiredSections?: string[];         // 必须包含的章节标题
  codeBlockRequired?: boolean;         // 是否必须包含代码块
  maxBulletPoints?: number;            // 最大条目数
  languageConsistency?: "zh" | "en";   // 语言一致性
}

export function checkRules(output: string, rules: CheckRules): RuleCheckResult {
  const violations: string[] = [];
  let totalChecks = 0;
  let passedChecks = 0;

  function check(condition: boolean, message: string): void {
    totalChecks++;
    if (condition) {
      passedChecks++;
    } else {
      violations.push(message);
    }
  }

  if (rules.minLength !== undefined) {
    check(
      output.length >= rules.minLength,
      `输出过短：${output.length} < ${rules.minLength} 字符`
    );
  }

  if (rules.maxLength !== undefined) {
    check(
      output.length <= rules.maxLength,
      `输出过长：${output.length} > ${rules.maxLength} 字符`
    );
  }

  if (rules.requiredKeywords) {
    for (const keyword of rules.requiredKeywords) {
      check(
        output.toLowerCase().includes(keyword.toLowerCase()),
        `缺少必要关键词：${keyword}`
      );
    }
  }

  if (rules.forbiddenPhrases) {
    for (const phrase of rules.forbiddenPhrases) {
      check(
        !output.toLowerCase().includes(phrase.toLowerCase()),
        `出现禁止短语：${phrase}`
      );
    }
  }

  if (rules.requiredSections) {
    for (const section of rules.requiredSections) {
      check(
        output.includes(section),
        `缺少必要章节：${section}`
      );
    }
  }

  if (rules.codeBlockRequired) {
    check(
      output.includes("```"),
      "缺少代码块（要求包含 ``` 代码块）"
    );
  }

  if (rules.languageConsistency === "zh") {
    const chineseChars = (output.match(/[\u4e00-\u9fff]/g) || []).length;
    const totalChars = output.replace(/\s/g, "").length;
    check(
      totalChars === 0 || chineseChars / totalChars > 0.3,
      "语言不一致：期望中文内容"
    );
  }

  const score = totalChecks > 0 ? passedChecks / totalChecks : 1;

  return {
    passed: violations.length === 0,
    violations,
    score,
  };
}

回归测试套件

// src/evaluation/regression.ts
import * as fs from "fs";
import { judgeOutput } from "./llm_judge";

export interface TestCase {
  id: string;
  question: string;
  groundTruth?: string;
  rules?: import("./rule_checker").CheckRules;
  minScore?: number;  // 该测试用例的最低分要求
}

export interface BenchmarkResult {
  testCaseId: string;
  question: string;
  answer: string;
  score: number;
  passed: boolean;
  timestamp: number;
}

export interface RegressionReport {
  runId: string;
  timestamp: number;
  totalTests: number;
  passedTests: number;
  avgScore: number;
  passRate: number;
  results: BenchmarkResult[];
  regressions: string[];   // 与基准相比下降的测试用例 ID
}

export class RegressionSuite {
  private testCases: TestCase[];
  private baselinePath: string;

  constructor(testCases: TestCase[], baselinePath: string = "./baseline.json") {
    this.testCases = testCases;
    this.baselinePath = baselinePath;
  }

  async run(
    agentFn: (question: string) => Promise<string>
  ): Promise<RegressionReport> {
    const results: BenchmarkResult[] = [];

    for (const tc of this.testCases) {
      console.log(`[Regression] 测试 ${tc.id}: ${tc.question.slice(0, 50)}...`);

      const answer = await agentFn(tc.question);
      const judgeResult = await judgeOutput(
        tc.question,
        answer,
        tc.groundTruth,
        undefined,
        tc.minScore || 7
      );

      results.push({
        testCaseId: tc.id,
        question: tc.question,
        answer,
        score: judgeResult.score,
        passed: judgeResult.passed,
        timestamp: Date.now(),
      });

      console.log(
        `  分数: ${judgeResult.score.toFixed(1)}/10 ${judgeResult.passed ? "✓" : "✗"}`
      );
    }

    // 与基准对比，检测回归
    const baseline = this.loadBaseline();
    const regressions = this.detectRegressions(results, baseline);

    const report: RegressionReport = {
      runId: `run_${Date.now()}`,
      timestamp: Date.now(),
      totalTests: results.length,
      passedTests: results.filter((r) => r.passed).length,
      avgScore: results.reduce((sum, r) => sum + r.score, 0) / results.length,
      passRate: results.filter((r) => r.passed).length / results.length,
      results,
      regressions,
    };

    // 如果整体分数提升，更新基准
    if (!baseline || report.avgScore > baseline.avgScore) {
      this.saveBaseline(report);
      console.log("[Regression] 基准已更新");
    }

    return report;
  }

  private loadBaseline(): RegressionReport | null {
    if (!fs.existsSync(this.baselinePath)) return null;
    return JSON.parse(fs.readFileSync(this.baselinePath, "utf-8"));
  }

  private saveBaseline(report: RegressionReport): void {
    fs.writeFileSync(this.baselinePath, JSON.stringify(report, null, 2));
  }

  private detectRegressions(
    current: BenchmarkResult[],
    baseline: RegressionReport | null
  ): string[] {
    if (!baseline) return [];

    const baselineMap = new Map(
      baseline.results.map((r) => [r.testCaseId, r.score])
    );

    return current
      .filter((r) => {
        const baselineScore = baselineMap.get(r.testCaseId);
        return baselineScore !== undefined && r.score < baselineScore - 1.0;
      })
      .map((r) => r.testCaseId);
  }
}

完整评估 Pipeline 演示

// src/evaluation/demo.ts
import OpenAI from "openai";
import { judgeOutput, batchJudge } from "./llm_judge";
import { checkRules } from "./rule_checker";
import { RegressionSuite } from "./regression";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// 被评估的 Agent（模拟）
async function myAgent(question: string): Promise<string> {
  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: "你是一个 AI 技术助手，用中文简洁回答问题。" },
      { role: "user", content: question },
    ],
    temperature: 0.7,
  });
  return response.choices[0].message.content || "";
}

async function runDemo() {
  // 1. 单次 LLM-as-Judge 评估
  console.log("=== 1. LLM-as-Judge 评估 ===");
  const answer = await myAgent("什么是 RAG？它解决了什么问题？");
  const judgeResult = await judgeOutput(
    "什么是 RAG？它解决了什么问题？",
    answer,
    "RAG（Retrieval-Augmented Generation）通过检索外部知识库来增强 LLM 的回答，解决 LLM 知识截止和幻觉问题"
  );
  console.log(`分数：${judgeResult.score}/10，通过：${judgeResult.passed}`);
  console.log(`理由：${judgeResult.reasoning}`);

  // 2. 规则检查
  console.log("\n=== 2. 规则检查 ===");
  const ruleResult = checkRules(answer, {
    minLength: 100,
    maxLength: 1000,
    requiredKeywords: ["检索", "生成", "知识库"],
    forbiddenPhrases: ["我不确定", "可能是"],
    languageConsistency: "zh",
  });
  console.log(`规则通过：${ruleResult.passed}，得分：${(ruleResult.score * 100).toFixed(0)}%`);
  if (ruleResult.violations.length > 0) {
    console.log("违规：", ruleResult.violations);
  }

  // 3. 批量评估
  console.log("\n=== 3. 批量评估 ===");
  const testCases = [
    { question: "解释 ReAct 模式", answer: await myAgent("解释 ReAct 模式") },
    { question: "什么是 Chain of Thought？", answer: await myAgent("什么是 Chain of Thought？") },
    { question: "Transformer 的核心机制是什么？", answer: await myAgent("Transformer 的核心机制是什么？") },
  ];
  const batchResult = await batchJudge(testCases);
  console.log(`平均分：${batchResult.summary.avgScore.toFixed(1)}/10`);
  console.log(`通过率：${(batchResult.summary.passRate * 100).toFixed(0)}%`);

  // 4. 回归测试
  console.log("\n=== 4. 回归测试 ===");
  const suite = new RegressionSuite(
    [
      { id: "rag-basics", question: "什么是 RAG？", minScore: 7 },
      { id: "react-pattern", question: "解释 ReAct 模式的 Thought-Action-Observation 循环", minScore: 7 },
      { id: "reflexion", question: "Reflexion 与普通重试有什么区别？", minScore: 6 },
    ],
    "./agent_baseline.json"
  );

  const report = await suite.run(myAgent);
  console.log(`\n回归测试报告：`);
  console.log(`  总测试数：${report.totalTests}`);
  console.log(`  通过数：${report.passedTests}`);
  console.log(`  平均分：${report.avgScore.toFixed(1)}`);
  console.log(`  通过率：${(report.passRate * 100).toFixed(0)}%`);
  if (report.regressions.length > 0) {
    console.log(`  ⚠️ 检测到回归：${report.regressions.join(", ")}`);
  } else {
    console.log(`  ✓ 无回归`);
  }
}

runDemo().catch(console.error);
