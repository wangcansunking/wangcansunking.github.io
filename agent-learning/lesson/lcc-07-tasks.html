第7课：Task System — 持久化依赖任务图

介绍

内存中的 Todo 列表在 Agent 重启或崩溃时消失。Task System 把任务持久化到 JSON 文件，支持任务间的依赖关系（DAG），实现跨会话连续工作。Agent 重启后可以继续未完成的任务，而不是从头开始。

源码参考：github.com/shareAI-lab/learn-claude-code — agents/s07_task_system.py

遇到什么问题

复杂工程任务存在天然依赖：

安装依赖
    ↓
初始化数据库 ← 必须在安装后
    ↓               ↓
实现 API     写数据迁移脚本
    ↓               ↓
         写集成测试 ← 依赖 API + 数据库
                ↓
          部署到 staging

如果 Agent 在"实现 API"步骤崩溃，重启后它怎么知道：

  1. 哪些任务已完成？
  2. 接下来该做什么？
  3. 哪些任务可以并行？

有哪些方案

方案 A：内存 Todo（s03 方案）

✅ 简单 ❌ 不持久化，重启即失

方案 B：纯文本任务文件

# tasks.txt
[done] 安装依赖
[wip]  初始化数据库
[ ]   实现 API

✅ 持久化 ❌ 无依赖关系，无法自动确定执行顺序

方案 C：JSON 依赖图（推荐）

{
  "tasks": [
    {"id": "t1", "title": "安装依赖", "status": "done", "deps": []},
    {"id": "t2", "title": "初始化数据库", "status": "pending", "deps": ["t1"]},
    {"id": "t3", "title": "实现 API", "status": "pending", "deps": ["t1"]}
  ]
}

✅ 持久化，Agent 重启可恢复 ✅ 依赖感知：自动找出"可执行"任务 ✅ 可并行：多个无依赖的 pending 任务同时运行

怎么选择

方案 C。JSON 依赖图是最小化的生产级任务系统。可以直接存文件，也可以迁移到 SQLite。

怎么执行

TaskSystem 实现

# agents/s07_task_system.py
import json
import time
from pathlib import Path
from dataclasses import dataclass, asdict, field
from typing import Literal


@dataclass
class Task:
    id: str
    title: str
    description: str = ""
    status: Literal["pending", "in_progress", "done", "failed"] = "pending"
    priority: Literal["high", "medium", "low"] = "medium"
    deps: list[str] = field(default_factory=list)   # 依赖的任务 ID 列表
    result: str = ""                                  # 完成时填写结果摘要
    created_at: float = field(default_factory=time.time)
    updated_at: float = field(default_factory=time.time)


class TaskSystem:
    def __init__(self, db_path: str = ".tasks/tasks.json"):
        self.db_path = Path(db_path)
        self.db_path.parent.mkdir(exist_ok=True)
        self.tasks: dict[str, Task] = {}
        self._load()

    def _load(self):
        if self.db_path.exists():
            data = json.loads(self.db_path.read_text())
            self.tasks = {t["id"]: Task(**t) for t in data.get("tasks", [])}

    def _save(self):
        data = {"tasks": [asdict(t) for t in self.tasks.values()]}
        self.db_path.write_text(json.dumps(data, indent=2, ensure_ascii=False))

    def create_task(self, title: str, description: str = "",
                    deps: list[str] = None, priority: str = "medium") -> Task:
        task_id = f"t{int(time.time() * 1000) % 100000}"
        task = Task(
            id=task_id, title=title, description=description,
            deps=deps or [], priority=priority,
        )
        self.tasks[task_id] = task
        self._save()
        return task

    def update_status(self, task_id: str, status: str, result: str = "") -> str:
        if task_id not in self.tasks:
            return f"Task {task_id} not found"
        task = self.tasks[task_id]
        task.status = status
        task.result = result
        task.updated_at = time.time()
        self._save()
        return f"Task {task_id} ({task.title[:40]}) → {status}"

    def get_ready_tasks(self) -> list[Task]:
        """返回所有依赖都已完成的 pending 任务"""
        done_ids = {t.id for t in self.tasks.values() if t.status == "done"}
        return [
            t for t in self.tasks.values()
            if t.status == "pending" and all(dep in done_ids for dep in t.deps)
        ]

    def get_status_summary(self) -> str:
        """可读的任务状态摘要，注入到系统 Prompt"""
        counts = {"pending": 0, "in_progress": 0, "done": 0, "failed": 0}
        lines = ["=== TASK SYSTEM ==="]
        for task in sorted(self.tasks.values(), key=lambda t: t.created_at):
            counts[task.status] = counts.get(task.status, 0) + 1
            icon = {"pending": "○", "in_progress": "◐", "done": "✓", "failed": "✗"}[task.status]
            deps_info = f" [deps: {','.join(task.deps)}]" if task.deps else ""
            lines.append(f"  {icon} [{task.id}] {task.title}{deps_info}")
            if task.result:
                lines.append(f"      → {task.result[:80]}")
        lines.append(f"\nReady to execute: {[t.id for t in self.get_ready_tasks()]}")
        return "\n".join(lines)

工具接口

task_system = TaskSystem()

TOOL_HANDLERS.update({
    "task_create": lambda **kw: str(asdict(task_system.create_task(**kw))),
    "task_update": lambda **kw: task_system.update_status(**kw),
    "task_list":   lambda **kw: task_system.get_status_summary(),
})

# 注册三个工具到 TOOLS 列表（task_create / task_update / task_list）

系统 Prompt 注入任务状态

def build_system_prompt() -> str:
    return f"""You are a persistent engineering agent at {WORKDIR}.

{task_system.get_status_summary()}

Instructions:
- Use task_create to break work into tracked tasks with dependencies
- Use task_update to mark progress as you work
- Always execute ready tasks (dependencies satisfied) next
- After restart, check task_list to resume from where you left off"""

跨会话恢复演示

# 第一次运行
s07 >> build a REST API for a todo app with SQLite backend

# Agent 创建任务图：
# ○ [t001] 安装依赖 (fastapi, sqlite3)
# ○ [t002] 初始化 SQLite 数据库 [deps: t001]
# ○ [t003] 实现 CRUD API [deps: t001, t002]
# ○ [t004] 写测试 [deps: t003]
# ○ [t005] 写 README [deps: t003]

# ... Agent 完成 t001, t002 后中断 ...

# 第二次运行（无需告诉 Agent 之前做了什么）
s07 >> continue

# Agent 读取任务状态，自动继续：
# ✓ [t001] 安装依赖 — done
# ✓ [t002] 初始化 SQLite 数据库 — done
# ◐ [t003] 实现 CRUD API — 开始执行...
