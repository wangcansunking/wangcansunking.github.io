第10课：Team Protocols — Agent 间通信标准

介绍

多 Agent 团队需要通信协议保证可靠性。本课定义消息类型规范、确认机制、超时重试策略，以及职责边界——防止 Agent 互相等待形成死锁，确保任务总能完成。

源码参考：github.com/shareAI-lab/learn-claude-code — agents/s10_team_protocols.py

遇到什么问题

无协议的 Agent 通信容易出现：

问题 1：死锁
  DevAgent   → SecurityAgent: "请审查代码"
  SecurityAgent → DevAgent:   "请先修复 lint 错误"
  DevAgent   等待 SecurityAgent，SecurityAgent 等待 DevAgent → 永久阻塞

问题 2：消息丢失
  发送消息后没有确认机制，无法知道对方是否收到

问题 3：职责越界
  两个 Agent 同时修改同一文件 → 冲突

问题 4：无超时保护
  等待慢速 Agent 响应，整个团队卡住

有哪些方案

方案 A：无协议（自由通信）

✅ 实现简单 ❌ 所有上述问题都会出现

方案 B：严格同步 RPC

每次调用等待响应，超时则失败。

✅ 简单明确 ❌ 降低并行性，高延迟

方案 C：异步消息 + 协议规范（推荐）

消息类型：TASK | RESULT | QUESTION | ANSWER | ACK | ERROR
超时重试：发送后 N 秒未收到 ACK 则重试
职责边界：每类操作指定负责 Agent

✅ 高并行，明确边界，可靠通信

怎么选择

方案 C。生产级多 Agent 系统必须有协议规范。本课实现轻量级协议层，不引入外部依赖。

消息类型枚举

# agents/s10_team_protocols.py
from enum import Enum


class MsgType(str, Enum):
    TASK      = "task"       # 分配任务
    RESULT    = "result"     # 任务完成结果
    QUESTION  = "question"   # 询问（需要 ANSWER）
    ANSWER    = "answer"     # 回复询问
    ACK       = "ack"        # 确认收到
    ERROR     = "error"      # 错误通知
    HEARTBEAT = "heartbeat"  # 存活检查


class Priority(int, Enum):
    LOW    = 0
    MEDIUM = 1
    HIGH   = 2
    URGENT = 3

协议消息格式

@dataclass
class ProtocolMessage:
    id: str                     # 唯一消息 ID
    sender: str
    recipient: str
    type: MsgType
    content: str
    priority: Priority = Priority.MEDIUM
    reply_to: str | None = None  # 回复哪条消息
    requires_ack: bool = False   # 是否需要确认
    timeout: float = 30.0        # 等待响应超时（秒）
    timestamp: float = field(default_factory=time.time)
    attempts: int = 0            # 重试次数

    def ack(self) -> "ProtocolMessage":
        """生成确认消息"""
        return ProtocolMessage(
            id=f"ack_{self.id}",
            sender=self.recipient,
            recipient=self.sender,
            type=MsgType.ACK,
            content=f"received:{self.id}",
            reply_to=self.id,
        )

协议层实现

class ProtocolBus:
    MAX_RETRIES = 3
    RETRY_DELAY = 5.0

    def __init__(self):
        self._queues: dict[str, queue.PriorityQueue] = {}
        self._pending_acks: dict[str, ProtocolMessage] = {}  # 等待确认的消息
        self._lock = threading.Lock()
        self._start_retry_daemon()

    def send(self, msg: ProtocolMessage):
        """发送消息，优先级队列排序"""
        with self._lock:
            q = self._queues.get(msg.recipient)
            if q:
                # 优先级越高越先处理（PriorityQueue 升序，取反实现降序）
                q.put((-msg.priority, msg.timestamp, msg))
            if msg.requires_ack:
                self._pending_acks[msg.id] = msg

    def receive(self, agent: str, timeout: float = 5.0) -> ProtocolMessage | None:
        q = self._queues.get(agent)
        if not q:
            return None
        try:
            _, _, msg = q.get(timeout=timeout)
            # 自动发送 ACK
            if msg.requires_ack:
                self.send(msg.ack())
            return msg
        except queue.Empty:
            return None

    def _start_retry_daemon(self):
        """后台线程：重试未确认的消息"""
        def retry_loop():
            while True:
                time.sleep(self.RETRY_DELAY)
                now = time.time()
                with self._lock:
                    to_retry = [
                        m for m in self._pending_acks.values()
                        if now - m.timestamp > m.timeout
                        and m.attempts < self.MAX_RETRIES
                    ]
                for msg in to_retry:
                    msg.attempts += 1
                    msg.timestamp = time.time()
                    self.send(msg)
        threading.Thread(target=retry_loop, daemon=True).start()

职责边界定义

# 明确每类操作由哪个 Agent 负责
RESPONSIBILITY_MAP = {
    "write_code":      "dev",        # 只有 dev 能写代码
    "security_review": "security",   # 只有 security 能做安全审查
    "run_profiler":    "perf",       # 只有 perf 能运行性能分析
    "write_docs":      "docs",       # 只有 docs 能写文档
    "deploy":          "ops",        # 只有 ops 能部署
}

def check_responsibility(action: str, agent: str) -> bool:
    """检查 Agent 是否有权执行该操作"""
    owner = RESPONSIBILITY_MAP.get(action)
    if owner and owner != agent:
        raise PermissionError(
            f"{agent} cannot {action} — that's {owner}'s responsibility"
        )
    return True

防死锁：超时 + 取消机制

class AgentWithProtocol(TeamAgent):
    def ask_teammate(self, recipient: str, question: str,
                     timeout: float = 30.0) -> str:
        """
        发送 QUESTION，等待 ANSWER，超时则返回默认值（不阻塞）。
        这是防死锁的关键：永远不无限等待。
        """
        msg = ProtocolMessage(
            id=f"q_{int(time.time()*1000)}",
            sender=self.name,
            recipient=recipient,
            type=MsgType.QUESTION,
            content=question,
            requires_ack=True,
            timeout=timeout,
        )
        self.bus.send(msg)

        # 等待 ANSWER（限时）
        deadline = time.time() + timeout
        while time.time() < deadline:
            incoming = self.bus.receive(self.name, timeout=1.0)
            if incoming and incoming.type == MsgType.ANSWER \
               and incoming.reply_to == msg.id:
                return incoming.content
        # 超时：返回默认，继续工作（不阻塞）
        return f"[timeout: {recipient} did not respond in {timeout}s, proceeding without answer]"
