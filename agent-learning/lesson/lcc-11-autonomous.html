第11课：Autonomous Agents — 自主任务认领

介绍

到目前为止，所有 Agent 都是被动响应指令。Autonomous Agent 主动轮询共享任务队列，认领适合自己能力的任务，自主执行，写回结果——无需人工介入，实现真正的 24/7 持续运行。

源码参考：github.com/shareAI-lab/learn-claude-code — agents/s11_autonomous_agents.py

遇到什么问题

人工调度的局限：

# 人工调度：每次都需要人介入
human → agent: "fix bug #123"     # 人工分配
human → agent: "add feature X"    # 人工分配
human → agent: "write tests for Y" # 人工分配

# 自主调度：Agent 自己认领
Agent 空闲时 → 查看任务队列 → 认领匹配任务 → 执行 → 报告 → 继续认领
            ↑_______________________________________|

有哪些方案

方案 A：固定任务分配

协调者明确告诉每个 Agent 做什么。

✅ 可预测 ❌ 需要人工协调，Agent 空闲时浪费资源

方案 B：随机任务分配

从队列随机取任务，不考虑能力匹配。

✅ 无需协调 ❌ 能力不匹配，效率低，失败率高

方案 C：能力匹配 + 乐观锁认领（推荐）

# Agent 申报自己的能力
agent.capabilities = ["python", "testing", "security"]

# 任务标记所需技能
task.required_skills = ["python", "testing"]

# 原子性认领（乐观锁防止竞争）
if task.status == "pending" and skills_match:
    task.claim(agent_id)  # CAS 操作

✅ 能力匹配提高成功率 ✅ 乐观锁防止多 Agent 竞争同一任务

怎么选择

方案 C。能力匹配 + 乐观锁是生产级自治系统的标准模式。任务队列持久化到 JSON，Agent 重启后可继续认领。

怎么执行

任务队列（持久化）

# agents/s11_autonomous_agents.py
import json
import uuid
import threading
from pathlib import Path
from dataclasses import dataclass, asdict, field
from typing import Optional


@dataclass
class AutonomousTask:
    id: str
    title: str
    description: str
    required_skills: list[str]    # 执行任务所需技能
    priority: int = 1             # 1-5，越高越优先
    status: str = "pending"       # pending | claimed | done | failed
    claimed_by: Optional[str] = None
    claimed_at: Optional[float] = None
    result: str = ""
    created_at: float = field(default_factory=time.time)


class AutonomousTaskQueue:
    def __init__(self, path: str = ".tasks/autonomous_queue.json"):
        self.path = Path(path)
        self.path.parent.mkdir(exist_ok=True)
        self._lock = threading.Lock()
        self._load()

    def _load(self):
        if self.path.exists():
            data = json.loads(self.path.read_text())
            self.tasks = {t["id"]: AutonomousTask(**t) for t in data}
        else:
            self.tasks = {}

    def _save(self):
        data = list(asdict(t) for t in self.tasks.values())
        self.path.write_text(json.dumps(data, indent=2, ensure_ascii=False))

    def add_task(self, title: str, description: str,
                 required_skills: list[str], priority: int = 1) -> str:
        task = AutonomousTask(
            id=str(uuid.uuid4())[:8],
            title=title, description=description,
            required_skills=required_skills, priority=priority,
        )
        with self._lock:
            self.tasks[task.id] = task
            self._save()
        return task.id

    def claim_task(self, agent_id: str, capabilities: list[str]) -> Optional[AutonomousTask]:
        """
        乐观锁认领：找到最高优先级的可认领任务。
        返回认领成功的任务，或 None（无可用任务）。
        """
        with self._lock:
            candidates = [
                t for t in self.tasks.values()
                if t.status == "pending"
                and all(s in capabilities for s in t.required_skills)
            ]
            if not candidates:
                return None

            # 按优先级降序取第一个
            task = max(candidates, key=lambda t: (t.priority, -t.created_at))
            task.status = "claimed"
            task.claimed_by = agent_id
            task.claimed_at = time.time()
            self._save()
            return task

    def complete_task(self, task_id: str, result: str):
        with self._lock:
            if task_id in self.tasks:
                self.tasks[task_id].status = "done"
                self.tasks[task_id].result = result
                self._save()

    def fail_task(self, task_id: str, reason: str):
        with self._lock:
            if task_id in self.tasks:
                t = self.tasks[task_id]
                t.status = "pending"  # 放回队列，可被其他 Agent 认领
                t.claimed_by = None
                t.result = f"failed: {reason}"
                self._save()

Autonomous Agent 主循环

class AutonomousAgent(threading.Thread):
    IDLE_POLL_INTERVAL = 10.0   # 空闲时每10秒轮询一次
    CLAIM_TIMEOUT = 300.0       # 5分钟未完成则释放任务

    def __init__(self, agent_id: str, capabilities: list[str],
                 queue: AutonomousTaskQueue):
        super().__init__(daemon=True)
        self.agent_id = agent_id
        self.capabilities = capabilities
        self.queue = queue
        self.current_task: Optional[AutonomousTask] = None

    def run(self):
        print(f"[{self.agent_id}] Started. Capabilities: {self.capabilities}")
        while True:
            task = self.queue.claim_task(self.agent_id, self.capabilities)
            if task:
                self._execute_task(task)
            else:
                # 无任务可认领，空闲等待
                print(f"[{self.agent_id}] Idle, polling in {self.IDLE_POLL_INTERVAL}s")
                time.sleep(self.IDLE_POLL_INTERVAL)

    def _execute_task(self, task: AutonomousTask):
        print(f"[{self.agent_id}] Claimed: [{task.id}] {task.title}")
        self.current_task = task
        try:
            result = self._run_agent(
                system=f"""You are an autonomous agent with skills: {self.capabilities}.
Complete this task independently. Report results when done.""",
                user_message=f"Task: {task.title}\n\n{task.description}",
            )
            self.queue.complete_task(task.id, result)
            print(f"[{self.agent_id}] Completed: [{task.id}]")
        except Exception as e:
            self.queue.fail_task(task.id, str(e))
            print(f"[{self.agent_id}] Failed: [{task.id}] {e}")
        finally:
            self.current_task = None

启动自主 Agent 集群

def start_autonomous_cluster():
    queue = AutonomousTaskQueue()

    # 添加初始任务
    queue.add_task(
        "Fix login bug", "Users report 403 on valid credentials",
        required_skills=["python", "auth"], priority=5,
    )
    queue.add_task(
        "Write API docs", "Document the /users endpoint",
        required_skills=["docs", "markdown"], priority=2,
    )
    queue.add_task(
        "Security audit", "Review JWT implementation",
        required_skills=["security", "python"], priority=4,
    )

    # 启动专业化 Agent（各自认领匹配任务）
    agents = [
        AutonomousAgent("dev-1",      ["python", "auth", "testing"], queue),
        AutonomousAgent("dev-2",      ["python", "api", "testing"],  queue),
        AutonomousAgent("security-1", ["security", "python"],         queue),
        AutonomousAgent("docs-1",     ["docs", "markdown"],           queue),
    ]

    for agent in agents:
        agent.start()

    # 主线程：动态添加新任务
    while True:
        new_task = input("Add task (or 'status'): ")
        if new_task == "status":
            for t in queue.tasks.values():
                print(f"  [{t.status}] {t.title} → {t.claimed_by or 'unclaimed'}")
