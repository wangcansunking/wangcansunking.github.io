第5课：MVP v3 — SubAgent 并行执行

介绍

v2 是单线程的：AI 一件事做完才做下一件。遇到"帮我检查所有 10 个模块的代码质量"这类任务，就会很慢。

本课引入 SubAgent 模式：主 Agent 负责分解和协调，多个子 Agent 并行完成子任务，最后汇总结果。

遇到什么问题

用户说："帮我审查 src/ 下所有模块，找出潜在的 Bug 和性能问题。"

单线程做法：

gantt
    title 串行处理：总时间 = N × 单个时间
    dateFormat X
    axisFormat %ss

    section 顺序执行
    审查 auth.ts   :a1, 0, 30
    审查 api.ts    :a2, after a1, 30
    审查 db.ts     :a3, after a2, 30
    审查 utils.ts  :a4, after a3, 30
    审查 cache.ts  :a5, after a4, 30

如果能并行：

gantt
    title 并行处理：总时间 ≈ 最慢那个的时间
    dateFormat X
    axisFormat %ss

    section 并行执行
    审查 auth.ts   :a1, 0, 30
    审查 api.ts    :a2, 0, 28
    审查 db.ts     :a3, 0, 32
    审查 utils.ts  :a4, 0, 25
    审查 cache.ts  :a5, 0, 29

问题：单个 Agent 无法并行执行，因为它是顺序处理工具调用的。

有哪些方案

方案 A：让主 Agent 用 bash 的后台进程

node review.js auth.ts &
node review.js api.ts &
wait

❌ 输出混乱难以收集 ❌ 错误处理复杂

方案 B：在代码层用 Promise.all 并行

// 主进程直接并行调用多个 Agent 实例
const results = await Promise.all(
  files.map(f => runSubAgent(f))
)

✅ 简单可靠 ✅ 输出结构化，易于收集

方案 C：消息队列 + Worker Pool

主 Agent → 任务队列 → N个 Worker 并行消费 → 结果队列

✅ 适合超大规模任务 ❌ 实现复杂，过度设计

怎么选择

用方案 B：在 Node.js 层用 Promise.all 并行运行多个 Agent 实例。

实现思路：

  1. 主 Agent 分析任务，输出 JSON 格式的子任务列表
  2. 主进程解析 JSON，用 Promise.all 并行启动多个子 Agent
  3. 每个子 Agent 独立运行，完成后返回结果
  4. 主进程收集所有结果，再让主 Agent 汇总

怎么执行

TodoManager：任务分解系统

主 Agent 通过 todo_write 工具输出结构化任务列表：

const TODO_TOOL: Anthropic.Tool = {
  name: "todo_write",
  description: `将大任务分解为子任务列表。
  每个子任务将被并行执行。
  只在需要并行处理多个独立任务时使用。`,
  input_schema: {
    type: "object",
    properties: {
      tasks: {
        type: "array",
        items: {
          type: "object",
          properties: {
            id:          { type: "string" },
            description: { type: "string", description: "子任务的完整描述" },
            files:       { type: "array", items: { type: "string" }, description: "相关文件列表" },
          },
          required: ["id", "description"],
        },
      },
    },
    required: ["tasks"],
  },
};

SubAgent 执行器

interface SubTask {
  id: string;
  description: string;
  files?: string[];
}

async function runSubAgent(task: SubTask): Promise<string> {
  console.log(`  [子Agent-${task.id}] 开始: ${task.description}`);

  const messages: Anthropic.MessageParam[] = [
    {
      role: "user",
      content: task.files?.length
        ? `${task.description}\n\n相关文件：\n${task.files.join("\n")}`
        : task.description,
    },
  ];

  // 子 Agent 有独立的 messages 历史，互不影响
  while (true) {
    const response = await client.messages.create({
      model: "claude-opus-4-6",
      max_tokens: 4096,
      tools: [bashTool, readTool, editTool, globTool],  // 子 Agent 工具集
      messages,
      system: "你是专注的代码分析助手，完成分配给你的单一任务后立即返回结果。",
    });

    messages.push({ role: "assistant", content: response.content });

    if (response.stop_reason === "end_turn") {
      const text = response.content
        .filter((b): b is Anthropic.TextBlock => b.type === "text")
        .map(b => b.text)
        .join("");
      console.log(`  [子Agent-${task.id}] 完成`);
      return `## 子任务 ${task.id}: ${task.description}\n\n${text}`;
    }

    // 执行工具（子 Agent 有独立的工具执行，不影响主 Agent）
    const toolResults: Anthropic.ToolResultBlockParam[] = [];
    for (const block of response.content) {
      if (block.type === "tool_use") {
        const result = await executeTool(block.name, block.input as any);
        toolResults.push({ type: "tool_result", tool_use_id: block.id, content: result });
      }
    }
    messages.push({ role: "user", content: toolResults });
  }
}

主 Agent Loop（含 SubAgent 触发）

async function runMainAgent(userMessage: string): Promise<void> {
  const messages: Anthropic.MessageParam[] = [
    { role: "user", content: userMessage },
  ];

  while (true) {
    const response = await client.messages.create({
      model: "claude-opus-4-6",
      max_tokens: 4096,
      tools: [...standardTools, TODO_TOOL],  // 标准工具 + 任务分解工具
      messages,
    });

    messages.push({ role: "assistant", content: response.content });

    if (response.stop_reason === "end_turn") break;

    const toolResults: Anthropic.ToolResultBlockParam[] = [];

    for (const block of response.content) {
      if (block.type !== "tool_use") continue;

      if (block.name === "todo_write") {
        // ✨ 核心：并行执行所有子任务
        const tasks = (block.input as any).tasks as SubTask[];
        console.log(`\n📋 分解为 ${tasks.length} 个子任务，并行执行中...\n`);

        const results = await Promise.all(tasks.map(runSubAgent));
        const summary = results.join("\n\n---\n\n");

        console.log(`\n✅ 所有子任务完成，汇总结果...\n`);

        toolResults.push({
          type: "tool_result",
          tool_use_id: block.id,
          content: summary,
        });
      } else {
        // 普通工具直接执行
        const result = await executeTool(block.name, block.input as any);
        toolResults.push({ type: "tool_result", tool_use_id: block.id, content: result });
      }
    }

    messages.push({ role: "user", content: toolResults });
  }
}
