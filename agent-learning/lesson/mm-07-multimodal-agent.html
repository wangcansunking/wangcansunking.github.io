第7课：多模态 Agent — 看图、听声、读文的综合 Agent

介绍

把视觉、音频、文档能力整合成一个 Agent，能处理混合输入（"帮我分析这张图和这份 PDF"）。多模态输入的路由、格式统一、上下文管理比单模态更复杂，需要系统性设计。本课实现输入类型检测（MIME type）、多模态 context 构建（文本+图像+文档混合）、工具选择策略和输出格式适配。

遇到什么问题

多模态输入的路由、格式统一、上下文管理比单模态更复杂：

  1. 输入类型检测：同一个 API 可能收到文本、图像、音频、PDF，需要先识别类型
  2. 格式统一：不同模态的处理结果格式不同（图像→Vision 输入，音频→转录文本，PDF→提取文本）
  3. Context 构建：把多种模态的内容整合成一个 LLM 可以理解的 context
  4. 模型选择：有图像时需要 Vision 模型，纯文本时用普通模型，混合时需要权衡

有哪些方案

方案 A：统一转文本（先全部转文本，再传给 LLM）

所有模态都先转换为文本描述，再统一传给 LLM。

✅ 简单，任何 LLM 都能处理 ❌ 图像转文字会损失信息（细节、颜色、空间关系等）

方案 B：原生多模态传递（图像直接传给 Vision 模型）

图像不转文字，直接作为 content parts 传给支持 Vision 的模型。

✅ 保留完整图像信息，理解更准确 ❌ 需要 Vision 模型，音频和文档仍需转换

方案 C：混合策略（推荐）

图像直接传 Vision 模型，音频先转录，文档先提取文本，混合构建 context。

✅ 最大化利用各模态的信息 ✅ 灵活，可针对不同任务选最优策略

怎么选择

选择方案 C（混合策略）。图像信息损失最敏感，应该直接传 Vision 模型；音频和文档必须转换，转换质量（Whisper STT、pdf-parse）已经足够好。

怎么执行

输入类型检测（MIME type）

// src/multimodal_agent/detector.ts
import * as fs from "fs";
import * as path from "path";

export type InputType = "text" | "image" | "audio" | "video" | "pdf" | "word" | "excel" | "unknown";

export interface DetectedInput {
  type: InputType;
  content: string;
  mimeType?: string;
  size?: number;
}

export function detectInputType(input: string): InputType {
  if (!fs.existsSync(input)) return "text"; // 非文件路径 = 纯文本

  const ext = path.extname(input).toLowerCase();
  const mimeMap: Record<string, InputType> = {
    ".jpg": "image", ".jpeg": "image", ".png": "image",
    ".gif": "image", ".webp": "image",
    ".mp3": "audio", ".wav": "audio", ".m4a": "audio",
    ".ogg": "audio", ".flac": "audio",
    ".mp4": "video", ".mov": "video", ".avi": "video",
    ".pdf": "pdf",
    ".docx": "word", ".doc": "word",
    ".xlsx": "excel", ".xls": "excel",
    ".txt": "text", ".md": "text",
  };
  return mimeMap[ext] || "unknown";
}

export function detectInputs(inputs: string[]): DetectedInput[] {
  return inputs.map((input) => ({
    type: detectInputType(input),
    content: input,
    size: fs.existsSync(input) ? fs.statSync(input).size : undefined,
  }));
}

多模态 Context 构建

// src/multimodal_agent/context_builder.ts
import * as fs from "fs";
import type { DetectedInput } from "./detector";

export interface MultimodalContext {
  textContent: string;
  hasImages: boolean;
  hasAudio: boolean;
  hasDocuments: boolean;
  imageContents: { base64: string; mimeType: string }[];
}

export async function buildContext(
  inputs: DetectedInput[],
  textQuery: string
): Promise<MultimodalContext> {
  const context: MultimodalContext = {
    textContent: textQuery,
    hasImages: false,
    hasAudio: false,
    hasDocuments: false,
    imageContents: [],
  };

  for (const input of inputs) {
    switch (input.type) {
      case "image": {
        const buffer = fs.readFileSync(input.content);
        const ext = input.content.split(".").pop()?.toLowerCase();
        const mimeType = ext === "png" ? "image/png" : ext === "webp" ? "image/webp" : "image/jpeg";
        context.hasImages = true;
        context.imageContents.push({ base64: buffer.toString("base64"), mimeType });
        break;
      }

      case "audio": {
        console.log(`[Multimodal] 转录音频: ${input.content}`);
        const { transcribe } = await import("../audio/whisper");
        const { text } = await transcribe(input.content);
        context.hasAudio = true;
        context.textContent += `\n\n[音频转录内容]\n${text}`;
        break;
      }

      case "pdf": {
        console.log(`[Multimodal] 解析 PDF: ${input.content}`);
        const { parsePDF } = await import("../documents/pdf_parser");
        const doc = await parsePDF(input.content);
        context.hasDocuments = true;
        context.textContent += `\n\n[PDF 文档内容]\n${doc.text}`;
        break;
      }

      case "word": {
        console.log(`[Multimodal] 解析 Word: ${input.content}`);
        const { parseWord } = await import("../documents/word_parser");
        const doc = await parseWord(input.content);
        context.hasDocuments = true;
        context.textContent += `\n\n[Word 文档内容]\n${doc.text}`;
        break;
      }

      case "excel": {
        console.log(`[Multimodal] 解析 Excel: ${input.content}`);
        const { parseExcel } = await import("../documents/excel_parser");
        const data = parseExcel(input.content);
        const tableText = Object.values(data.sheets)
          .map((s: any) => `## ${s.name}\n${s.markdown}`)
          .join("\n\n");
        context.hasDocuments = true;
        context.textContent += `\n\n[Excel 数据]\n${tableText}`;
        break;
      }
    }
  }

  return context;
}

多模态 Agent 主类

// src/multimodal_agent/agent.ts
import OpenAI from "openai";
import type { MultimodalContext } from "./context_builder";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export async function processMultimodalQuery(
  context: MultimodalContext,
  systemPrompt?: string
): Promise<string> {
  const system =
    systemPrompt ||
    "你是一个多模态 AI 助手，能分析图像、音频转录和文档内容。请综合所有输入材料，给出全面准确的回答。";

  if (context.hasImages) {
    // 有图像：构建 Vision 请求（图像直接传，不转文字）
    const userContent: OpenAI.ChatCompletionContentPart[] = [];

    for (const img of context.imageContents) {
      userContent.push({
        type: "image_url",
        image_url: { url: `data:${img.mimeType};base64,${img.base64}` },
      });
    }

    userContent.push({ type: "text", text: context.textContent });

    const response = await openai.chat.completions.create({
      model: "gpt-4o",
      messages: [
        { role: "system", content: system },
        { role: "user", content: userContent },
      ],
      max_tokens: 2000,
    });

    return response.choices[0].message.content || "";
  } else {
    // 纯文本：根据长度选择模型
    const model = context.textContent.length > 30000 ? "gpt-4o" : "gpt-4o-mini";

    const response = await openai.chat.completions.create({
      model,
      messages: [
        { role: "system", content: system },
        { role: "user", content: context.textContent },
      ],
      max_tokens: 2000,
    });

    return response.choices[0].message.content || "";
  }
}

完整使用示例

// src/multimodal_agent/demo.ts
import { detectInputs } from "./detector";
import { buildContext } from "./context_builder";
import { processMultimodalQuery } from "./agent";

export async function multimodalChat(
  query: string,
  attachments: string[] = []
): Promise<string> {
  console.log(`\n[Agent] 查询: "${query}"`);

  // 1. 检测输入类型
  const detectedInputs = detectInputs(attachments);
  for (const input of detectedInputs) {
    console.log(`  附件: ${input.type} - ${input.content}`);
  }

  // 2. 构建多模态 context
  const context = await buildContext(detectedInputs, query);
  console.log(
    `[Agent] Context 构建完成: ` +
    `图像=${context.imageContents.length}张, ` +
    `音频=${context.hasAudio}, 文档=${context.hasDocuments}`
  );

  // 3. LLM 处理
  return processMultimodalQuery(context);
}

async function runDemo() {
  // 场景1：图像分析
  const r1 = await multimodalChat("描述图片内容，并说明可能的应用场景", ["./chart.png"]);
  console.log("结果:", r1);

  // 场景2：图像 + 文档混合
  const r2 = await multimodalChat(
    "对比图片和文档中的数据，找出差异",
    ["./data_chart.png", "./data_report.pdf"]
  );
  console.log("结果:", r2);

  // 场景3：音频 + 文档
  const r3 = await multimodalChat(
    "根据会议录音和会议记录文档，总结关键决策",
    ["./meeting.mp3", "./meeting_notes.docx"]
  );
  console.log("结果:", r3);
}

runDemo().catch(console.error);
