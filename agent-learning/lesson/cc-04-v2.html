第4课：MVP v2 — 完整工具 + 权限 + CLAUDE.md

介绍

v1 能跑，但在真实项目里用它你会很快碰壁：AI 想读一个文件，却只能绕路用 cat；想改代码，却只能先读全文再整体覆写——容易把不该改的行也改掉。

本课升级三件事：完整工具集、权限保护、项目上下文（CLAUDE.md）。

遇到什么问题

v1 的三个痛点：

1. 只有 Bash，文件操作笨拙

# AI 想精确修改第 42 行，只能这么做：
# 先读整个文件
cat src/index.ts
# 再整体覆写（危险！容易丢失其他内容）
cat > src/index.ts << 'EOF'
... 整个文件内容 ...
EOF

2. 没有权限保护

# AI 一旦理解错需求，可能执行：
rm -rf dist/    # 删掉整个构建目录
git push --force # 强制推送覆盖远程

v1 对此毫无防御。

3. 没有项目上下文

AI 不知道：这个项目用什么框架？有哪些约定？哪些目录不能动？每次都要靠对话重新说明。

有哪些方案

方案 A：把所有防护逻辑写进 System Prompt

"你不能运行 rm -rf，你不能修改 .env，你应该用 Edit 而不是 Write..."

❌ Prompt 越来越长，AI 容易遗忘 ❌ 用户无法动态配置

方案 B：在工具执行层拦截（硬编码黑名单）

if (command.includes('rm -rf')) throw new Error('禁止操作')

✅ 可靠，AI 绕不过去 ❌ 黑名单难以穷举，灵活性差

方案 C：权限分级 + 用户确认（白名单 + 询问）

低风险操作  → 直接执行（读文件、跑测试）
中风险操作  → 询问用户是否确认（git push）
高危操作    → 直接拒绝（rm -rf /）

✅ 安全且灵活 ✅ 用户保持控制权

怎么选择

采用方案 C，同时加入：

  • 专用文件工具（Read/Write/Edit/Glob）替代纯 Bash 文件操作
  • CLAUDE.md 自动加载到 System Prompt

权限系统

import * as readline from "readline";

// 危险命令黑名单（直接拒绝）
const DENY_PATTERNS = [
  /rm\s+-rf\s+\//, /mkfs/, /dd\s+if=.*of=\/dev/,
];

// 需要确认的命令（询问用户）
const ASK_PATTERNS = [
  /git\s+push/, /rm\s+-rf/, /git\s+reset\s+--hard/,
  /sudo/, /chmod\s+777/,
];

async function checkPermission(command: string): Promise<boolean> {
  // 硬拒绝
  if (DENY_PATTERNS.some(p => p.test(command))) {
    console.log(`🚫 拒绝执行危险命令: ${command}`);
    return false;
  }

  // 询问用户
  if (ASK_PATTERNS.some(p => p.test(command))) {
    const rl = readline.createInterface({ input: process.stdin, output: process.stdout });
    const answer = await new Promise<string>(resolve =>
      rl.question(`⚠️  确认执行 "${command}"? (y/n): `, resolve)
    );
    rl.close();
    return answer.toLowerCase() === 'y';
  }

  return true; // 默认允许
}

工具执行（含权限检查）

async function executeTool(
  name: string,
  input: Record<string, any>
): Promise<string> {
  switch (name) {
    case "bash": {
      if (!(await checkPermission(input.command))) {
        return "操作已被用户取消。";
      }
      try {
        const { stdout, stderr } = await execAsync(input.command, {
          cwd: process.cwd(), timeout: 30_000,
        });
        return stdout || stderr || "(无输出)";
      } catch (err: any) {
        return `错误: ${err.message}`;
      }
    }

    case "read": {
      if (!existsSync(input.path)) return `文件不存在: ${input.path}`;
      const content = readFileSync(input.path, "utf-8");
      if (input.start_line || input.end_line) {
        const lines = content.split("\n");
        const start = (input.start_line ?? 1) - 1;
        const end   = input.end_line ?? lines.length;
        return lines.slice(start, end).join("\n");
      }
      return content;
    }

    case "write": {
      writeFileSync(input.path, input.content, "utf-8");
      return `已写入: ${input.path}`;
    }

    case "edit": {
      if (!existsSync(input.path)) return `文件不存在: ${input.path}`;
      const content = readFileSync(input.path, "utf-8");
      if (!content.includes(input.old_string)) {
        return `错误: 未找到要替换的文本。请检查 old_string 是否与文件内容完全匹配。`;
      }
      const updated = content.replace(input.old_string, input.new_string);
      writeFileSync(input.path, updated, "utf-8");
      return `已修改: ${input.path}`;
    }

    case "glob": {
      const files = await glob(input.pattern, { ignore: "node_modules/**" });
      return files.length ? files.join("\n") : "(无匹配文件)";
    }

    default:
      return `未知工具: ${name}`;
  }
}

CLAUDE.md 自动加载

function loadClaudeMd(): string {
  const paths = ["./CLAUDE.md", "../CLAUDE.md"];
  for (const p of paths) {
    if (existsSync(p)) {
      const content = readFileSync(p, "utf-8");
      console.log(`📋 已加载 CLAUDE.md (${content.length} 字符)`);
      return `\n\n## 项目规范（来自 CLAUDE.md）\n${content}`;
    }
  }
  return "";
}
