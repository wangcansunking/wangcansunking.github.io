第5课：Plan-and-Execute — 先规划后执行

介绍

Plan-and-Execute 是两阶段 Agent 模式：先用强模型做全局规划（生成步骤列表），再逐步执行（可以用弱模型）。比 ReAct 更适合长任务。ReAct 是"走一步看一步"，容易在长任务中偏离目标；Plan-Execute 先规划确保方向正确。

遇到什么问题

ReAct 是"走一步看一步"，长任务容易偏离目标；Plan-Execute 先规划确保方向正确：

flowchart TD
    A[用户任务] --> B[Planner\n强模型 GPT-4o/Claude]
    B --> C[结构化步骤列表\nStep 1-N]
    C --> D{步骤执行循环}
    D --> E[Executor\n可用弱模型]
    E --> F{执行结果}
    F -->|成功| G{还有步骤?}
    F -->|失败| H[Re-Planner\n修订计划]
    H --> D
    G -->|是| D
    G -->|否| I[汇总结果]
    I --> J[最终输出]

有哪些方案

方案 A：静态规划（一次性生成完整计划）

const plan = await planner.createPlan(task);
for (const step of plan.steps) {
  await executor.execute(step);
}

✅ 逻辑清晰，计划可审查 ❌ 无法适应执行过程中的变化

方案 B：动态规划（执行失败时重规划）

while (!done) {
  const nextStep = await planner.getNextStep(task, completedSteps, lastResult);
  const result = await executor.execute(nextStep);
  if (!result.success) {
    plan = await planner.replan(task, completedSteps, result.error);
  }
}

✅ 能适应执行中的意外情况 ❌ 重规划增加 LLM 调用成本

方案 C：并行执行（无依赖的步骤并行）

const parallelSteps = plan.steps.filter((s) => !s.dependsOn?.length);
await Promise.all(parallelSteps.map(executor.execute));

✅ 大幅加速独立任务的执行 ❌ 需要正确识别依赖关系

怎么选择

选择结合方案 A + B + C：静态规划为主，执行失败时动态重规划，独立步骤并行执行。这是生产级 Plan-Execute Agent 的完整实现。

怎么执行

Planner（结构化输出步骤列表）

// src/plan_execute/planner.ts
import OpenAI from "openai";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export interface Step {
  id: string;
  description: string;
  tool?: string;           // 需要使用的工具（可选）
  input?: string;          // 工具输入提示
  dependsOn?: string[];    // 依赖的步骤 ID（用于并行执行）
  expectedOutput?: string; // 预期输出描述
}

export interface Plan {
  objective: string;
  steps: Step[];
  totalEstimatedSteps: number;
}

export async function createPlan(task: string, context?: string): Promise<Plan> {
  const response = await openai.chat.completions.create({
    model: "gpt-4o", // Planner 用强模型
    messages: [
      {
        role: "system",
        content: `你是一个任务规划专家。将复杂任务分解为清晰的执行步骤。

输出 JSON 格式：
{
  "objective": "任务目标",
  "steps": [
    {
      "id": "step_1",
      "description": "具体步骤描述",
      "tool": "可选工具名称",
      "input": "工具输入提示",
      "dependsOn": [],
      "expectedOutput": "期望得到什么结果"
    }
  ],
  "totalEstimatedSteps": 步骤数
}

原则：
- 每个步骤要具体可执行
- 明确标注步骤间的依赖关系
- 无依赖的步骤可以并行执行
- 步骤数量控制在 3-10 个`,
      },
      {
        role: "user",
        content: `任务：${task}${context ? `\n背景：${context}` : ""}`,
      },
    ],
    response_format: { type: "json_object" },
    temperature: 0.3,
  });

  return JSON.parse(response.choices[0].message.content!) as Plan;
}

// 执行失败后重规划
export async function replan(
  originalTask: string,
  completedSteps: { step: Step; result: string }[],
  failedStep: Step,
  errorMessage: string
): Promise<Plan> {
  const completedSummary = completedSteps
    .map((s) => `- ${s.step.description}: ${s.result.slice(0, 100)}`)
    .join("\n");

  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      {
        role: "user",
        content: `原始任务：${originalTask}

已完成步骤：
${completedSummary}

失败步骤：${failedStep.description}
错误：${errorMessage}

请重新规划剩余的执行步骤（从失败点之后继续），使用同样的 JSON 格式。`,
      },
    ],
    response_format: { type: "json_object" },
    temperature: 0.3,
  });

  return JSON.parse(response.choices[0].message.content!) as Plan;
}

Executor（逐步执行，可并行）

// src/plan_execute/executor.ts
import OpenAI from "openai";
import type { Step } from "./planner";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export interface ExecutionResult {
  stepId: string;
  success: boolean;
  output: string;
  error?: string;
  duration: number;
}

// 执行单个步骤（可用弱模型，降低成本）
export async function executeStep(
  step: Step,
  previousResults: Map<string, string>,
  tools: Map<string, (input: string) => Promise<string>>
): Promise<ExecutionResult> {
  const start = Date.now();

  try {
    // 构建上下文：已完成步骤的结果
    const contextStr = Array.from(previousResults.entries())
      .map(([id, result]) => `[${id}] ${result.slice(0, 200)}`)
      .join("\n");

    let output: string;

    if (step.tool && tools.has(step.tool)) {
      // 有指定工具，调用工具
      const toolFn = tools.get(step.tool)!;
      const toolInput = step.input || step.description;
      output = await toolFn(toolInput);
    } else {
      // 用 LLM 执行（Executor 可以用 gpt-4o-mini 降低成本）
      const response = await openai.chat.completions.create({
        model: "gpt-4o-mini",
        messages: [
          {
            role: "system",
            content: "你是任务执行助手。基于前面步骤的结果，执行当前步骤。",
          },
          {
            role: "user",
            content: `${contextStr ? `前面步骤的结果：\n${contextStr}\n\n` : ""}当前步骤：${step.description}`,
          },
        ],
        temperature: 0.3,
      });
      output = response.choices[0].message.content || "";
    }

    return {
      stepId: step.id,
      success: true,
      output,
      duration: Date.now() - start,
    };
  } catch (error) {
    return {
      stepId: step.id,
      success: false,
      output: "",
      error: error instanceof Error ? error.message : String(error),
      duration: Date.now() - start,
    };
  }
}

// 并行执行无依赖的步骤
export async function executeParallel(
  steps: Step[],
  previousResults: Map<string, string>,
  tools: Map<string, (input: string) => Promise<string>>
): Promise<ExecutionResult[]> {
  return Promise.all(steps.map((step) => executeStep(step, previousResults, tools)));
}

进度追踪 + 完整 Pipeline

// src/plan_execute/pipeline.ts
import { createPlan, replan, type Plan, type Step } from "./planner";
import { executeStep, executeParallel } from "./executor";

export class PlanExecutePipeline {
  private tools: Map<string, (input: string) => Promise<string>>;
  private maxReplans: number;

  constructor(
    tools: Map<string, (input: string) => Promise<string>>,
    maxReplans: number = 2
  ) {
    this.tools = tools;
    this.maxReplans = maxReplans;
  }

  async run(task: string): Promise<{
    success: boolean;
    finalOutput: string;
    plan: Plan;
    results: { step: Step; result: string }[];
    totalDuration: number;
  }> {
    const start = Date.now();
    console.log(`[Plan-Execute] 任务：${task}\n`);

    // 1. 规划
    let plan = await createPlan(task);
    console.log(`[Plan] 生成了 ${plan.steps.length} 个步骤：`);
    plan.steps.forEach((s, i) => console.log(`  ${i + 1}. ${s.description}`));

    const completedResults: { step: Step; result: string }[] = [];
    const resultMap = new Map<string, string>();
    let replansUsed = 0;

    // 2. 执行步骤（支持并行）
    const remainingSteps = [...plan.steps];

    while (remainingSteps.length > 0) {
      // 找出所有依赖已完成的步骤（可以并行执行）
      const ready = remainingSteps.filter(
        (step) =>
          !step.dependsOn?.length ||
          step.dependsOn.every((dep) => resultMap.has(dep))
      );

      if (ready.length === 0) {
        console.error("[Plan-Execute] 步骤依赖死锁");
        break;
      }

      console.log(`\n[Execute] 并行执行 ${ready.length} 个步骤...`);

      const results = await executeParallel(ready, resultMap, this.tools);

      for (const result of results) {
        const step = ready.find((s) => s.id === result.stepId)!;

        if (!result.success && replansUsed < this.maxReplans) {
          // 执行失败：重规划
          console.log(`[Replan] 步骤 ${result.stepId} 失败: ${result.error}`);
          plan = await replan(task, completedResults, step, result.error!);
          replansUsed++;

          // 更新剩余步骤
          const completedIds = new Set(completedResults.map((r) => r.step.id));
          remainingSteps.length = 0;
          remainingSteps.push(
            ...plan.steps.filter((s) => !completedIds.has(s.id))
          );
          break;
        }

        console.log(`  ✓ ${step.description} (${result.duration}ms)`);
        resultMap.set(result.stepId, result.output);
        completedResults.push({ step, result: result.output });

        // 从剩余步骤中移除
        const idx = remainingSteps.findIndex((s) => s.id === result.stepId);
        if (idx !== -1) remainingSteps.splice(idx, 1);
      }
    }

    // 3. 汇总结果
    const finalOutput = completedResults.map((r) => r.result).join("\n\n");

    return {
      success: true,
      finalOutput,
      plan,
      results: completedResults,
      totalDuration: Date.now() - start,
    };
  }
}

完整演示

// src/plan_execute/demo.ts
import { PlanExecutePipeline } from "./pipeline";

async function runDemo() {
  // 定义可用工具
  const tools = new Map<string, (input: string) => Promise<string>>();

  tools.set("search", async (query) => {
    return `搜索"${query}"的结果：找到相关信息...`;
  });

  tools.set("write", async (instruction) => {
    const { default: OpenAI } = await import("openai");
    const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
    const response = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [{ role: "user", content: instruction }],
    });
    return response.choices[0].message.content || "";
  });

  const pipeline = new PlanExecutePipeline(tools);

  const result = await pipeline.run(
    "研究 2025 年 AI Agent 的最新发展，写一篇 500 字的技术简报"
  );

  console.log("\n=== 执行完成 ===");
  console.log(`总耗时: ${result.totalDuration}ms`);
  console.log(`执行步骤: ${result.results.length}`);
  console.log("\n最终输出:");
  console.log(result.finalOutput.slice(0, 500) + "...");
}

runDemo().catch(console.error);
